34 /// Perform the actions on the input file indicated by the command line flags
35 /// within the specified context.
36 ///
37 /// This typically parses the main source file, runs zero or more optimization
38 /// passes, then prints the output.
39 ///
40 static LogicalResult performActions(raw_ostream &os, bool verifyDiagnostics,
41  bool verifyPasses, SourceMgr &sourceMgr,
42  MLIRContext *context,
43  const PassPipelineCLParser &passPipeline) {
44  OwningModuleRef module(parseSourceFile(sourceMgr, context));
45  if (!module)
46  return failure();
47 
48  // Apply any pass manager command line options.
49  PassManager pm(context, verifyPasses);
50  applyPassManagerCLOptions(pm);
51 
52  // Build the provided pipeline.
53  if (failed(passPipeline.addToPipeline(pm)))
54  return failure();
55 
56  // Run the pipeline.
57  if (failed(pm.run(*module)))
58  return failure();
59 
60  // Print the output.
61  module->print(os);
62  return success();
63 }

65 /// Parses the memory buffer. If successfully, run a series of passes against
66 /// it and print the result.
67 static LogicalResult processBuffer(raw_ostream &os,
68  std::unique_ptr<MemoryBuffer> ownedBuffer,
69  bool verifyDiagnostics, bool verifyPasses,
70  const PassPipelineCLParser &passPipeline) {
71  // Tell sourceMgr about this buffer, which is what the parser will pick up.
72  SourceMgr sourceMgr;
73  sourceMgr.AddNewSourceBuffer(std::move(ownedBuffer), SMLoc());
74 
75  // Parse the input file.
76  MLIRContext context;
77 
78  // If we are in verify diagnostics mode then we have a lot of work to do,
79  // otherwise just perform the actions without worrying about it.
80  if (!verifyDiagnostics) {
81  SourceMgrDiagnosticHandler sourceMgrHandler(sourceMgr, &context);
82  return performActions(os, verifyDiagnostics, verifyPasses, sourceMgr,
83  &context, passPipeline);
84  }
85 
86  SourceMgrDiagnosticVerifierHandler sourceMgrHandler(sourceMgr, &context);
87 
88  // Do any processing requested by command line flags. We don't care whether
89  // these actions succeed or fail, we only care what diagnostics they produce
90  // and whether they match our expectations.
91  performActions(os, verifyDiagnostics, verifyPasses, sourceMgr, &context,
92  passPipeline);
93 
94  // Verify the diagnostic handler to make sure that each of the diagnostics
95  // matched.
96  return sourceMgrHandler.verify();
97 }
