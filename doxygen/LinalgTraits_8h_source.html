22 /// This class provides the API for ops that are known to have a specified
23 /// number of inputs, all passed as operands. Use as a trait as follows:
24 ///
25 /// class DotOp : public Op<DotOp, OpTrait::NInputs<2>::Impl> {
26 ///
27 template <unsigned N> class NInputs {
28 public:
29  template <typename ConcreteType>
30  class Impl : public OpTrait::TraitBase<ConcreteType, NInputs<N>::Impl> {
31  public:
32  static unsigned getNumInputs() { return N; }
33  };
34 };

50 /// This class provides the API for structured ops that are known to operate on
51 /// buffers or tensors. This trait must be used in conjunction with an op
52 /// definition or a trait that provides the methods `getNumInputs` and
53 /// `getNumOutputs`. Use as a trait as follows:
54 ///
55 /// class DotOp : public Op<DotOp, OpTrait::StructuredOpTraits> {
56 ///
57 template <typename ConcreteType>
58 class StructuredOpTraits
59  : public OpTrait::TraitBase<ConcreteType, StructuredOpTraits> {
60 private:
61  /// Return the number of inputs, irrespective of their buffer or tensor type.
62  /// For internal use only.
63  unsigned nInputs() {
64  return cast<ConcreteType>(this->getOperation()).getNumInputs();
65  }
66  /// Return the number of outputs, irrespective of their buffer or tensor type.
67  /// For internal use only.
68  unsigned nOutputs() {
69  return cast<ConcreteType>(this->getOperation()).getNumOutputs();
70  }
71 
72 public:
73  //==========================================================================//
74  // Loop types handling.
75  //==========================================================================//
76  unsigned getNumParallelLoops() {
77  return getNumIterators(
78  getParallelIteratorTypeName(),
79  cast<ConcreteType>(this->getOperation()).iterator_types());
80  }
81  unsigned getNumReductionLoops() {
82  return getNumIterators(
83  getReductionIteratorTypeName(),
84  cast<ConcreteType>(this->getOperation()).iterator_types());
85  }
86  unsigned getNumWindowLoops() {
87  return getNumIterators(
88  getWindowIteratorTypeName(),
89  cast<ConcreteType>(this->getOperation()).iterator_types());
90  }
91  unsigned getNumLoops() {
92  return getNumIterators(
93  cast<ConcreteType>(this->getOperation()).iterator_types());
94  }
95 
96  //==========================================================================//
97  // Input arguments handling.
98  //==========================================================================//
99  // The `i^th` input argument is always the `i^th` operand regardless of
100  // whether we have tensors or buffers.
101  //
102  /// Return the `i`-th input value.
103  Value getInput(unsigned i) {
104  assert(i < nInputs());
105  return this->getOperation()->getOperand(i);
106  }
107  /// Return the index of `value` in the list of inputs if found, llvm::None
108  /// otherwise.
109  Optional<unsigned> getIndexOfInput(Value value) {
110  auto it = llvm::find(getInputs(), value);
111  if (it != getInputs().end())
112  return it - getInputs().begin();
113  return llvm::None;
114  }
115  /// Return the `i`-th input buffer type.
116  ShapedType getInputShapedType(unsigned i) {
117  return getInput(i).getType().template cast<ShapedType>();
118  }
119  /// Return the range over inputs.
120  Operation::operand_range getInputs() {
121  auto range = this->getOperation()->getOperands();
122  return {range.begin(), range.begin() + nInputs()};
123  }
124  /// Query the subset of input operands that are of ranked tensor type.
125  SmallVector<RankedTensorType, 4> getInputTensorTypes() {
126  SmallVector<RankedTensorType, 4> res;
127  for (Type type : getInputs().getTypes())
128  if (auto t = type.template dyn_cast<RankedTensorType>())
129  res.push_back(t);
130  return res;
131  }
132 
133  //==========================================================================//
134  // Output arguments handling.
135  //==========================================================================//
136  // The `i^th` output argument is an operand (resp. a return value) iff it is
137  // a value of buffer type (resp. a return value of tensor type).
138 
139  /// Return the `i`-th output, asserts that this is a buffer operand and not
140  /// a tensor result.
141  Value getOutputBuffer(unsigned i) {
142  assert(i + this->getOperation()->getNumResults() < nOutputs() &&
143  "overflowing output buffer index");
144  return this->getOperation()->getOperand(nInputs() + i);
145  }
146  /// Return the index of `value` in the list of output buffers if found,
147  /// llvm::None otherwise.
148  Optional<unsigned> getIndexOfOutputBuffer(Value value) {
149  auto it = llvm::find(getOutputBuffers(), value);
150  if (it != getOutputBuffers().end())
151  return it - getOutputBuffers().begin();
152  return llvm::None;
153  }
154  /// Return the `i`-th output buffer type.
155  MemRefType getOutputBufferType(unsigned i) {
156  return getOutputBuffer(i).getType().template cast<MemRefType>();
157  }
158  /// Return the `i`-th output shaped type, irrespective of buffer of tensor
159  /// type.
160  ShapedType getOutputShapedType(unsigned i) {
161  return getShapedType(i + nInputs());
162  }
163  /// Query the subset of results that are of ranked tensor type.
164  SmallVector<RankedTensorType, 4> getOutputTensorTypes() {
165  SmallVector<RankedTensorType, 4> res;
166  for (Type type : this->getOperation()->getResults().getTypes())
167  res.push_back(type.template cast<RankedTensorType>());
168  return res;
169  }
170  /// Return the range over outputs.
171  Operation::operand_range getOutputBuffers() {
172  auto range = this->getOperation()->getOperands();
173  return {range.begin() + nInputs(),
174  range.begin() + getNumInputsAndOutputBuffers()};
175  }
176 
177  //==========================================================================//
178  // Input and Output arguments handling.
179  //==========================================================================//
180  /// Return the number of inputs and outputs, irrespective of their buffer or
181  /// tensor type.
182  unsigned getNumInputsAndOutputs() { return nInputs() + nOutputs(); }
183  /// Return the number of inputs, irrespective of their buffer or tensor type,
184  /// and output buffers.
185  unsigned getNumInputsAndOutputBuffers() {
186  assert(this->getOperation()->getNumResults() <= nOutputs());
187  return nInputs() + nOutputs() - this->getOperation()->getNumResults();
188  }
189  /// Return the range over inputs (irrespective of type) and output buffers.
190  Operation::operand_range getInputsAndOutputBuffers() {
191  auto range = this->getOperation()->getOperands();
192  return {range.begin(), range.begin() + getNumInputsAndOutputBuffers()};
193  }
194  /// Return the `i`-th shaped type, there are 3 cases:
195  /// 1. if `i < nInputs()` then return `getInputShapedType(i)`; otherwise
196  /// 2. if `i < getNumInputsAndOutputBuffers()` then return the
197  /// `getOutputBufferType(i - nInputs())`; otherwise
198  /// 3. return the `i - getNumInputsAndOutputBuffers()` result type.
199  ShapedType getShapedType(unsigned i) {
200  if (i < nInputs())
201  return getInputShapedType(i);
202  if (i < getNumInputsAndOutputBuffers())
203  return getOutputBufferType(i - nInputs()).template cast<ShapedType>();
204  return getOutputTensorTypes()[i - getNumInputsAndOutputBuffers()]
205  .template cast<ShapedType>();
206  }
207 
208  //==========================================================================//
209  // Other interface methods.
210  //==========================================================================//
211  /// Query whether the op has only buffer inputs and no returns.
212  bool hasBufferSemantics() {
213  return this->getOperation()->getNumResults() == 0 &&
214  llvm::all_of(getInputs(),
215  [](Value v) { return v.getType().isa<MemRefType>(); });
216  }
217 
218  //==========================================================================//
219  // Other static interface methods.
220  //==========================================================================//
221  static LogicalResult verifyTrait(Operation *op) {
222  auto nOperands = cast<ConcreteType>(op).getNumInputsAndOutputBuffers();
223  if (failed(OpTrait::impl::verifyAtLeastNOperands(op, nOperands)))
224  return failure();
225  return success();
226  }
227 };
