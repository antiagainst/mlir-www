47 /// Given the 'indices' of an load/store operation where the memref is a result
48 /// of a subview op, returns the indices w.r.t to the source memref of the
49 /// subview op. For example
50 ///
51 /// %0 = ... : memref<12x42xf32>
52 /// %1 = subview %0[%arg0, %arg1][][%stride1, %stride2] : memref<12x42xf32> to
53 /// memref<4x4xf32, offset=?, strides=[?, ?]>
54 /// %2 = load %1[%i1, %i2] : memref<4x4xf32, offset=?, strides=[?, ?]>
55 ///
56 /// could be folded into
57 ///
58 /// %2 = load %0[%arg0 + %i1 * %stride1][%arg1 + %i2 * %stride2] :
59 /// memref<12x42xf32>
60 static LogicalResult
61 resolveSourceIndices(Location loc, PatternRewriter &rewriter,
62  SubViewOp subViewOp, ValueRange indices,
63  SmallVectorImpl<Value> &sourceIndices) {
64  // TODO: Aborting when the offsets are static. There might be a way to fold
65  // the subview op with load even if the offsets have been canonicalized
66  // away.
67  if (subViewOp.getNumOffsets() == 0)
68  return failure();
69 
70  ValueRange opOffsets = subViewOp.offsets();
71  SmallVector<Value, 2> opStrides;
72  if (subViewOp.getNumStrides()) {
73  // If the strides are dynamic, get the stride operands.
74  opStrides = llvm::to_vector<2>(subViewOp.strides());
75  } else {
76  // When static, the stride operands can be retrieved by taking the strides
77  // of the result of the subview op, and dividing the strides of the base
78  // memref.
79  SmallVector<int64_t, 2> staticStrides;
80  if (failed(subViewOp.getStaticStrides(staticStrides))) {
81  return failure();
82  }
83  opStrides.reserve(opOffsets.size());
84  for (auto stride : staticStrides) {
85  auto constValAttr = rewriter.getIntegerAttr(
86  IndexType::get(rewriter.getContext()), stride);
87  opStrides.emplace_back(rewriter.create<ConstantOp>(loc, constValAttr));
88  }
89  }
90  assert(opOffsets.size() == opStrides.size());
91 
92  // New indices for the load are the current indices * subview_stride +
93  // subview_offset.
94  assert(indices.size() == opStrides.size());
95  sourceIndices.resize(indices.size());
96  for (auto index : llvm::enumerate(indices)) {
97  auto offset = opOffsets[index.index()];
98  auto stride = opStrides[index.index()];
99  auto mul = rewriter.create<MulIOp>(loc, index.value(), stride);
100  sourceIndices[index.index()] =
101  rewriter.create<AddIOp>(loc, offset, mul).getResult();
102  }
103  return success();
104 }

110 PatternMatchResult
111 LoadOpOfSubViewFolder::matchAndRewrite(LoadOp loadOp,
112  PatternRewriter &rewriter) const {
113  auto subViewOp = dyn_cast_or_null<SubViewOp>(loadOp.memref().getDefiningOp());
114  if (!subViewOp) {
115  return matchFailure();
116  }
117  SmallVector<Value, 4> sourceIndices;
118  if (failed(resolveSourceIndices(loadOp.getLoc(), rewriter, subViewOp,
119  loadOp.indices(), sourceIndices)))
120  return matchFailure();
121 
122  rewriter.replaceOpWithNewOp<LoadOp>(loadOp, subViewOp.source(),
123  sourceIndices);
124  return matchSuccess();
125 }
