21 /// Specifies an interface for basic side-effect modelling that is used by the
22 /// loop-invariant code motion pass.
23 ///
24 /// TODO: This interface should be replaced by a more general solution.
25 class SideEffectsDialectInterface
26  : public DialectInterface::Base<SideEffectsDialectInterface> {
27 public:
28  SideEffectsDialectInterface(Dialect *dialect) : Base(dialect) {}
29 
30  enum SideEffecting {
31  Never, /* the operation has no side-effects */
32  Recursive, /* the operation has side-effects if a contained operation has */
33  Always /* the operation has side-effects */
34  };
35 
36  /// Checks whether the given operation has side-effects.
37  virtual SideEffecting isSideEffecting(Operation *op) const {
38  if (op->hasNoSideEffect())
39  return Never;
40  return Always;
41  };
42 };
43 
44 class SideEffectsInterface
45  : public DialectInterfaceCollection<SideEffectsDialectInterface> {
46 public:
47  using SideEffecting = SideEffectsDialectInterface::SideEffecting;
48  explicit SideEffectsInterface(MLIRContext *ctx)
49  : DialectInterfaceCollection<SideEffectsDialectInterface>(ctx) {}
50 
51  SideEffecting isSideEffecting(Operation *op) const {
52  // First check generic trait.
53  if (op->hasNoSideEffect())
54  return SideEffecting::Never;
55  if (auto handler = getInterfaceFor(op))
56  return handler->isSideEffecting(op);
57 
58  return SideEffecting::Always;
59  }
60 };
