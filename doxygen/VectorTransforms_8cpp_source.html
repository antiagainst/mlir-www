44 /// Given a shape with sizes greater than 0 along all dimensions,
45 /// returns the distance, in number of elements, between a slice in a dimension
46 /// and the next slice in the same dimension.
47 /// e.g. shape[3, 4, 5] -> linearization_basis[20, 5, 1]
48 static SmallVector<int64_t, 8> computeStrides(ArrayRef<int64_t> shape) {
49  if (shape.empty())
50  return {};
51  SmallVector<int64_t, 8> tmp;
52  tmp.reserve(shape.size());
53  int64_t running = 1;
54  for (auto size : llvm::reverse(shape)) {
55  assert(size > 0 && "size must be nonnegative");
56  tmp.push_back(running);
57  running *= size;
58  }
59  return SmallVector<int64_t, 8>(tmp.rbegin(), tmp.rend());
60 }

62 static int64_t computeMaxLinearIndex(ArrayRef<int64_t> basis) {
63  if (basis.empty())
64  return 0;
65  int64_t res = 1;
66  for (auto b : basis)
67  res *= b;
68  return res;
69 }

71 /// Computes and returns the linearized index of 'offsets' w.r.t. 'basis'.
72 static int64_t linearize(ArrayRef<int64_t> offsets, ArrayRef<int64_t> basis) {
73  assert(offsets.size() == basis.size());
74  int64_t linearIndex = 0;
75  for (unsigned idx = 0, e = basis.size(); idx < e; ++idx)
76  linearIndex += offsets[idx] * basis[idx];
77  return linearIndex;
78 }

91 // Populates 'resultElements[indexMap[i]]' with elements from 'inputElements[i]'
92 // for each index 'i' in inputElements with a valid mapping in 'indexMap'.
93 static void getMappedElements(const DenseMap<int64_t, int64_t> &indexMap,
94  ArrayRef<int64_t> inputElements,
95  SmallVectorImpl<int64_t> &resultElements) {
96  assert(indexMap.size() == resultElements.size());
97  assert(inputElements.size() >= resultElements.size());
98  for (unsigned i = 0, e = inputElements.size(); i < e; ++i) {
99  auto it = indexMap.find(i);
100  if (it != indexMap.end())
101  resultElements[it->second] = inputElements[i];
102  }
103 }

105 // Returns a tuple type with vector element types for each resulting slice
106 // of 'vectorType' unrolled by 'sizes' and 'strides'.
107 // TODO(andydavis) Move this to a utility function and share it with
108 // Extract/InsertSlicesOp verification.
109 static TupleType generateExtractSlicesOpResultType(VectorType vectorType,
110  ArrayRef<int64_t> sizes,
111  ArrayRef<int64_t> strides,
112  PatternRewriter &builder) {
113  assert(llvm::all_of(strides, [](int64_t s) { return s == 1; }));
114  assert(static_cast<int64_t>(sizes.size()) == vectorType.getRank());
115  assert(static_cast<int64_t>(strides.size()) == vectorType.getRank());
116 
117  // Compute shape ratio of 'shape' and 'sizes'.
118  auto shape = vectorType.getShape();
119  auto maybeDimSliceCounts = shapeRatio(shape, sizes);
120  assert(maybeDimSliceCounts.hasValue());
121  auto sliceDimCounts = *maybeDimSliceCounts;
122 
123  // Compute strides w.r.t number of slices in each dimension.
124  auto sliceStrides = computeStrides(sliceDimCounts);
125  int64_t sliceCount = computeMaxLinearIndex(sliceDimCounts);
126  SmallVector<Type, 4> vectorTypes(sliceCount);
127  for (unsigned i = 0; i < sliceCount; ++i) {
128  auto vectorOffsets = delinearize(sliceStrides, i);
129  auto elementOffsets =
130  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
131  auto sliceSizes = computeSliceSizes(shape, sizes, elementOffsets);
132  // Create Vector type and add to 'vectorTypes[i]'.
133  vectorTypes[i] = VectorType::get(sliceSizes, vectorType.getElementType());
134  }
135  return TupleType::get(vectorTypes, builder.getContext());
136 }

148 // Populates 'state' with unrolled shape, unroll factors, basis and
149 // num unrolled instances for 'vectorType'.
150 static void initUnrolledVectorState(VectorType vectorType, Value initValue,
151  const DenseMap<int64_t, int64_t> &indexMap,
152  ArrayRef<int64_t> targetShape,
153  UnrolledVectorState &state,
154  PatternRewriter &builder) {
155  // Compute unrolled shape of 'vectorType'.
156  state.unrolledShape.resize(vectorType.getRank());
157  getMappedElements(indexMap, targetShape, state.unrolledShape);
158  // Compute unroll factors for unrolled shape.
159  auto maybeUnrollFactors =
160  shapeRatio(vectorType.getShape(), state.unrolledShape);
161  assert(maybeUnrollFactors.hasValue());
162  state.unrollFactors = *maybeUnrollFactors;
163  // Compute 'basis' and 'numInstances' based on 'state.unrollFactors'.
164  state.basis = computeStrides(state.unrollFactors);
165  state.numInstances = computeMaxLinearIndex(state.unrollFactors);
166  state.slicesTuple = nullptr;
167  if (initValue != nullptr) {
168  // Create ExtractSlicesOp.
169  SmallVector<int64_t, 4> sizes(state.unrolledShape);
170  SmallVector<int64_t, 4> strides(state.unrollFactors.size(), 1);
171  auto tupleType =
172  generateExtractSlicesOpResultType(vectorType, sizes, strides, builder);
173  state.slicesTuple = builder.create<vector::ExtractSlicesOp>(
174  initValue.getLoc(), tupleType, initValue, sizes, strides);
175  }
176 }

178 // Computes and returns the linear index of the unrolled vector at
179 // 'vectorOffsets' within the vector represented by 'state'.
180 static int64_t
181 getUnrolledVectorLinearIndex(UnrolledVectorState &state,
182  ArrayRef<int64_t> vectorOffsets,
183  DenseMap<int64_t, int64_t> &indexMap) {
184  // Compute vector offsets.
185  SmallVector<int64_t, 4> sliceOffsets(state.unrolledShape.size());
186  getMappedElements(indexMap, vectorOffsets, sliceOffsets);
187  // Compute and return linear index of 'sliceOffsets' w.r.t 'state.basis'.
188  return linearize(sliceOffsets, state.basis);
189 }

191 // Returns an unrolled vector at 'vectorOffsets' within the vector
192 // represented by 'state'. The vector is created from a slice of 'initValue'
193 // if not present in 'cache'.
194 static Value getOrCreateUnrolledVectorSlice(
195  Location loc, UnrolledVectorState &state, ArrayRef<int64_t> vectorOffsets,
196  ArrayRef<int64_t> offsets, DenseMap<int64_t, int64_t> &indexMap,
197  Value initValue, SmallVectorImpl<Value> &cache, PatternRewriter &builder) {
198  // Compute slice offsets.
199  SmallVector<int64_t, 4> sliceOffsets(state.unrolledShape.size());
200  getMappedElements(indexMap, offsets, sliceOffsets);
201  // TODO(b/144845578) Support non-1 strides.
202  SmallVector<int64_t, 4> sliceStrides(state.unrolledShape.size(), 1);
203  // Compute linear index of 'sliceOffsets' w.r.t 'state.basis'.
204  int64_t sliceLinearIndex =
205  getUnrolledVectorLinearIndex(state, vectorOffsets, indexMap);
206  assert(sliceLinearIndex < static_cast<int64_t>(cache.size()));
207  auto valueSlice = cache[sliceLinearIndex];
208  if (valueSlice == nullptr) {
209  // Return tuple element at 'sliceLinearIndex'.
210  auto tupleIndex = builder.getI64IntegerAttr(sliceLinearIndex);
211  auto initValueType = initValue.getType().cast<VectorType>();
212  auto vectorType =
213  VectorType::get(state.unrolledShape, initValueType.getElementType());
214  // Initialize 'cache' with slice from 'initValue'.
215  valueSlice = builder.create<vector::TupleGetOp>(
216  loc, vectorType, state.slicesTuple, tupleIndex);
217  // Store value back to 'cache'.
218  cache[sliceLinearIndex] = valueSlice;
219  }
220  return valueSlice;
221 }

282 // TODO(andydavis) Generalize this to support structured ops beyond
283 // vector ContractionOp, and merge it with 'unrollSingleResultOpMatchingType'
284 static Value unrollSingleResultStructuredOp(Operation *op,
285  ArrayRef<int64_t> iterationBounds,
286  std::vector<VectorState> &vectors,
287  unsigned resultIndex,
288  ArrayRef<int64_t> targetShape,
289  PatternRewriter &builder) {
290  auto shapedType = op->getResult(0).getType().dyn_cast_or_null<ShapedType>();
291  if (!shapedType || !shapedType.hasStaticShape())
292  assert(false && "Expected a statically shaped result type");
293 
294  // Compute unroll factors for 'iterationBounds' based on 'targetShape'
295  auto maybeUnrollFactors = shapeRatio(iterationBounds, targetShape);
296  if (!maybeUnrollFactors.hasValue())
297  assert(false && "Failed to compute unroll factors for target shape");
298  auto unrollFactors = *maybeUnrollFactors;
299 
300  // Compute unrolled vector state for each vector in 'vectors'.
301  unsigned numVectors = vectors.size();
302  SmallVector<UnrolledVectorState, 3> unrolledVectorState(numVectors);
303  for (unsigned i = 0; i < numVectors; ++i) {
304  int64_t operandIndex = vectors[i].operandIndex;
305  auto operand = operandIndex >= 0 ? op->getOperand(operandIndex) : nullptr;
306  initUnrolledVectorState(vectors[i].type, operand, vectors[i].indexMap,
307  targetShape, unrolledVectorState[i], builder);
308  }
309  // Compute number of total unrolled instances.
310  auto numUnrolledInstances = computeMaxLinearIndex(unrollFactors);
311  auto sliceStrides = computeStrides(unrollFactors);
312 
313  auto &resultValueState = unrolledVectorState[resultIndex];
314  auto unrolledResultType = VectorType::get(resultValueState.unrolledShape,
315  shapedType.getElementType());
316 
317  // Initialize caches for intermediate vector results.
318  std::vector<SmallVector<Value, 4>> caches(numVectors);
319  for (unsigned i = 0; i < numVectors; ++i)
320  caches[i].resize(unrolledVectorState[i].numInstances);
321 
322  // Unroll 'numUnrolledInstances' of 'op', storing results in 'caches'.
323  for (unsigned i = 0; i < numUnrolledInstances; ++i) {
324  auto vectorOffsets = delinearize(sliceStrides, i);
325  auto elementOffsets =
326  computeElementOffsetsFromVectorSliceOffsets(targetShape, vectorOffsets);
327  // Get cached slice (or create slice) for each operand at 'offsets'.
328  SmallVector<Value, 3> operands;
329  operands.resize(op->getNumOperands());
330  for (unsigned i = 0; i < numVectors; ++i) {
331  int64_t operandIndex = vectors[i].operandIndex;
332  if (operandIndex < 0)
333  continue; // Output
334  auto operand = op->getOperand(operandIndex);
335  operands[operandIndex] = getOrCreateUnrolledVectorSlice(
336  op->getLoc(), unrolledVectorState[i], vectorOffsets, elementOffsets,
337  vectors[i].indexMap, operand, caches[i], builder);
338  }
339  // Create op on sliced vector arguments.
340  auto resultVector =
341  cloneOpWithOperandsAndTypes(builder, op->getLoc(), op, operands,
342  unrolledResultType)
343  ->getResult(0);
344 
345  // Compute linear result index.
346  int64_t linearIndex = getUnrolledVectorLinearIndex(
347  resultValueState, vectorOffsets, vectors[resultIndex].indexMap);
348  // Update result cache at 'linearIndex'.
349  caches[resultIndex][linearIndex] = resultVector;
350  }
351 
352  // Create TupleOp of unrolled result vectors.
353  SmallVector<Type, 4> vectorTupleTypes(resultValueState.numInstances);
354  SmallVector<Value, 4> vectorTupleValues(resultValueState.numInstances);
355  for (unsigned i = 0; i < resultValueState.numInstances; ++i) {
356  vectorTupleTypes[i] = caches[resultIndex][i].getType().cast<VectorType>();
357  vectorTupleValues[i] = caches[resultIndex][i];
358  }
359  TupleType tupleType = builder.getTupleType(vectorTupleTypes);
360  Value tupleOp = builder.create<vector::TupleOp>(op->getLoc(), tupleType,
361  vectorTupleValues);
362 
363  // Create InsertSlicesOp(Tuple(result_vectors)).
364  auto resultVectorType = op->getResult(0).getType().cast<VectorType>();
365  SmallVector<int64_t, 4> sizes(resultValueState.unrolledShape);
366  SmallVector<int64_t, 4> strides(resultValueState.unrollFactors.size(), 1);
367 
368  Value insertSlicesOp = builder.create<vector::InsertSlicesOp>(
369  op->getLoc(), resultVectorType, tupleOp, builder.getI64ArrayAttr(sizes),
370  builder.getI64ArrayAttr(strides));
371  return insertSlicesOp;
372 }

374 static void getVectorContractionOpUnrollState(
375  vector::ContractionOp contractionOp, ArrayRef<int64_t> targetShape,
376  SmallVectorImpl<int64_t> &iterationBounds,
377  std::vector<VectorState> &vectors, unsigned &resultIndex) {
378  // Get contraction op iteration bounds.
379  contractionOp.getIterationBounds(iterationBounds);
380  assert(iterationBounds.size() == targetShape.size());
381  // Get map from iteration space index to lhs/rhs/result shape index.
382  std::vector<DenseMap<int64_t, int64_t>> iterationIndexMapList;
383  contractionOp.getIterationIndexMap(iterationIndexMapList);
384  unsigned numIterators = iterationIndexMapList.size();
385  vectors.resize(numIterators);
386  unsigned accOperandIndex = vector::ContractionOp::getAccOperandIndex();
387  for (unsigned i = 0; i < numIterators; ++i) {
388  vectors[i].type = contractionOp.getOperand(i).getType().cast<VectorType>();
389  vectors[i].indexMap = iterationIndexMapList[i];
390  vectors[i].operandIndex = i;
391  vectors[i].isAcc = i == accOperandIndex ? true : false;
392  }
393 
394  if (llvm::size(contractionOp.masks()) == 2) {
395  // Add vectors for lhs/rhs vector mask arguments. Masks have the
396  // same vector shape lhs/rhs args, so copy their index maps.
397  vectors.push_back({contractionOp.getLHSVectorMaskType(),
398  vectors[0].indexMap, accOperandIndex + 1, false});
399  vectors.push_back({contractionOp.getRHSVectorMaskType(),
400  vectors[1].indexMap, accOperandIndex + 2, false});
401  }
402  // Unroll 'op' 'iterationBounds' to 'targetShape'.
403  // TODO(andydavis) Use linalg style 'args_in'/'args_out' to partition
404  // 'vectors' instead of 'resultIndex'.
405  resultIndex = accOperandIndex;
406 }

408 static void
409 getVectorElementwiseOpUnrollState(Operation *op, ArrayRef<int64_t> targetShape,
410  SmallVectorImpl<int64_t> &iterationBounds,
411  std::vector<VectorState> &vectors,
412  unsigned &resultIndex) {
413  // Verify that operation and operands all have the same vector shape.
414  auto resultType = op->getResult(0).getType().dyn_cast_or_null<VectorType>();
415  assert(resultType && "Expected op with vector result type");
416  auto resultShape = resultType.getShape();
417  // Verify that all operands have the same vector type as result.
418  assert(llvm::all_of(op->getOperandTypes(),
419  [=](Type type) { return type == resultType; }));
420  // Populate 'iterationBounds' with 'resultShape' for elementwise operations.
421  iterationBounds.assign(resultShape.begin(), resultShape.end());
422 
423  // Create trivial elementwise identity index map based on 'resultShape'.
424  DenseMap<int64_t, int64_t> indexMap;
425  indexMap.reserve(resultShape.size());
426  for (unsigned i = 0; i < resultShape.size(); ++i)
427  indexMap[i] = i;
428 
429  // Create VectorState each operand and single result.
430  unsigned numVectors = op->getNumOperands() + op->getNumResults();
431  vectors.resize(numVectors);
432  for (unsigned i = 0; i < op->getNumOperands(); ++i)
433  vectors[i] = {resultType, indexMap, i, false};
434  vectors[numVectors - 1] = {resultType, indexMap, -1, false};
435  resultIndex = numVectors - 1;
436 }

439 SmallVector<Value, 1> mlir::vector::unrollSingleResultOpMatchingType(
440  PatternRewriter &builder, Operation *op, ArrayRef<int64_t> targetShape) {
441  assert(op->getNumResults() == 1 && "Expected single result operation");
442 
443  // Populate 'iterationBounds', 'vectors' and 'resultIndex' to unroll 'op'.
444  SmallVector<int64_t, 6> iterationBounds;

463 // Generates slices of 'vectorType' according to 'sizes' and 'strides, and
464 // calls 'fn' with linear index and indices for each slice.
465 static void
466 generateTransferOpSlices(VectorType vectorType, TupleType tupleType,
467  ArrayRef<int64_t> sizes, ArrayRef<int64_t> strides,
468  ArrayRef<Value> indices, PatternRewriter &rewriter,
469  function_ref<void(unsigned, ArrayRef<Value>)> fn) {
470  // Compute strides w.r.t. to slice counts in each dimension.
471  auto maybeDimSliceCounts = shapeRatio(vectorType.getShape(), sizes);
472  assert(maybeDimSliceCounts.hasValue());
473  auto sliceDimCounts = *maybeDimSliceCounts;
474  auto sliceStrides = computeStrides(sliceDimCounts);
475 
476  int64_t numSlices = tupleType.size();
477  unsigned numSliceIndices = indices.size();
478  auto *ctx = rewriter.getContext();
479  for (unsigned i = 0; i < numSlices; ++i) {
480  auto vectorOffsets = delinearize(sliceStrides, i);
481  auto elementOffsets =
482  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
483  // Compute 'sliceIndices' by adding 'sliceOffsets[i]' to 'indices[i]'.
484  SmallVector<Value, 4> sliceIndices(numSliceIndices);
485  for (auto it : llvm::enumerate(indices)) {
486  auto expr = getAffineDimExpr(0, ctx) +
487  getAffineConstantExpr(elementOffsets[it.index()], ctx);
488  auto map = AffineMap::get(/*dimCount=*/1, /*symbolCount=*/0, expr);
489  sliceIndices[it.index()] = rewriter.create<AffineApplyOp>(
490  it.value().getLoc(), map, ArrayRef<Value>(it.value()));
491  }
492  // Call 'fn' to generate slice 'i' at 'sliceIndices'.
493  fn(i, sliceIndices);
494  }
495 }

500 struct SplitTransferReadOp : public OpRewritePattern<vector::TransferReadOp> {
501  using OpRewritePattern<vector::TransferReadOp>::OpRewritePattern;
502 
503  PatternMatchResult matchAndRewrite(vector::TransferReadOp xferReadOp,
504  PatternRewriter &rewriter) const override {
505  // TODO(andydavis, ntv) Support splitting TransferReadOp with non-identity
506  // permutation maps. Repurpose code from MaterializeVectors transformation.
507  if (!xferReadOp.permutation_map().isIdentity())
508  return matchFailure();
509  // Return unless the unique 'xferReadOp' user is an ExtractSlicesOp.
510  Value xferReadResult = xferReadOp.getResult();
511  auto extractSlicesOp =
512  dyn_cast<vector::ExtractSlicesOp>(*xferReadResult.getUsers().begin());
513  if (!xferReadResult.hasOneUse() || !extractSlicesOp)
514  return matchFailure();
515 
516  // Get 'sizes' and 'strides' parameters from ExtractSlicesOp user.
517  auto sourceVectorType = extractSlicesOp.getSourceVectorType();
518  auto resultTupleType = extractSlicesOp.getResultTupleType();
519  SmallVector<int64_t, 4> sizes;
520  extractSlicesOp.getSizes(sizes);
521  SmallVector<int64_t, 4> strides;
522  extractSlicesOp.getStrides(strides);
523  assert(llvm::all_of(strides, [](int64_t s) { return s == 1; }));
524 
525  Location loc = xferReadOp.getLoc();
526  int64_t numSlices = resultTupleType.size();
527  SmallVector<Value, 4> vectorTupleValues(numSlices);
528  SmallVector<Value, 4> indices(xferReadOp.indices().begin(),
529  xferReadOp.indices().end());
530  auto createSlice = [&](unsigned index, ArrayRef<Value> sliceIndices) {
531  // Get VectorType for slice 'i'.
532  auto sliceVectorType = resultTupleType.getType(index);
533  // Create split TransferReadOp for 'sliceUser'.
534  vectorTupleValues[index] = rewriter.create<vector::TransferReadOp>(
535  loc, sliceVectorType, xferReadOp.memref(), sliceIndices,

539  indices, rewriter, createSlice);

542  Value tupleOp = rewriter.create<vector::TupleOp>(loc, resultTupleType,
543  vectorTupleValues);

545  rewriter.replaceOpWithNewOp<vector::InsertSlicesOp>(

553 struct SplitTransferWriteOp : public OpRewritePattern<vector::TransferWriteOp> {
554  using OpRewritePattern<vector::TransferWriteOp>::OpRewritePattern;
555 
556  PatternMatchResult matchAndRewrite(vector::TransferWriteOp xferWriteOp,
557  PatternRewriter &rewriter) const override {
558  // TODO(andydavis, ntv) Support splitting TransferWriteOp with non-identity
559  // permutation maps. Repurpose code from MaterializeVectors transformation.
560  if (!xferWriteOp.permutation_map().isIdentity())
561  return matchFailure();
562  // Return unless the 'xferWriteOp' 'vector' operand is an 'InsertSlicesOp'.
563  auto *vectorDefOp = xferWriteOp.vector().getDefiningOp();
564  auto insertSlicesOp = dyn_cast_or_null<vector::InsertSlicesOp>(vectorDefOp);
565  if (!insertSlicesOp)
566  return matchFailure();
567 
568  // Get TupleOp operand of 'insertSlicesOp'.
569  auto tupleOp = dyn_cast_or_null<vector::TupleOp>(
570  insertSlicesOp.vectors().getDefiningOp());
571  if (!tupleOp)
572  return matchFailure();
573 
574  // Get 'sizes' and 'strides' parameters from InsertSlicesOp user.
575  auto sourceTupleType = insertSlicesOp.getSourceTupleType();
576  auto resultVectorType = insertSlicesOp.getResultVectorType();
577  SmallVector<int64_t, 4> sizes;
578  insertSlicesOp.getSizes(sizes);
579  SmallVector<int64_t, 4> strides;
580  insertSlicesOp.getStrides(strides);
581 
582  Location loc = xferWriteOp.getLoc();
583  SmallVector<Value, 4> indices(xferWriteOp.indices().begin(),
584  xferWriteOp.indices().end());
585  auto createSlice = [&](unsigned index, ArrayRef<Value> sliceIndices) {
586  // Create split TransferWriteOp for source vector 'tupleOp.operand[i]'.
587  rewriter.create<vector::TransferWriteOp>(
588  loc, tupleOp.getOperand(index), xferWriteOp.memref(), sliceIndices,
589  xferWriteOp.permutation_map());
590  };

592  indices, rewriter, createSlice);
593 
594  // Erase old 'xferWriteOp'.
595  rewriter.eraseOp(xferWriteOp);
596  return matchSuccess();
597  }

633 /// Progressive lowering of ExtractSlicesOp to tuple of StridedSliceOp.
634 /// One:
635 /// %x = vector.extract_slices %0
636 /// is replaced by:
637 /// %a = vector.strided_slice %0
638 /// %b = vector.strided_slice %0
639 /// ..
640 /// %x = vector.tuple %a, %b, ..
641 class ExtractSlicesOpLowering
642  : public OpRewritePattern<vector::ExtractSlicesOp> {
643 public:
644  using OpRewritePattern<vector::ExtractSlicesOp>::OpRewritePattern;
645 
646  PatternMatchResult matchAndRewrite(vector::ExtractSlicesOp op,
647  PatternRewriter &rewriter) const override {
648  auto loc = op.getLoc();
649 
650  VectorType vectorType = op.getSourceVectorType();
651  auto shape = vectorType.getShape();
652 
653  SmallVector<int64_t, 4> sizes;
654  op.getSizes(sizes);
655  SmallVector<int64_t, 4> strides;
656  op.getStrides(strides); // all-ones at the moment
657 
658  // For each element in the tuple, generate the proper strided slice.
659  TupleType tupleType = op.getResultTupleType();
660  int64_t tupleSize = tupleType.size();
661  SmallVector<Value, 4> tupleValues(tupleSize);
662  auto sliceStrides = computeStrides(shape, sizes);
663  for (int64_t i = 0; i < tupleSize; ++i) {
664  auto vectorOffsets = delinearize(sliceStrides, i);
665  auto elementOffsets =
666  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
667  auto sliceSizes = computeSliceSizes(shape, sizes, elementOffsets);
668  // Insert in tuple.
669  tupleValues[i] = rewriter.create<vector::StridedSliceOp>(
670  loc, op.vector(), elementOffsets, sliceSizes, strides);
671  }
672 
673  rewriter.replaceOpWithNewOp<vector::TupleOp>(op, tupleType, tupleValues);
674  return matchSuccess();
675  }
676 };

678 /// Progressive lowering of InsertSlicesOp to series of InsertStridedSliceOp.
679 /// One:
680 /// %x = vector.insert_slices %0
681 /// is replaced by:
682 /// %r0 = vector.splat 0
683 // %t1 = vector.tuple_get %0, 0
684 /// %r1 = vector.insert_strided_slice %r0, %t1
685 // %t2 = vector.tuple_get %0, 1
686 /// %r2 = vector.insert_strided_slice %r1, %t2
687 /// ..
688 /// %x = ..
689 class InsertSlicesOpLowering : public OpRewritePattern<vector::InsertSlicesOp> {
690 public:
691  using OpRewritePattern<vector::InsertSlicesOp>::OpRewritePattern;
692 
693  PatternMatchResult matchAndRewrite(vector::InsertSlicesOp op,
694  PatternRewriter &rewriter) const override {
695  auto loc = op.getLoc();
696 
697  VectorType vectorType = op.getResultVectorType();
698  auto shape = vectorType.getShape();
699 
700  SmallVector<int64_t, 4> sizes;
701  op.getSizes(sizes);
702  SmallVector<int64_t, 4> strides;
703  op.getStrides(strides); // all-ones at the moment
704 
705  // Prepare result.
706  auto elemType = vectorType.getElementType();
707  Value zero = rewriter.create<ConstantOp>(loc, elemType,
708  rewriter.getZeroAttr(elemType));
709  Value result = rewriter.create<SplatOp>(loc, vectorType, zero);
710 
711  // For each element in the tuple, extract the proper strided slice.
712  TupleType tupleType = op.getSourceTupleType();
713  int64_t tupleSize = tupleType.size();
714  auto sliceStrides = computeStrides(shape, sizes);
715  for (int64_t i = 0; i < tupleSize; ++i) {
716  auto vectorOffsets = delinearize(sliceStrides, i);
717  auto elementOffsets =
718  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
719  // Extract from tuple into the result.
720  auto index = rewriter.getI64IntegerAttr(i);
721  auto tupleGet = rewriter.create<vector::TupleGetOp>(
722  loc, tupleType.getType(i), op.getOperand(), index);
723  result = rewriter.create<vector::InsertStridedSliceOp>(
724  loc, tupleGet, result, elementOffsets, strides);
725  }
726 
727  rewriter.replaceOp(op, result);
728  return matchSuccess();
729  }
730 };
