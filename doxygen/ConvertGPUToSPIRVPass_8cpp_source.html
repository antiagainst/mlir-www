27 /// Pass to lower GPU Dialect to SPIR-V. The pass only converts those functions
28 /// that have the "gpu.kernel" attribute, i.e. those functions that are
29 /// referenced in gpu::LaunchKernelOp operations. For each such function
30 ///
31 /// 1) Create a spirv::ModuleOp, and clone the function into spirv::ModuleOp
32 /// (the original function is still needed by the gpu::LaunchKernelOp, so cannot
33 /// replace it).
34 ///
35 /// 2) Lower the body of the spirv::ModuleOp.
36 class GPUToSPIRVPass : public ModulePass<GPUToSPIRVPass> {
37 public:
38  GPUToSPIRVPass() = default;
39  GPUToSPIRVPass(const GPUToSPIRVPass &) {}
40  GPUToSPIRVPass(ArrayRef<int64_t> workGroupSize) {
41  this->workGroupSize = workGroupSize;
42  }
43 
44  void runOnModule() override;
45 
46 private:
47  /// Command line option to specify the workgroup size.
48  ListOption<int64_t> workGroupSize{
49  *this, "workgroup-size",
50  llvm::cl::desc(
51  "Workgroup Sizes in the SPIR-V module for x, followed by y, followed "
52  "by z dimension of the dispatch (others will be ignored)"),
53  llvm::cl::ZeroOrMore, llvm::cl::MiscFlags::CommaSeparated};
54 };
55 } // namespace

57 void GPUToSPIRVPass::runOnModule() {
58  MLIRContext *context = &getContext();
59  ModuleOp module = getModule();
60 
61  SmallVector<Operation *, 1> kernelModules;
62  OpBuilder builder(context);
63  module.walk([&builder, &kernelModules](gpu::GPUModuleOp moduleOp) {
64  // For each kernel module (should be only 1 for now, but that is not a
65  // requirement here), clone the module for conversion because the
66  // gpu.launch function still needs the kernel module.
67  builder.setInsertionPoint(moduleOp.getOperation());
68  kernelModules.push_back(builder.clone(*moduleOp.getOperation()));
69  });
70 
71  SPIRVTypeConverter typeConverter;
72  OwningRewritePatternList patterns;
73  populateGPUToSPIRVPatterns(context, typeConverter, patterns, workGroupSize);
74  populateStandardToSPIRVPatterns(context, typeConverter, patterns);
75 
76  std::unique_ptr<ConversionTarget> target = spirv::SPIRVConversionTarget::get(
77  spirv::lookupTargetEnvOrDefault(module), context);
78  target->addDynamicallyLegalOp<FuncOp>(
79  [&](FuncOp op) { return typeConverter.isSignatureLegal(op.getType()); });
80 
81  if (failed(applyFullConversion(kernelModules, *target, patterns,
82  &typeConverter))) {
83  return signalPassFailure();
84  }
85 }
