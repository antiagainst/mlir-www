34 ArrayRef<Extension> spirv::getImpliedExtensions(Version version) {
35  // Note: the following lists are from "Appendix A: Changes" of the spec.
36 
37 #define V_1_3_IMPLIED_EXTS \
38  Extension::SPV_KHR_shader_draw_parameters, Extension::SPV_KHR_16bit_storage, \
39  Extension::SPV_KHR_device_group, Extension::SPV_KHR_multiview, \
40  Extension::SPV_KHR_storage_buffer_storage_class, \
41  Extension::SPV_KHR_variable_pointers
42 
43 #define V_1_4_IMPLIED_EXTS \
44  Extension::SPV_KHR_no_integer_wrap_decoration, \
45  Extension::SPV_GOOGLE_decorate_string, \
46  Extension::SPV_GOOGLE_hlsl_functionality1, \
47  Extension::SPV_KHR_float_controls
48 
49 #define V_1_5_IMPLIED_EXTS \
50  Extension::SPV_KHR_8bit_storage, Extension::SPV_EXT_descriptor_indexing, \
51  Extension::SPV_EXT_shader_viewport_index_layer, \
52  Extension::SPV_EXT_physical_storage_buffer, \
53  Extension::SPV_KHR_physical_storage_buffer, \
54  Extension::SPV_KHR_vulkan_memory_model
55 
56  switch (version) {
57  default:
58  return {};
59  case Version::V_1_3: {
60  static const Extension exts[] = {V_1_3_IMPLIED_EXTS};
61  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
62  }
63  case Version::V_1_4: {
64  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS};
65  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
66  }
67  case Version::V_1_5: {
68  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS,
69  V_1_5_IMPLIED_EXTS};
70  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
71  }
72  }
73 
74 #undef V_1_5_IMPLIED_EXTS
75 #undef V_1_4_IMPLIED_EXTS
76 #undef V_1_3_IMPLIED_EXTS
77 }

447 struct spirv::detail::StructTypeStorage : public TypeStorage {
448  StructTypeStorage(
449  unsigned numMembers, Type const *memberTypes,
450  StructType::LayoutInfo const *layoutInfo, unsigned numMemberDecorations,
451  StructType::MemberDecorationInfo const *memberDecorationsInfo)
452  : TypeStorage(numMembers), memberTypes(memberTypes),
453  layoutInfo(layoutInfo), numMemberDecorations(numMemberDecorations),
454  memberDecorationsInfo(memberDecorationsInfo) {}
455 
456  using KeyTy = std::tuple<ArrayRef<Type>, ArrayRef<StructType::LayoutInfo>,
457  ArrayRef<StructType::MemberDecorationInfo>>;
458  bool operator==(const KeyTy &key) const {
459  return key ==
460  KeyTy(getMemberTypes(), getLayoutInfo(), getMemberDecorationsInfo());
461  }
462 
463  static StructTypeStorage *construct(TypeStorageAllocator &allocator,
464  const KeyTy &key) {
465  ArrayRef<Type> keyTypes = std::get<0>(key);
466 
467  // Copy the member type and layout information into the bump pointer
468  const Type *typesList = nullptr;
469  if (!keyTypes.empty()) {
470  typesList = allocator.copyInto(keyTypes).data();
471  }
472 
473  const StructType::LayoutInfo *layoutInfoList = nullptr;
474  if (!std::get<1>(key).empty()) {
475  ArrayRef<StructType::LayoutInfo> keyLayoutInfo = std::get<1>(key);
476  assert(keyLayoutInfo.size() == keyTypes.size() &&
477  "size of layout information must be same as the size of number of "
478  "elements");
479  layoutInfoList = allocator.copyInto(keyLayoutInfo).data();
480  }
481 
482  const StructType::MemberDecorationInfo *memberDecorationList = nullptr;
483  unsigned numMemberDecorations = 0;
484  if (!std::get<2>(key).empty()) {
485  auto keyMemberDecorations = std::get<2>(key);
486  numMemberDecorations = keyMemberDecorations.size();
487  memberDecorationList = allocator.copyInto(keyMemberDecorations).data();
488  }
489  return new (allocator.allocate<StructTypeStorage>())
490  StructTypeStorage(keyTypes.size(), typesList, layoutInfoList,
491  numMemberDecorations, memberDecorationList);
492  }
493 
494  ArrayRef<Type> getMemberTypes() const {
495  return ArrayRef<Type>(memberTypes, getSubclassData());
496  }
497 
498  ArrayRef<StructType::LayoutInfo> getLayoutInfo() const {
499  if (layoutInfo) {
500  return ArrayRef<StructType::LayoutInfo>(layoutInfo, getSubclassData());
501  }
502  return {};
503  }
