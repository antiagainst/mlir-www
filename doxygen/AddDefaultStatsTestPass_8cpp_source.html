63 void AddDefaultStatsPass::runWithConfig(SolverContext &solverContext,
64  const TargetConfiguration &config) {
65  auto func = getFunction();
66 
67  // Insert stats for each argument.
68  for (auto arg : func.getArguments()) {
69  if (!config.isHandledType(arg.getType()))
70  continue;
71  OpBuilder b(func.getBody());
72  APFloat minValue(-1.0f);
73  APFloat maxValue(1.0f);
74  ElementsAttr layerStats = DenseFPElementsAttr::get(
75  RankedTensorType::get({2}, b.getF32Type()), {minValue, maxValue});
76  auto statsOp = b.create<StatisticsOp>(func.getLoc(), arg, layerStats,
77  nullptr, nullptr);
78  arg.replaceAllUsesWith(statsOp);
79 
80  // StatsOp contained a use to 'arg' so make sure to reset it after replacing
81  // all of the uses of 'arg'.
82  statsOp.getOperation()->replaceUsesOfWith(statsOp, arg);
83  }
84 
85  // Walk the ops and insert stats.
86  func.walk([&](Operation *op) {
87  if (!config.isRequireStatsOp(op)) {
88  return;
89  }
90  assert(op->getNumResults() == 1);
91 
92  auto originalResult = op->getResult(0);
93  if (!config.isHandledType(originalResult.getType()))
94  return;
95 
96  OpBuilder b(op->getBlock(), ++op->getIterator());
97 
98  APFloat minValue(-1.0f);
99  APFloat maxValue(1.0f);
100  ElementsAttr layerStats = DenseFPElementsAttr::get(
101  RankedTensorType::get({2}, b.getF32Type()), {minValue, maxValue});
102  auto statsOp = b.create<StatisticsOp>(op->getLoc(), op->getResult(0),
103  layerStats, nullptr, nullptr);
104  originalResult.replaceAllUsesWith(statsOp);
105 
106  // StatsOp contained a use to 'op' so make sure to reset it after replacing
107  // all of the uses of 'op'.
108  statsOp.getOperation()->replaceUsesOfWith(statsOp, originalResult);
109  });
110 }
