19 spirv::StructType
20 VulkanLayoutUtils::decorateType(spirv::StructType structType,
21  VulkanLayoutUtils::Size &size,
22  VulkanLayoutUtils::Size &alignment) {
23  if (structType.getNumElements() == 0) {
24  return structType;
25  }
26 
27  SmallVector<Type, 4> memberTypes;
28  SmallVector<VulkanLayoutUtils::Size, 4> layoutInfo;
29  SmallVector<spirv::StructType::MemberDecorationInfo, 4> memberDecorations;
30 
31  VulkanLayoutUtils::Size structMemberOffset = 0;
32  VulkanLayoutUtils::Size maxMemberAlignment = 1;
33 
34  for (uint32_t i = 0, e = structType.getNumElements(); i < e; ++i) {
35  VulkanLayoutUtils::Size memberSize = 0;
36  VulkanLayoutUtils::Size memberAlignment = 1;
37 
38  auto memberType = VulkanLayoutUtils::decorateType(
39  structType.getElementType(i), memberSize, memberAlignment);
40  structMemberOffset = llvm::alignTo(structMemberOffset, memberAlignment);
41  memberTypes.push_back(memberType);
42  layoutInfo.push_back(structMemberOffset);
43  // According to the Vulkan spec:
44  // "A structure has a base alignment equal to the largest base alignment of
45  // any of its members."
46  structMemberOffset += memberSize;
47  maxMemberAlignment = std::max(maxMemberAlignment, memberAlignment);
48  }
49 
50  // According to the Vulkan spec:
51  // "The Offset decoration of a member must not place it between the end of a
52  // structure or an array and the next multiple of the alignment of that
53  // structure or array."
54  size = llvm::alignTo(structMemberOffset, maxMemberAlignment);
55  alignment = maxMemberAlignment;
56  structType.getMemberDecorations(memberDecorations);
57  return spirv::StructType::get(memberTypes, layoutInfo, memberDecorations);
58 }

60 Type VulkanLayoutUtils::decorateType(Type type, VulkanLayoutUtils::Size &size,
61  VulkanLayoutUtils::Size &alignment) {
62  if (spirv::SPIRVDialect::isValidScalarType(type)) {
63  alignment = VulkanLayoutUtils::getScalarTypeAlignment(type);
64  // Vulkan spec does not specify any padding for a scalar type.
65  size = alignment;
66  return type;
67  }
68 
69  switch (type.getKind()) {
70  case spirv::TypeKind::Struct:
71  return VulkanLayoutUtils::decorateType(type.cast<spirv::StructType>(), size,
72  alignment);
73  case spirv::TypeKind::Array:
74  return VulkanLayoutUtils::decorateType(type.cast<spirv::ArrayType>(), size,
75  alignment);
76  case StandardTypes::Vector:
77  return VulkanLayoutUtils::decorateType(type.cast<VectorType>(), size,
78  alignment);
79  default:
80  llvm_unreachable("unhandled SPIR-V type");
81  }
82 }

84 Type VulkanLayoutUtils::decorateType(VectorType vectorType,
85  VulkanLayoutUtils::Size &size,
86  VulkanLayoutUtils::Size &alignment) {
87  const auto numElements = vectorType.getNumElements();
88  auto elementType = vectorType.getElementType();
89  VulkanLayoutUtils::Size elementSize = 0;
90  VulkanLayoutUtils::Size elementAlignment = 1;
91 
92  auto memberType = VulkanLayoutUtils::decorateType(elementType, elementSize,
93  elementAlignment);
94  // According to the Vulkan spec:
95  // 1. "A two-component vector has a base alignment equal to twice its scalar
96  // alignment."
97  // 2. "A three- or four-component vector has a base alignment equal to four
98  // times its scalar alignment."
99  size = elementSize * numElements;
100  alignment = numElements == 2 ? elementAlignment * 2 : elementAlignment * 4;
101  return VectorType::get(numElements, memberType);
102 }

122 VulkanLayoutUtils::Size
123 VulkanLayoutUtils::getScalarTypeAlignment(Type scalarType) {
124  // According to the Vulkan spec:
125  // 1. "A scalar of size N has a scalar alignment of N."
126  // 2. "A scalar has a base alignment equal to its scalar alignment."
127  // 3. "A scalar, vector or matrix type has an extended alignment equal to its
128  // base alignment."
129  auto bitWidth = scalarType.getIntOrFloatBitWidth();
130  if (bitWidth == 1)
131  return 1;
132  return bitWidth / 8;
133 }

135 bool VulkanLayoutUtils::isLegalType(Type type) {
136  auto ptrType = type.dyn_cast<spirv::PointerType>();
137  if (!ptrType) {
138  return true;
139  }
140 
141  auto storageClass = ptrType.getStorageClass();
142  auto structType = ptrType.getPointeeType().dyn_cast<spirv::StructType>();
143  if (!structType) {
144  return true;
145  }
146 
147  switch (storageClass) {
148  case spirv::StorageClass::Uniform:
149  case spirv::StorageClass::StorageBuffer:
150  case spirv::StorageClass::PushConstant:
151  case spirv::StorageClass::PhysicalStorageBuffer:
152  return structType.hasLayout() || !structType.getNumElements();
153  default:
154  return true;
155  }
