25 /// Utility class for implementing users of storage classes uniqued by a
26 /// StorageUniquer. Clients are not expected to interact with this class
27 /// directly.
28 template <typename ConcreteT, typename BaseT, typename StorageT,
29  typename UniquerT>
30 class StorageUserBase : public BaseT {
31 public:
32  using BaseT::BaseT;
33 
34  /// Utility declarations for the concrete attribute class.
35  using Base = StorageUserBase<ConcreteT, BaseT, StorageT, UniquerT>;
36  using ImplType = StorageT;
37 
38  /// Return a unique identifier for the concrete type.
39  static ClassID *getClassID() { return ClassID::getID<ConcreteT>(); }
40 
41  /// Provide a default implementation of 'classof' that invokes a 'kindof'
42  /// method on the concrete type.
43  template <typename T> static bool classof(T val) {
44  static_assert(std::is_convertible<ConcreteT, T>::value,
45  "casting from a non-convertible type");
46  return ConcreteT::kindof(val.getKind());
47  }
48 
49 protected:
50  /// Get or create a new ConcreteT instance within the ctx. This
51  /// function is guaranteed to return a non null object and will assert if
52  /// the arguments provided are invalid.
53  template <typename... Args>
54  static ConcreteT get(MLIRContext *ctx, unsigned kind, Args... args) {
55  // Ensure that the invariants are correct for construction.
56  assert(succeeded(
57  ConcreteT::verifyConstructionInvariants(llvm::None, ctx, args...)));
58  return UniquerT::template get<ConcreteT>(ctx, kind, args...);
59  }
60 
61  /// Get or create a new ConcreteT instance within the ctx, defined at
62  /// the given, potentially unknown, location. If the arguments provided are
63  /// invalid then emit errors and return a null object.
64  template <typename... Args>
65  static ConcreteT getChecked(const Location &loc, MLIRContext *ctx,
66  unsigned kind, Args... args) {
67  // If the construction invariants fail then we return a null attribute.
68  if (failed(ConcreteT::verifyConstructionInvariants(loc, ctx, args...)))
69  return ConcreteT();
70  return UniquerT::template get<ConcreteT>(ctx, kind, args...);
71  }
72 
73  /// Default implementation that just returns success.
74  template <typename... Args>
75  static LogicalResult verifyConstructionInvariants(Args... args) {
76  return success();
77  }
78 
79  /// Utility for easy access to the storage instance.
80  ImplType *getImpl() const { return static_cast<ImplType *>(this->impl); }
81 };
82 } // namespace detail
