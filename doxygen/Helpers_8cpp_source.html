16 static SmallVector<ValueHandle, 8> getMemRefSizes(Value memRef) {
17  MemRefType memRefType = memRef.getType().cast<MemRefType>();
18  assert(isStrided(memRefType) && "Expected strided MemRef type");
19 
20  SmallVector<ValueHandle, 8> res;
21  res.reserve(memRefType.getShape().size());
22  const auto &shape = memRefType.getShape();
23  for (unsigned idx = 0, n = shape.size(); idx < n; ++idx) {
24  if (shape[idx] == -1) {
25  res.push_back(ValueHandle::create<DimOp>(memRef, idx));
26  } else {
27  res.push_back(static_cast<index_type>(shape[idx]));
28  }
29  }
30  return res;
31 }
