40 static SmallVector<ValueHandle, 8>
41 makeCanonicalAffineApplies(OpBuilder &b, Location loc, AffineMap map,
42  ArrayRef<Value> vals) {
43  assert(map.getNumSymbols() == 0);
44  assert(map.getNumInputs() == vals.size());
45  SmallVector<ValueHandle, 8> res;
46  res.reserve(map.getNumResults());
47  auto dims = map.getNumDims();
48  for (auto e : map.getResults()) {
49  auto exprMap = AffineMap::get(dims, 0, e);
50  SmallVector<Value, 4> operands(vals.begin(), vals.end());
51  canonicalizeMapAndOperands(&exprMap, &operands);
52  res.push_back(affine_apply(exprMap, operands));
53  }
54  return res;
55 }

65 // Creates a number of ranges equal to the number of results in `map`.
66 // The returned ranges correspond to the loop ranges, in the proper order, for
67 // which new loops will be created.
68 static SmallVector<Value, 4> emitLoopRanges(OpBuilder &b, Location loc,
69  AffineMap map,
70  ArrayRef<Value> allViewSizes);
71 SmallVector<Value, 4> emitLoopRanges(OpBuilder &b, Location loc, AffineMap map,
72  ArrayRef<Value> allViewSizes) {
73  // Apply `map` to get view sizes in loop order.
74  auto sizes = applyMapToValues(b, loc, map, allViewSizes);
75  // Create a new range with the applied tile sizes.
76  ScopedContext scope(b, loc);
77  SmallVector<Value, 4> res;
78  for (unsigned idx = 0, e = map.getNumResults(); idx < e; ++idx) {
79  res.push_back(
80  linalg_range(constant_index(0), sizes[idx], constant_index(1)));
81  }
82  return res;
83 }

89 template <typename IndexedValueType>
90 class LinalgScopedEmitter<IndexedValueType, CopyOp> {
91 public:
92  static void emitScalarImplementation(ArrayRef<Value> allIvs, CopyOp copyOp) {
93  assert(copyOp.hasBufferSemantics() &&
94  "expected linalg op with buffer semantics");
95  auto nPar = copyOp.getNumParallelLoops();
96  assert(nPar == allIvs.size());
97  auto inputIvs =
98  permuteIvs(allIvs.take_front(nPar), copyOp.inputPermutation());
99  auto outputIvs =
100  permuteIvs(allIvs.take_front(nPar), copyOp.outputPermutation());
101  SmallVector<IndexHandle, 8> iivs(inputIvs.begin(), inputIvs.end());
102  SmallVector<IndexHandle, 8> oivs(outputIvs.begin(), outputIvs.end());
103  IndexedValueType O(copyOp.getOutputBuffer(0)), I(copyOp.getInput(0));
104  // Emit the proper scalar assignment, whether we are dealing with a 0-D or
105  // an n-D loop nest; with or without permutations.
106  // clang-format off
107  nPar > 0 ? O(oivs) = I(iivs) :
108  O() = I();
109  // clang-format on
110  }
111 };

199 // Emits the MLIR for the scalar part of the generic op by:
200 // 1. Emitting std_load and std_store ops for each input and output
201 // view in order. This is achieved by applying the appropriate input or
202 // output map to the enclosing induction variables.
203 // 2. Emitting a call to `op.fun()` that takes as arguments the scalars
204 // from point 1. above.
205 // 3. Emitting std_store to store the results of 2. to the output
206 // views.
207 //
208 // An example output may resemble:
209 //
210 // ```
211 // loop.for %i = %c0 to %0 step %c1 {
212 // loop.for %j = %c0 to %1 step %c1 {
213 // loop.for %k = %c0 to %4 step %c1 {
214 // %11 = load %arg0[%i, %j] :
215 // memref<?x?xf32, stride_specification>
216 // %12 = load %arg1[%i, %j, %k] :
217 // memref<?x?x?xf32, stride_specification>
218 // %13 = load %arg2[%i, %k, %j] :
219 // memref<?x?x?xf32, stride_specification>
220 // %14:2 = call @foo(%11, %12, %13) : (f32, f32, f32) -> (f32, f32)
221 // store %14#0, %arg1[%i, %j, %k] :
222 // memref<?x?x?Xf32, stride_specification>
223 // store %14#1, %arg2[%i, %k, %j] :
224 // memref<?x?x?Xf32, stride_specification>
225 // }
226 // }
227 // }
228 // ```
229 template <typename IndexedValueType>
230 class LinalgScopedEmitter<IndexedValueType, GenericOp> {
231 public:
232  static void emitScalarImplementation(ArrayRef<Value> allIvs,
233  GenericOp genericOp) {
234  assert(genericOp.hasBufferSemantics() &&
235  "expected linalg op with buffer semantics");
236  auto b = ScopedContext::getBuilder();
237  auto loc = ScopedContext::getLocation();
238  using edsc::intrinsics::detail::ValueHandleArray;
239  unsigned nInputs = genericOp.getNumInputs();
240  unsigned nOutputs = genericOp.getNumOutputs();
241  SmallVector<Value, 4> indexedValues(nInputs + nOutputs);
242 
243  // 1.a. Emit std_load from input views.
244  for (unsigned i = 0; i < nInputs; ++i) {
245  ValueHandleArray indexing(makeCanonicalAffineApplies(
246  b, loc, genericOp.getInputIndexingMap(i), allIvs));
247  indexedValues[i] = std_load(genericOp.getInput(i), indexing);
248  }
249 
250  // 1.b. Emit std_load from output views.
251  for (unsigned i = 0; i < nOutputs; ++i) {
252  ValueHandleArray indexing(makeCanonicalAffineApplies(
253  b, loc, genericOp.getOutputIndexingMap(i), allIvs));
254  indexedValues[nInputs + i] =
255  std_load(genericOp.getOutputBuffer(i), indexing);
256  }
257 
258  auto funcOp = genericOp.getFunction();
259  if (funcOp) {
260  // 2. Emit call.
261  Operation *callOp = call(funcOp, indexedValues);
262  assert(callOp->getNumResults() == genericOp.getNumOutputs());
263 
264  // 3. Emit std_store.
265  for (unsigned i = 0; i < nOutputs; ++i) {
266  ValueHandleArray indexing(makeCanonicalAffineApplies(
267  b, loc, genericOp.getOutputIndexingMap(i), allIvs));
268  std_store(callOp->getResult(i), genericOp.getOutputBuffer(i), indexing);
269  }
270  return;
271  }
272  // TODO(ntv): When a region inliner exists, use it.
273  // 2. Inline region, currently only works for a single basic block.
274  BlockAndValueMapping map;
275  auto &block = genericOp.region().front();
276  for (auto it : llvm::zip(block.getArguments(), indexedValues))
277  map.map(std::get<0>(it), std::get<1>(it));
278  for (auto &op : block.without_terminator()) {
279  assert(op.getNumRegions() == 0);
280  auto *newOp = b.clone(op, map);
281  for (auto it : llvm::zip(op.getResults(), newOp->getResults()))
282  map.map(std::get<0>(it), std::get<1>(it));
283  }
284 
285  // 3. Emit std_store.
286  auto *yieldOp = cast<YieldOp>(block.back()).getOperation();
287  assert(yieldOp->getNumOperands() == nOutputs);
288  for (unsigned i = 0; i < nOutputs; ++i) {
289  ValueHandleArray indexing(makeCanonicalAffineApplies(
290  b, loc, genericOp.getOutputIndexingMap(i), allIvs));
291  std_store(map.lookup(yieldOp->getOperand(i)),
292  genericOp.getOutputBuffer(i), indexing);
293  }
294  }
295 };

297 // Emits the MLIR for the scalar part of the indexed generic op by:
298 // 1. Emitting std_load and std_store ops for each input and output view in
299 // order. This is achieved by applying the appropriate input or output map
300 // to the enclosing induction variables.
301 // 2. Emitting a call to `op.fun()` that takes as arguments the induction
302 // variables and the scalars from point 1. above.
303 // 3. Emitting std_store to store the results of 2. to the output views.
304 //
305 // An example output may resemble:
306 //
307 // ```
308 // loop.for %i = %c0 to %0 step %c1 {
309 // loop.for %j = %c0 to %1 step %c1 {

327 template <typename IndexedValueType>
328 class LinalgScopedEmitter<IndexedValueType, IndexedGenericOp> {
329 public:
330  static void emitScalarImplementation(ArrayRef<Value> allIvs,
331  IndexedGenericOp indexedGenericOp) {
332  assert(indexedGenericOp.hasBufferSemantics() &&
333  "expected linalg op with buffer semantics");
334  auto b = ScopedContext::getBuilder();
335  auto loc = ScopedContext::getLocation();
336  using edsc::intrinsics::detail::ValueHandleArray;
337  unsigned nInputs = indexedGenericOp.getNumInputs();
338  unsigned nOutputs = indexedGenericOp.getNumOutputs();
339  unsigned nLoops = allIvs.size();
340  SmallVector<Value, 4> indexedValues(nLoops + nInputs + nOutputs);
341 
342  for (unsigned i = 0; i < nLoops; ++i) {
343  indexedValues[i] = allIvs[i];
344  }
345 
346  // 1.a. Emit std_load from input views.
347  for (unsigned i = 0; i < nInputs; ++i) {
348  ValueHandleArray indexing(makeCanonicalAffineApplies(
349  b, loc, indexedGenericOp.getInputIndexingMap(i), allIvs));
350  indexedValues[nLoops + i] =
351  std_load(indexedGenericOp.getInput(i), indexing);
352  }
353 
354  // 1.b. Emit std_load from output views.
355  for (unsigned i = 0; i < nOutputs; ++i) {
356  ValueHandleArray indexing(makeCanonicalAffineApplies(
357  b, loc, indexedGenericOp.getOutputIndexingMap(i), allIvs));
358  indexedValues[nLoops + nInputs + i] =
359  std_load(indexedGenericOp.getOutputBuffer(i), indexing);
360  }
361 
362  if (auto funcOp = indexedGenericOp.getFunction()) {
363  // 2. Emit call.
364  Operation *callOp = call(funcOp, indexedValues);
365  assert(callOp->getNumResults() == indexedGenericOp.getNumOutputs());
366 
367  // 3. Emit std_store.
368  for (unsigned i = 0; i < nOutputs; ++i) {
369  ValueHandleArray indexing(makeCanonicalAffineApplies(
370  b, loc, indexedGenericOp.getOutputIndexingMap(i), allIvs));
371  std_store(callOp->getResult(i), indexedGenericOp.getOutputBuffer(i),
372  indexing);
373  }
374  return;
375  }
376  // TODO(ntv): When a region inliner exists, use it.
377  // 2. Inline region, currently only works for a single basic block.
378  BlockAndValueMapping map;
379  auto &block = indexedGenericOp.region().front();
380  for (auto it : llvm::zip(block.getArguments(), indexedValues))
381  map.map(std::get<0>(it), std::get<1>(it));
382  for (auto &op : block.without_terminator()) {
383  assert(op.getNumRegions() == 0);
384  auto *newOp = b.clone(op, map);
385  for (auto it : llvm::zip(op.getResults(), newOp->getResults()))
386  map.map(std::get<0>(it), std::get<1>(it));
387  }
388 
389  // 3. Emit std_store.
390  auto *yieldOp = cast<YieldOp>(block.back()).getOperation();
391  assert(yieldOp->getNumOperands() == nOutputs);
392  for (unsigned i = 0; i < nOutputs; ++i) {
393  ValueHandleArray indexing(makeCanonicalAffineApplies(
394  b, loc, indexedGenericOp.getOutputIndexingMap(i), allIvs));
395  std_store(map.lookup(yieldOp->getOperand(i)),
396  indexedGenericOp.getOutputBuffer(i), indexing);
397  }
398  }

401 // This struct is for factoring out the implementation and support template
402 // instantiations in the following 2 cases:
403 // 1. Appending to a list of patterns via RewritePatternList.
404 // 2. Direct invocation via `linalgOpToLoops` and `linalgOpToAffineLoops`.
405 // The implementation must work both in DRR and inside a RewritePattern. As a
406 // consequence, (1) it is only allowed to emit new ops if the match is
407 // guaranteed to be a success, (2) it is not allowed erase/replace, and (3) an
408 // encompassing pattern must take care of the erasure logic.
409 template <typename LoopTy, typename IndexedValueTy, typename ConcreteOpTy>
410 class LinalgOpToLoopsImpl {
411 public:
412  static LogicalResult doit(Operation *op, PatternRewriter &rewriter);
413 };
414 
415 template <typename LoopTy, typename IndexedValueTy, typename ConcreteOpTy>
416 LogicalResult LinalgOpToLoopsImpl<LoopTy, IndexedValueTy, ConcreteOpTy>::doit(
417  Operation *op, PatternRewriter &rewriter) {
418  OpBuilder b(op);
419  ScopedContext scope(b, op->getLoc());
420 
421  // The flattened loopToOperandRangesMaps is expected to be an invertible
422  // permutation map (which is asserted in the inverse calculation).
423  auto linalgOp = cast<ConcreteOpTy>(op);
424  assert(linalgOp.hasBufferSemantics() &&
425  "expected linalg op with buffer semantics");
426  auto invertedMap =
427  inversePermutation(concatAffineMaps(loopToOperandRangesMaps(linalgOp)));
428  if (!invertedMap) {
429  LinalgScopedEmitter<IndexedValueTy, ConcreteOpTy>::emitScalarImplementation(
430  {}, linalgOp);
431  return success();
432  }
433 
434  auto nPar = linalgOp.getNumParallelLoops();
435  auto nRed = linalgOp.getNumReductionLoops();
436  auto nWin = linalgOp.getNumWindowLoops();
437  SmallVector<IndexHandle, 4> allIvs(nPar + nRed + nWin);
438  SmallVector<ValueHandle *, 4> allPIvs =
439  makeHandlePointers(MutableArrayRef<IndexHandle>(allIvs));
440  auto loopRanges = emitLoopRanges(scope.getBuilder(), scope.getLocation(),
441  invertedMap, getViewSizes(b, linalgOp));
442  assert(loopRanges.size() == allIvs.size());
443 
444  GenericLoopNestRangeBuilder<LoopTy>(allPIvs, loopRanges)([&] {
445  auto allIvValues = extractValues(allIvs);
446  LinalgScopedEmitter<IndexedValueTy, ConcreteOpTy>::emitScalarImplementation(
447  allIvValues, linalgOp);
448  });
449  return success();
450 }

508 // Local folding pattern for AffineApplyOp that we can apply greedily.
509 // This replaces AffineApplyOp by the proper value in cases where the associated
510 // map is trivial. A trivial map here is defined as a map with a single result
511 // and either:
512 // 1. Zero operand + returns a single AffineConstantExpr
513 // 2. One operand + returns a single AffineDimExpr
514 // 3. One operands + returns a single AffineSymbolExpr
515 //
516 // In the first case, the AffineApplyOp is replaced by a new constant. In the
517 // other cases, it is replaced by its unique operand.
518 struct FoldAffineOp : public RewritePattern {
519  FoldAffineOp(MLIRContext *context)
520  : RewritePattern(AffineApplyOp::getOperationName(), 0, context) {}
521 
522  PatternMatchResult matchAndRewrite(Operation *op,
523  PatternRewriter &rewriter) const override {
524  AffineApplyOp affineApplyOp = cast<AffineApplyOp>(op);
525  auto map = affineApplyOp.getAffineMap();
526  if (map.getNumResults() != 1 || map.getNumInputs() > 1)
527  return matchFailure();
528 
529  AffineExpr expr = map.getResult(0);
530  if (map.getNumInputs() == 0) {
531  if (auto val = expr.dyn_cast<AffineConstantExpr>()) {
532  rewriter.replaceOpWithNewOp<ConstantIndexOp>(op, val.getValue());
533  return matchSuccess();
534  }
535  return matchFailure();
536  }
537  if (expr.dyn_cast<AffineDimExpr>() || expr.dyn_cast<AffineSymbolExpr>()) {
538  rewriter.replaceOp(op, op->getOperand(0));
539  return matchSuccess();
540  }
541  return matchFailure();
542  }
543 };
544 } // namespace

546 template <typename LoopType, typename IndexedValueType>
547 void LowerLinalgToLoopsPass<LoopType, IndexedValueType>::runOnFunction() {
548  auto *context = &this->getContext();
549  OwningRewritePatternList patterns;
550  // Canonicalization and folding patterns applied greedily allow cleaning up
551  // the emitted IR on the fly.
552  // TODO(ntv) fold view and subview ops?
553  FillRewritePatterns<LoopType, IndexedValueType>(patterns, context);
554  DimOp::getCanonicalizationPatterns(patterns, context);
555  AffineApplyOp::getCanonicalizationPatterns(patterns, context);
556  patterns.insert<FoldAffineOp>(context);
557  // Just apply the patterns greedily.
558  applyPatternsGreedily(this->getFunction(), patterns);
559 }
