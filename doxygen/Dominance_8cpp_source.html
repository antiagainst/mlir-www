29 template <bool IsPostDom>
30 void DominanceInfoBase<IsPostDom>::recalculate(Operation *op) {
31  dominanceInfos.clear();
32 
33  /// Build the dominance for each of the operation regions.
34  op->walk([&](Operation *op) {
35  for (auto &region : op->getRegions()) {
36  // Don't compute dominance if the region is empty.
37  if (region.empty())
38  continue;
39  auto opDominance = std::make_unique<base>();
40  opDominance->recalculate(region);
41  dominanceInfos.try_emplace(&region, std::move(opDominance));
42  }
43  });
44 }

46 /// Return true if the specified block A properly dominates block B.
47 template <bool IsPostDom>
48 bool DominanceInfoBase<IsPostDom>::properlyDominates(Block *a, Block *b) {
49  // A block dominates itself but does not properly dominate itself.
50  if (a == b)
51  return false;
52 
53  // If either a or b are null, then conservatively return false.
54  if (!a || !b)
55  return false;
56 
57  // If both blocks are not in the same region, 'a' properly dominates 'b' if
58  // 'b' is defined in an operation region that (recursively) ends up being
59  // dominated by 'a'. Walk up the list of containers enclosing B.
60  auto *regionA = a->getParent(), *regionB = b->getParent();
61  if (regionA != regionB) {
62  Operation *bAncestor;
63  do {
64  bAncestor = regionB->getParentOp();
65  // If 'bAncestor' is the top level region, then 'a' is a block that post
66  // dominates 'b'.
67  if (!bAncestor || !bAncestor->getBlock())
68  return IsPostDom;
69 
70  regionB = bAncestor->getBlock()->getParent();
71  } while (regionA != regionB);
72 
73  // Check to see if the ancestor of 'b' is the same block as 'a'.
74  b = bAncestor->getBlock();
75  if (a == b)
76  return true;
77  }
78 
79  // Otherwise, use the standard dominance functionality.
80 
81  // If we don't have a dominance information for this region, assume that b is
82  // dominated by anything.
83  auto baseInfoIt = dominanceInfos.find(regionA);
84  if (baseInfoIt == dominanceInfos.end())
85  return true;
86  return baseInfoIt->second->properlyDominates(a, b);
87 }

96 /// Return true if operation A properly dominates operation B.
97 bool DominanceInfo::properlyDominates(Operation *a, Operation *b) {
98  auto *aBlock = a->getBlock(), *bBlock = b->getBlock();
99 
100  // If a or b are not within a block, then a does not dominate b.
101  if (!aBlock || !bBlock)
102  return false;
103 
104  // If the blocks are the same, then check if b is before a in the block.
105  if (aBlock == bBlock)
106  return a->isBeforeInBlock(b);
107 
108  // Traverse up b's hierarchy to check if b's block is contained in a's.
109  if (auto *bAncestor = aBlock->findAncestorOpInBlock(*b)) {
110  // Since we already know that aBlock != bBlock, here bAncestor != b.
111  // a and bAncestor are in the same block; check if 'a' dominates
112  // bAncestor.
113  return dominates(a, bAncestor);
114  }
115 
116  // If the blocks are different, check if a's block dominates b's.
117  return properlyDominates(aBlock, bBlock);
118 }

120 /// Return true if value A properly dominates operation B.
121 bool DominanceInfo::properlyDominates(Value a, Operation *b) {
122  if (auto *aOp = a.getDefiningOp()) {
123  // The values defined by an operation do *not* dominate any nested
124  // operations.
125  if (aOp->getParentRegion() != b->getParentRegion() && aOp->isAncestor(b))
126  return false;
127  return properlyDominates(aOp, b);
128  }
129 
130  // block arguments properly dominate all operations in their own block, so
131  // we use a dominates check here, not a properlyDominates check.
132  return dominates(a.cast<BlockArgument>().getOwner(), b->getBlock());
133 }
