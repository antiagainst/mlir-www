36 struct FxpMathTargetConfigImpl : public FxpMathTargetConfig {
37  FxpMathTargetConfigImpl(SolverContext &context)
38  : FxpMathTargetConfig(context) {
39  Builder b(&context.getMlirContext());
40  IntegerType i8Type = b.getIntegerType(8);
41  IntegerType i16Type = b.getIntegerType(16);
42  IntegerType i32Type = b.getIntegerType(32);
43 
44  q8 = addCandidateType(
45  AnyQuantizedType::get(QuantizationFlags::Signed, i8Type, nullptr,
46  std::numeric_limits<int8_t>::min(),
47  std::numeric_limits<int8_t>::max()),
48  CandidateQuantizedType::Scheme::UniformPerLayer);
49  q16 = addCandidateType(
50  AnyQuantizedType::get(QuantizationFlags::Signed, i16Type, nullptr,
51  std::numeric_limits<int16_t>::min(),
52  std::numeric_limits<int16_t>::max()),
53  CandidateQuantizedType::Scheme::UniformPerLayer);
54  q32ExplicitFixedPoint = addCandidateType(
55  AnyQuantizedType::get(QuantizationFlags::Signed, i32Type, nullptr,
56  std::numeric_limits<int32_t>::min(),
57  std::numeric_limits<int32_t>::max()),
58  CandidateQuantizedType::Scheme::UniformExplicitFixedPointScale);
59 
60  // Op handlers.
61  addOpHandler<ConstantOp>(
62  std::bind(&FxpMathTargetConfigImpl::handleConstant, this, _1, _2));
63  addOpHandler<ReturnOp>(
64  std::bind(&FxpMathTargetConfigImpl::handleTerminal, this, _1, _2));
65  addOpHandler<quant::StatisticsOp>(
66  std::bind(&FxpMathTargetConfigImpl::handleStats, this, _1, _2));

68  // FxpMathOps.

87  bool isHandledType(Type t) const final {
88  if (t.isa<FloatType>())
89  return true;
90  return (t.isa<VectorType>() || t.isa<TensorType>()) &&
91  t.cast<ShapedType>().getElementType().isa<FloatType>();
92  }

107  void handleValueIdentity(Operation *op, CAGSlice &cag) const {
108  assert(op->getNumResults() == 1);
109  if (!isHandledType(op->getResult(0).getType()))
110  return;
111 
112  auto resultNode = cag.getResultAnchor(op, 0);
113  resultNode->setTypeTransformRule(
114  CAGAnchorNode::TypeTransformRule::DirectStorage);
115 
116  for (unsigned opIdx = 0, e = op->getNumOperands(); opIdx < e; ++opIdx) {
117  if (!isHandledType(op->getOperand(opIdx).getType()))
118  continue;
119  auto operandNode = cag.getOperandAnchor(op, opIdx);
120  operandNode->setTypeTransformRule(
121  CAGAnchorNode::TypeTransformRule::DirectStorage);
122  UniformConstraintsBuilder(cag).coupleAnchors(operandNode, resultNode);
123  }
124  }

126  void handleConstant(Operation *op, CAGSlice &cag) const {
127  if (!isHandledType(op->getResult(0).getType()))
128  return;
129 
130  auto resultNode = cag.getResultAnchor(op, 0);
131  resultNode->setTypeTransformRule(
132  CAGAnchorNode::TypeTransformRule::ExpressedOnly);
133  Attribute valueAttr;
134  if (!matchPattern(op, m_Constant(&valueAttr))) {
135  return;
136  }
137 
138  AttributeTensorStatistics stats(valueAttr);
139  TensorAxisStatistics layerStats;
140  if (!stats.get(layerStats)) {
141  op->emitOpError("could not compute statistics");
142  return;
143  }
144 
145  UniformConstraintsBuilder(cag).applyStats(resultNode, layerStats);
146  }

156  void handleStats(Operation *op, CAGSlice &cag) const {
157  if (!isHandledType(op->getResult(0).getType()))
158  return;
159 
160  auto argNode = cag.getOperandAnchor(op, 0);
161  auto resultNode = cag.getResultAnchor(op, 0);
162  UniformConstraintsBuilder(cag).coupleAnchors(argNode, resultNode);
163 
164  TensorAxisStatistics layerStats;
165  auto statsOp = cast<quant::StatisticsOp>(op);
166  auto layerStatsAttr = statsOp.layerStats();
167  layerStats.minValue =
168  layerStatsAttr.getValue<FloatAttr>(0).getValueAsDouble();
169  layerStats.maxValue =
170  layerStatsAttr.getValue<FloatAttr>(1).getValueAsDouble();
171  UniformConstraintsBuilder(cag).applyStats(resultNode, layerStats);
172  }

174  void handleAdd(Operation *op, CAGSlice &cag) const {
175  if (!isHandledType(op->getResult(0).getType()))
176  return;
177 
178  auto lhs = cag.getOperandAnchor(op, 0);
179  auto rhs = cag.getOperandAnchor(op, 1);
180  auto resultNode = cag.getResultAnchor(op, 0);
181  // Add supports 8/16 bit math.
182  llvm::SmallBitVector disableMask =
183  getCandidateTypeDisabledExceptMask({q8, q16});
184  lhs->getUniformMetadata().disabledCandidateTypes = disableMask;
185  rhs->getUniformMetadata().disabledCandidateTypes = disableMask;
186  resultNode->getUniformMetadata().disabledCandidateTypes = disableMask;
187  // NOTE: We couple the add such that the scale/zeroPoint match between
188  // both args and the result. This is overly constrained in that it is
189  // possible to write efficient add kernels with a bit more freedom (i.e.
190  // zeroPoints can vary, scales can differ by a power of two, etc).
191  // However, fully coupled yields the simples solutions on the fast path.
192  // Further efficiency can be had by constraining the zeroPoint to 0, but
193  // there isn't a constraint for this yet (and there are tradeoffs).
194  UniformConstraintsBuilder(cag).coupleAnchors(lhs, resultNode);
195  UniformConstraintsBuilder(cag).coupleAnchors(rhs, resultNode);
196  addRealMathOptionalConstraints(op, resultNode, cag);
197  }

231  void handleMatMulBias(Operation *op, CAGSlice &cag) const {
232  if (!isHandledType(op->getResult(0).getType()))
233  return;
234 
235  auto lhs = cag.getOperandAnchor(op, 0);
236  auto rhs = cag.getOperandAnchor(op, 1);
237  auto bias = cag.getOperandAnchor(op, 2);
238  bias->getUniformMetadata().disabledCandidateTypes =
239  getCandidateTypeDisabledExceptMask({q32ExplicitFixedPoint});
240 
241  auto resultNode = cag.getResultAnchor(op, 0);
242  UniformConstraintsBuilder(cag).propagateExplicitScale(resultNode, bias);
243 
244  // Mul supports 8/16 bit math.
245  llvm::SmallBitVector disableMask =
246  getCandidateTypeDisabledExceptMask({q8, q16});
247  lhs->getUniformMetadata().disabledCandidateTypes = disableMask;
248  rhs->getUniformMetadata().disabledCandidateTypes = disableMask;
249  resultNode->getUniformMetadata().disabledCandidateTypes = disableMask;
250  addRealMathOptionalConstraints(op, resultNode, cag);

253  void addRealMathOptionalConstraints(Operation *op, CAGAnchorNode *anchor,
254  CAGSlice &cag) const {
255  // TODO: It would be nice if these all extended some base trait instead
256  // of requiring name lookup.
257  auto clampMinAttr = op->getAttrOfType<FloatAttr>("clamp_min");
258  auto clampMaxAttr = op->getAttrOfType<FloatAttr>("clamp_max");
259 
260  if (clampMinAttr || clampMaxAttr) {
261  auto nan = APFloat::getQNaN(APFloat::IEEEdouble());
262  auto clampMin = clampMinAttr ? clampMinAttr.getValue() : nan;
263  auto clampMax = clampMaxAttr ? clampMaxAttr.getValue() : nan;
264  UniformConstraintsBuilder(cag).clamp(anchor, clampMin, clampMax);
265  }
266  }
267 
268  unsigned q8;
269  unsigned q16;
270  unsigned q32ExplicitFixedPoint;
