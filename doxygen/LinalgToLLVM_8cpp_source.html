81 // Convert the given type to the LLVM IR Dialect type. The following
82 // conversions are supported:
83 // - an Index type is converted into an LLVM integer type with pointer
84 // bitwidth (analogous to intptr_t in C);
85 // - an Integer type is converted into an LLVM integer type of the same width;
86 // - an F32 type is converted into an LLVM float type
87 // - a Buffer, Range or View is converted into an LLVM structure type
88 // containing the respective dynamic values.
89 static Type convertLinalgType(Type t, LLVMTypeConverter &lowering) {
90  auto *context = t.getContext();
91  auto int64Ty = lowering.convertType(IntegerType::get(64, context))
92  .cast<LLVM::LLVMType>();
93 
94  // Range descriptor contains the range bounds and the step as 64-bit integers.
95  //
96  // struct {
97  // int64_t min;
98  // int64_t max;
99  // int64_t step;
100  // };
101  if (t.isa<RangeType>())
102  return LLVMType::getStructTy(int64Ty, int64Ty, int64Ty);
103 
104  return Type();
105 }

143 // RangeOp creates a new range descriptor.
144 class RangeOpConversion : public LLVMOpLowering {
145 public:
146  explicit RangeOpConversion(MLIRContext *context, LLVMTypeConverter &lowering_)
147  : LLVMOpLowering(RangeOp::getOperationName(), context, lowering_) {}
148 
149  PatternMatchResult
150  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
151  ConversionPatternRewriter &rewriter) const override {
152  auto rangeOp = cast<RangeOp>(op);
153  auto rangeDescriptorTy =
154  convertLinalgType(rangeOp.getResult().getType(), lowering);
155 
156  edsc::ScopedContext context(rewriter, op->getLoc());
157 
158  // Fill in an aggregate value of the descriptor.
159  RangeOpOperandAdaptor adaptor(operands);
160  Value desc = llvm_undef(rangeDescriptorTy);
161  desc = insertvalue(desc, adaptor.min(), rewriter.getI64ArrayAttr(0));
162  desc = insertvalue(desc, adaptor.max(), rewriter.getI64ArrayAttr(1));
163  desc = insertvalue(desc, adaptor.step(), rewriter.getI64ArrayAttr(2));
164  rewriter.replaceOp(op, desc);
165  return matchSuccess();
166  }

172 class ReshapeOpConversion : public LLVMOpLowering {
173 public:

175  LLVMTypeConverter &lowering_)
176  : LLVMOpLowering(ReshapeOp::getOperationName(), context, lowering_) {}
177 
178  PatternMatchResult
179  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
180  ConversionPatternRewriter &rewriter) const override {
181  auto reshapeOp = cast<ReshapeOp>(op);
182  MemRefType dstType = reshapeOp.getResult().getType().cast<MemRefType>();
183 
184  if (!dstType.hasStaticShape())
185  return matchFailure();
186 
187  int64_t offset;
188  SmallVector<int64_t, 4> strides;
189  auto res = getStridesAndOffset(dstType, strides, offset);
190  if (failed(res) || llvm::any_of(strides, [](int64_t val) {
191  return ShapedType::isDynamicStrideOrOffset(val);
192  }))
193  return matchFailure();
194 
195  edsc::ScopedContext context(rewriter, op->getLoc());
196  ReshapeOpOperandAdaptor adaptor(operands);
197  BaseViewConversionHelper baseDesc(adaptor.view());
198  BaseViewConversionHelper desc(lowering.convertType(dstType));
199  desc.setAllocatedPtr(baseDesc.allocatedPtr());
200  desc.setAlignedPtr(baseDesc.alignedPtr());
201  desc.setOffset(baseDesc.offset());
202  for (auto en : llvm::enumerate(dstType.getShape()))
203  desc.setConstantSize(en.index(), en.value());
204  for (auto en : llvm::enumerate(strides))
205  desc.setConstantStride(en.index(), en.value());

211 /// Conversion pattern that transforms a linalg.slice op into:
212 /// 1. A function entry `alloca` operation to allocate a ViewDescriptor.
213 /// 2. A load of the ViewDescriptor from the pointer allocated in 1.
214 /// 3. Updates to the ViewDescriptor to introduce the data ptr, offset, size
215 /// and stride corresponding to the region of memory within the bounds of
216 /// the parent view.
217 /// 4. A store of the resulting ViewDescriptor to the alloca'ed pointer.
218 /// The linalg.slice op is replaced by the alloca'ed pointer.
219 class SliceOpConversion : public LLVMOpLowering {
220 public:
221  explicit SliceOpConversion(MLIRContext *context, LLVMTypeConverter &lowering_)
222  : LLVMOpLowering(SliceOp::getOperationName(), context, lowering_) {}
223 
224  PatternMatchResult
225  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
226  ConversionPatternRewriter &rewriter) const override {
227  edsc::ScopedContext context(rewriter, op->getLoc());
228  SliceOpOperandAdaptor adaptor(operands);
229  BaseViewConversionHelper baseDesc(adaptor.view());
230 
231  auto sliceOp = cast<SliceOp>(op);
232  auto memRefType = sliceOp.getBaseViewType();
233  auto int64Ty = lowering.convertType(rewriter.getIntegerType(64))
234  .cast<LLVM::LLVMType>();
235 
236  BaseViewConversionHelper desc(
237  lowering.convertType(sliceOp.getShapedType()));
238 
239  // TODO(ntv): extract sizes and emit asserts.
240  SmallVector<Value, 4> strides(memRefType.getRank());
241  for (int i = 0, e = memRefType.getRank(); i < e; ++i)
242  strides[i] = baseDesc.stride(i);
243 
244  auto pos = [&rewriter](ArrayRef<int64_t> values) {
245  return rewriter.getI64ArrayAttr(values);
246  };
247 
248  // Compute base offset.
249  Value baseOffset = baseDesc.offset();
250  for (int i = 0, e = memRefType.getRank(); i < e; ++i) {
251  Value indexing = adaptor.indexings()[i];
252  Value min = indexing;
253  if (sliceOp.indexing(i).getType().isa<RangeType>())
254  min = extractvalue(int64Ty, indexing, pos(0));
255  baseOffset = add(baseOffset, mul(min, strides[i]));
256  }
257 
258  // Insert the base and aligned pointers.
259  desc.setAllocatedPtr(baseDesc.allocatedPtr());
260  desc.setAlignedPtr(baseDesc.alignedPtr());
261 
262  // Insert base offset.
263  desc.setOffset(baseOffset);
264 
265  // Corner case, no sizes or strides: early return the descriptor.
266  if (sliceOp.getShapedType().getRank() == 0)
267  return rewriter.replaceOp(op, {desc}), matchSuccess();
268 
269  Value zero =
270  constant(int64Ty, rewriter.getIntegerAttr(rewriter.getIndexType(), 0));
271  // Compute and insert view sizes (max - min along the range) and strides.
272  // Skip the non-range operands as they will be projected away from the view.
273  int numNewDims = 0;
274  for (auto en : llvm::enumerate(sliceOp.indexings())) {
275  Value indexing = en.value();
276  if (indexing.getType().isa<RangeType>()) {
277  int rank = en.index();
278  Value rangeDescriptor = adaptor.indexings()[rank];
279  Value min = extractvalue(int64Ty, rangeDescriptor, pos(0));
280  Value max = extractvalue(int64Ty, rangeDescriptor, pos(1));
281  Value step = extractvalue(int64Ty, rangeDescriptor, pos(2));
282  Value baseSize = baseDesc.size(rank);
283 
284  // Bound upper by base view upper bound.
285  max = llvm_select(llvm_icmp(ICmpPredicate::slt, max, baseSize), max,
286  baseSize);
287  Value size = sub(max, min);
288  // Bound lower by zero.
289  size =
290  llvm_select(llvm_icmp(ICmpPredicate::slt, size, zero), zero, size);
291  Value stride = mul(strides[rank], step);
292  desc.setSize(numNewDims, size);
293  desc.setStride(numNewDims, stride);
294  ++numNewDims;
295  }
296  }
297 
298  rewriter.replaceOp(op, {desc});
299  return matchSuccess();
300  }
301 };

303 /// Conversion pattern that transforms a linalg.transpose op into:
304 /// 1. A function entry `alloca` operation to allocate a ViewDescriptor.
305 /// 2. A load of the ViewDescriptor from the pointer allocated in 1.
306 /// 3. Updates to the ViewDescriptor to introduce the data ptr, offset, size
307 /// and stride. Size and stride are permutations of the original values.
308 /// 4. A store of the resulting ViewDescriptor to the alloca'ed pointer.
309 /// The linalg.transpose op is replaced by the alloca'ed pointer.
310 class TransposeOpConversion : public LLVMOpLowering {
311 public:
312  explicit TransposeOpConversion(MLIRContext *context,
313  LLVMTypeConverter &lowering_)
314  : LLVMOpLowering(TransposeOp::getOperationName(), context, lowering_) {}
315 
316  PatternMatchResult
317  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
318  ConversionPatternRewriter &rewriter) const override {
319  // Initialize the common boilerplate and alloca at the top of the FuncOp.
320  edsc::ScopedContext context(rewriter, op->getLoc());
321  TransposeOpOperandAdaptor adaptor(operands);
322  BaseViewConversionHelper baseDesc(adaptor.view());
323 
324  auto transposeOp = cast<TransposeOp>(op);
325  // No permutation, early exit.
326  if (transposeOp.permutation().isIdentity())
327  return rewriter.replaceOp(op, {baseDesc}), matchSuccess();
328 
329  BaseViewConversionHelper desc(
330  lowering.convertType(transposeOp.getShapedType()));
331 
332  // Copy the base and aligned pointers from the old descriptor to the new
333  // one.
334  desc.setAllocatedPtr(baseDesc.allocatedPtr());
335  desc.setAlignedPtr(baseDesc.alignedPtr());
336 
337  // Copy the offset pointer from the old descriptor to the new one.
338  desc.setOffset(baseDesc.offset());
339 
340  // Iterate over the dimensions and apply size/stride permutation.
341  for (auto en : llvm::enumerate(transposeOp.permutation().getResults())) {
342  int sourcePos = en.index();
343  int targetPos = en.value().cast<AffineDimExpr>().getPosition();
344  desc.setSize(targetPos, baseDesc.size(sourcePos));
345  desc.setStride(targetPos, baseDesc.stride(sourcePos));
346  }
347 
348  rewriter.replaceOp(op, {desc});
349  return matchSuccess();
350  }
351 };

390 // Get a SymbolRefAttr containing the library function name for the LinalgOp.
391 // If the library function does not exist, insert a declaration.
392 template <typename LinalgOp>
393 static FlatSymbolRefAttr getLibraryCallSymbolRef(Operation *op,
394  PatternRewriter &rewriter) {
395  auto linalgOp = cast<LinalgOp>(op);
396  auto fnName = linalgOp.getLibraryCallName();
397  if (fnName.empty()) {
398  op->emitWarning("No library call defined for: ") << *op;
399  return {};
400  }
401 
402  // fnName is a dynamic std::String, unique it via a SymbolRefAttr.
403  FlatSymbolRefAttr fnNameAttr = rewriter.getSymbolRefAttr(fnName);
404  auto module = op->getParentOfType<ModuleOp>();
405  if (module.lookupSymbol(fnName)) {
406  return fnNameAttr;
407  }
408 
409  SmallVector<Type, 4> inputTypes(ExtractOperandTypes<LinalgOp>(op));
410  assert(op->getNumResults() == 0 &&
411  "Library call for linalg operation can be generated only for ops that "
412  "have void return types");
413  auto libFnType = FunctionType::get(inputTypes, {}, rewriter.getContext());
414 
415  OpBuilder::InsertionGuard guard(rewriter);
416  // Insert before module terminator.
417  rewriter.setInsertionPoint(module.getBody(),
418  std::prev(module.getBody()->end()));
419  rewriter.create<FuncOp>(op->getLoc(), fnNameAttr.getValue(), libFnType,
420  ArrayRef<NamedAttribute>{});
421  return fnNameAttr;
422 }

512 /// A non-conversion rewrite pattern kicks in to convert CopyOp with
513 /// permutations into a sequence of TransposeOp and permutation-free CopyOp.
514 /// This interplays together with TransposeOpConversion and
515 /// LinalgConversion<CopyOp> to create a path to the LLVM dialect.
516 class CopyTransposeConversion : public OpRewritePattern<CopyOp> {
517 public:
518  using OpRewritePattern<CopyOp>::OpRewritePattern;
519 
520  PatternMatchResult matchAndRewrite(CopyOp op,
521  PatternRewriter &rewriter) const override {
522  Value in = op.input(), out = op.output();
523 
524  // If either inputPerm or outputPerm are non-identities, insert transposes.
525  auto inputPerm = op.inputPermutation();
526  if (inputPerm.hasValue() && !inputPerm->isIdentity())
527  in = rewriter.create<linalg::TransposeOp>(op.getLoc(), in,
528  AffineMapAttr::get(*inputPerm));
529  auto outputPerm = op.outputPermutation();
530  if (outputPerm.hasValue() && !outputPerm->isIdentity())
531  out = rewriter.create<linalg::TransposeOp>(
532  op.getLoc(), out, AffineMapAttr::get(*outputPerm));
533 
534  // If nothing was transposed, fail and let the conversion kick in.
535  if (in == op.input() && out == op.output())
536  return matchFailure();
537 
538  rewriter.replaceOpWithNewOp<CopyOp>(op, in, out);
539  return matchSuccess();
540  }

573 void ConvertLinalgToLLVMPass::runOnModule() {
574  auto module = getModule();
575 
576  // Convert to the LLVM IR dialect using the converter defined above.
577  OwningRewritePatternList patterns;
578  LinalgTypeConverter converter(&getContext());
579  populateAffineToStdConversionPatterns(patterns, &getContext());
580  populateLoopToStdConversionPatterns(patterns, &getContext());
581  populateStdToLLVMConversionPatterns(converter, patterns);
582  populateVectorToLLVMConversionPatterns(converter, patterns);
583  populateLinalgToStandardConversionPatterns(patterns, &getContext());
584  populateLinalgToLLVMConversionPatterns(converter, patterns, &getContext());
585 
586  ConversionTarget target(getContext());
587  target.addLegalDialect<LLVM::LLVMDialect>();
588  target.addDynamicallyLegalOp<FuncOp>(
589  [&](FuncOp op) { return converter.isSignatureLegal(op.getType()); });
590  target.addLegalOp<ModuleOp, ModuleTerminatorOp>();
591  if (failed(applyFullConversion(module, target, patterns, &converter)))
592  signalPassFailure();
593 }
