54 /// Loop unrolling pass. Unrolls all innermost loops unless full unrolling and a
55 /// full unroll threshold was specified, in which case, fully unrolls all loops
56 /// with trip count less than the specified threshold. The latter is for testing
57 /// purposes, especially for testing outer loop unrolling.
58 struct LoopUnroll : public FunctionPass<LoopUnroll> {
59  const Optional<unsigned> unrollFactor;
60  const Optional<bool> unrollFull;
61  // Callback to obtain unroll factors; if this has a callable target, takes
62  // precedence over command-line argument or passed argument.
63  const std::function<unsigned(AffineForOp)> getUnrollFactor;
64 
65  explicit LoopUnroll(
66  Optional<unsigned> unrollFactor = None, Optional<bool> unrollFull = None,
67  const std::function<unsigned(AffineForOp)> &getUnrollFactor = nullptr)
68  : unrollFactor(unrollFactor), unrollFull(unrollFull),
69  getUnrollFactor(getUnrollFactor) {}
70 
71  void runOnFunction() override;
72 
73  /// Unroll this for op. Returns failure if nothing was done.
74  LogicalResult runOnAffineForOp(AffineForOp forOp);
75 
76  static const unsigned kDefaultUnrollFactor = 4;
77 };
78 } // end anonymous namespace

80 void LoopUnroll::runOnFunction() {
81  // Gathers all innermost loops through a post order pruned walk.
82  struct InnermostLoopGatherer {
83  // Store innermost loops as we walk.
84  std::vector<AffineForOp> loops;
85 
86  void walkPostOrder(FuncOp f) {
87  for (auto &b : f)
88  walkPostOrder(b.begin(), b.end());
89  }
90 
91  bool walkPostOrder(Block::iterator Start, Block::iterator End) {
92  bool hasInnerLoops = false;
93  // We need to walk all elements since all innermost loops need to be
94  // gathered as opposed to determining whether this list has any inner
95  // loops or not.
96  while (Start != End)
97  hasInnerLoops |= walkPostOrder(&(*Start++));
98  return hasInnerLoops;
99  }
100  bool walkPostOrder(Operation *opInst) {
101  bool hasInnerLoops = false;
102  for (auto &region : opInst->getRegions())
103  for (auto &block : region)
104  hasInnerLoops |= walkPostOrder(block.begin(), block.end());
105  if (isa<AffineForOp>(opInst)) {
106  if (!hasInnerLoops)
107  loops.push_back(cast<AffineForOp>(opInst));
108  return true;
109  }
110  return hasInnerLoops;
111  }
112  };
113 
114  if (clUnrollFull.getNumOccurrences() > 0 &&
115  clUnrollFullThreshold.getNumOccurrences() > 0) {
116  // Store short loops as we walk.
117  std::vector<AffineForOp> loops;
118 
119  // Gathers all loops with trip count <= minTripCount. Do a post order walk
120  // so that loops are gathered from innermost to outermost (or else unrolling
121  // an outer one may delete gathered inner ones).
122  getFunction().walk([&](AffineForOp forOp) {
123  Optional<uint64_t> tripCount = getConstantTripCount(forOp);
124  if (tripCount.hasValue() && tripCount.getValue() <= clUnrollFullThreshold)
125  loops.push_back(forOp);
126  });
127  for (auto forOp : loops)
128  loopUnrollFull(forOp);
129  return;
130  }
131 
132  unsigned numRepetitions = clUnrollNumRepetitions.getNumOccurrences() > 0
133  ? clUnrollNumRepetitions
134  : 1;
135  // If the call back is provided, we will recurse until no loops are found.
136  FuncOp func = getFunction();
137  for (unsigned i = 0; i < numRepetitions || getUnrollFactor; i++) {
138  InnermostLoopGatherer ilg;
139  ilg.walkPostOrder(func);
140  auto &loops = ilg.loops;
141  if (loops.empty())
142  break;
143  bool unrolled = false;
144  for (auto forOp : loops)
145  unrolled |= succeeded(runOnAffineForOp(forOp));
146  if (!unrolled)
147  // Break out if nothing was unrolled.
148  break;
149  }
150 }

152 /// Unrolls a 'affine.for' op. Returns success if the loop was unrolled,
153 /// failure otherwise. The default unroll factor is 4.
154 LogicalResult LoopUnroll::runOnAffineForOp(AffineForOp forOp) {
155  // Use the function callback if one was provided.
156  if (getUnrollFactor) {
157  return loopUnrollByFactor(forOp, getUnrollFactor(forOp));
158  }
159  // Unroll by the factor passed, if any.
160  if (unrollFactor.hasValue())
161  return loopUnrollByFactor(forOp, unrollFactor.getValue());
162  // Unroll by the command line factor if one was specified.
163  if (clUnrollFactor.getNumOccurrences() > 0)
164  return loopUnrollByFactor(forOp, clUnrollFactor);
165  // Unroll completely if full loop unroll was specified.
166  if (clUnrollFull.getNumOccurrences() > 0 ||
167  (unrollFull.hasValue() && unrollFull.getValue()))
168  return loopUnrollFull(forOp);
169 
170  // Unroll by four otherwise.
171  return loopUnrollByFactor(forOp, kDefaultUnrollFactor);
172 }
