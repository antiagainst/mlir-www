54 /// Returns the linearized list of all view dimensions in a linalgOp. Applying
55 /// the inverse, concatenated loopToOperandRangeMaps to this list allows the
56 /// derivation of loop ranges for any linalgOp.
57 template <typename ConcreteOp>
58 SmallVector<Value, 8> getViewSizes(OpBuilder &builder, ConcreteOp linalgOp) {
59  auto loc = linalgOp.getLoc();
60  SmallVector<Value, 8> res;
61  for (auto v : linalgOp.getInputsAndOutputBuffers()) {
62  MemRefType t = v.getType().template cast<MemRefType>();
63  for (unsigned i = 0; i < t.getRank(); ++i)
64  res.push_back(builder.create<DimOp>(loc, v, i));
65  }
66  return res;
67 }

82 /// Performs standalone tiling of a single LinalgOp by `tileSizes`.
83 /// and permute the loop nest according to `permutation`
84 /// The permutation is expressed as a list of integers that specify
85 /// the new ordering of the loop nest. The length of `permutation`
86 /// must be equal to the length of `tileSizes`.
87 /// E.g. the permutation `(i,j,k) -> (j,k,i)` will be expressed with
88 /// `permutation = [1,2,0]`. All values in `permutation` must be
89 /// integers, in the range 0..`tileSizes.size()` without duplications
90 /// (i.e. `[1,1,2]` is an invalid permutation). An empty list
91 /// states for the identity permutation.
92 /// Returns a struct containing the tiled loops in the specified order
93 /// and the cloned op if successful, llvm::None otherwise.

95 /// `createAndFold` builder method. If `folder` is null, the regular `create`
96 /// method is called.

154 /// Apply the permutation defined by `permutation` to `inVec`.
155 /// Element `i` in `inVec` is mapped to location `j = permutation[i]`.
156 /// E.g.: for an input vector `inVec = ['a', 'b', 'c']` and a permutation vector
157 /// `permutation = [2, 0, 1]`, this function leaves `inVec = ['c', 'a', 'b']`.
158 template <typename T, unsigned N>
159 void applyPermutationToVector(SmallVector<T, N> &inVec,
160  ArrayRef<unsigned> permutation) {
161  SmallVector<T, N> auxVec(inVec.size());
162  for (unsigned i = 0; i < permutation.size(); ++i)
163  auxVec[i] = inVec[permutation[i]];
164  inVec = auxVec;
165 }
