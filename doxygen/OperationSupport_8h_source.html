58 enum class OperationProperty {
59  /// This bit is set for an operation if it is a commutative operation: that
60  /// is a binary operator (two inputs) where "a op b" and "b op a" produce the
61  /// same results.
62  Commutative = 0x1,
63 
64  /// This bit is set for operations that have no side effects: that means that
65  /// they do not read or write memory, or access any hidden state.
66  NoSideEffect = 0x2,
67 
68  /// This bit is set for an operation if it is a terminator: that means
69  /// an operation at the end of a block.
70  Terminator = 0x4,
71 
72  /// This bit is set for operations that are completely isolated from above.
73  /// This is used for operations whose regions are explicit capture only, i.e.
74  /// they are never allowed to implicitly reference values defined above the
75  /// parent operation.
76  IsolatedFromAbove = 0x8,
77 };

79 /// This is a "type erased" representation of a registered operation. This
80 /// should only be used by things like the AsmPrinter and other things that need
81 /// to be parameterized by generic operation hooks. Most user code should use
82 /// the concrete operation types.
83 class AbstractOperation {
84 public:
85  using OperationProperties = uint32_t;
86 
87  /// This is the name of the operation.
88  const StringRef name;
89 
90  /// This is the dialect that this operation belongs to.
91  Dialect &dialect;
92 
93  /// The unique identifier of the derived Op class.
94  ClassID *classID;
95 
96  /// Use the specified object to parse this ops custom assembly format.
97  ParseResult (&parseAssembly)(OpAsmParser &parser, OperationState &result);
98 
99  /// This hook implements the AsmPrinter for this operation.
100  void (&printAssembly)(Operation *op, OpAsmPrinter &p);
101 
102  /// This hook implements the verifier for this operation. It should emits an
103  /// error message and returns failure if a problem is detected, or returns
104  /// success if everything is ok.
105  LogicalResult (&verifyInvariants)(Operation *op);
106 
107  /// This hook implements a generalized folder for this operation. Operations
108  /// can implement this to provide simplifications rules that are applied by
109  /// the Builder::createOrFold API and the canonicalization pass.
110  ///
111  /// This is an intentionally limited interface - implementations of this hook
112  /// can only perform the following changes to the operation:
113  ///
114  /// 1. They can leave the operation alone and without changing the IR, and
115  /// return failure.
116  /// 2. They can mutate the operation in place, without changing anything else
117  /// in the IR. In this case, return success.
118  /// 3. They can return a list of existing values that can be used instead of
119  /// the operation. In this case, fill in the results list and return
120  /// success. The caller will remove the operation and use those results
121  /// instead.
122  ///
123  /// This allows expression of some simple in-place canonicalizations (e.g.
124  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
125  /// generalized constant folding.
126  LogicalResult (&foldHook)(Operation *op, ArrayRef<Attribute> operands,
127  SmallVectorImpl<OpFoldResult> &results);
128 
129  /// This hook returns any canonicalization pattern rewrites that the operation
130  /// supports, for use by the canonicalization pass.
131  void (&getCanonicalizationPatterns)(OwningRewritePatternList &results,
132  MLIRContext *context);
133 
134  /// Returns whether the operation has a particular property.
135  bool hasProperty(OperationProperty property) const {
136  return opProperties & static_cast<OperationProperties>(property);
137  }
138 
139  /// Returns an instance of the concept object for the given interface if it
140  /// was registered to this operation, null otherwise. This should not be used
141  /// directly.
142  template <typename T> typename T::Concept *getInterface() const {
143  return reinterpret_cast<typename T::Concept *>(
144  getRawInterface(T::getInterfaceID()));
145  }
146 
147  /// Returns if the operation has a particular trait.
148  template <template <typename T> class Trait> bool hasTrait() const {
149  return hasRawTrait(ClassID::getID<Trait>());
150  }
151 
152  /// Look up the specified operation in the specified MLIRContext and return a
153  /// pointer to it if present. Otherwise, return a null pointer.
154  static const AbstractOperation *lookup(StringRef opName,
155  MLIRContext *context);
156 
157  /// This constructor is used by Dialect objects when they register the list of
158  /// operations they contain.
159  template <typename T> static AbstractOperation get(Dialect &dialect) {
160  return AbstractOperation(
161  T::getOperationName(), dialect, T::getOperationProperties(),
162  ClassID::getID<T>(), T::parseAssembly, T::printAssembly,
163  T::verifyInvariants, T::foldHook, T::getCanonicalizationPatterns,
164  T::getRawInterface, T::hasTrait);
165  }
166 
167 private:
168  AbstractOperation(
169  StringRef name, Dialect &dialect, OperationProperties opProperties,
170  ClassID *classID,
171  ParseResult (&parseAssembly)(OpAsmParser &parser, OperationState &result),
172  void (&printAssembly)(Operation *op, OpAsmPrinter &p),
173  LogicalResult (&verifyInvariants)(Operation *op),
174  LogicalResult (&foldHook)(Operation *op, ArrayRef<Attribute> operands,
175  SmallVectorImpl<OpFoldResult> &results),
176  void (&getCanonicalizationPatterns)(OwningRewritePatternList &results,
177  MLIRContext *context),
178  void *(&getRawInterface)(ClassID *interfaceID),
179  bool (&hasTrait)(ClassID *traitID))
180  : name(name), dialect(dialect), classID(classID),
181  parseAssembly(parseAssembly), printAssembly(printAssembly),
182  verifyInvariants(verifyInvariants), foldHook(foldHook),
183  getCanonicalizationPatterns(getCanonicalizationPatterns),
184  opProperties(opProperties), getRawInterface(getRawInterface),
185  hasRawTrait(hasTrait) {}
186 
187  /// The properties of the operation.
188  const OperationProperties opProperties;
189 
190  /// Returns a raw instance of the concept for the given interface id if it is
191  /// registered to this operation, nullptr otherwise. This should not be used
192  /// directly.
193  void *(&getRawInterface)(ClassID *interfaceID);
194 
195  /// This hook returns if the operation contains the trait corresponding
196  /// to the given ClassID.
197  bool (&hasRawTrait)(ClassID *traitID);
198 };

274  bool resizableOperandList = false;

334 /// A utility class holding the information necessary to dynamically resize
335 /// operands.
336 struct ResizableStorage {
337  ResizableStorage(OpOperand *opBegin, unsigned numOperands)
338  : firstOpAndIsDynamic(opBegin, false), capacity(numOperands) {}
339 
340  ~ResizableStorage() { cleanupStorage(); }
341 
342  /// Cleanup any allocated storage.
343  void cleanupStorage() {
344  // If the storage is dynamic, then we need to free the storage.
345  if (isStorageDynamic())
346  free(firstOpAndIsDynamic.getPointer());
347  }
348 
349  /// Sets the storage pointer to a new dynamically allocated block.
350  void setDynamicStorage(OpOperand *opBegin) {
351  /// Cleanup the old storage if necessary.
352  cleanupStorage();
353  firstOpAndIsDynamic.setPointerAndInt(opBegin, true);
354  }
355 
356  /// Returns the current storage pointer.
357  OpOperand *getPointer() { return firstOpAndIsDynamic.getPointer(); }
358 
359  /// Returns if the current storage of operands is in the trailing objects is
360  /// in a dynamically allocated memory block.
361  bool isStorageDynamic() const { return firstOpAndIsDynamic.getInt(); }
362 
363  /// A pointer to the first operand element. This is either to the trailing
364  /// objects storage, or a dynamically allocated block of memory.
365  llvm::PointerIntPair<OpOperand *, 1, bool> firstOpAndIsDynamic;
366 
367  // The maximum number of operands that can be currently held by the storage.
368  unsigned capacity;
369 };

371 /// This class handles the management of operation operands. Operands are
372 /// stored similarly to the elements of a SmallVector except for two key
373 /// differences. The first is the inline storage, which is a trailing objects
374 /// array. The second is that being able to dynamically resize the operand list
375 /// is optional.
376 class OperandStorage final
377  : private llvm::TrailingObjects<OperandStorage, ResizableStorage,
378  OpOperand> {
379 public:
380  OperandStorage(unsigned numOperands, bool resizable)
381  : numOperands(numOperands), resizable(resizable) {
382  // Initialize the resizable storage.
383  if (resizable) {
384  new (&getResizableStorage())
385  ResizableStorage(getTrailingObjects<OpOperand>(), numOperands);
386  }
387  }
388 
389  ~OperandStorage() {
390  // Manually destruct the operands.
391  for (auto &operand : getOperands())
392  operand.~OpOperand();
393 
394  // If the storage is resizable then destruct the utility.
395  if (resizable)
396  getResizableStorage().~ResizableStorage();
397  }
398 
399  /// Replace the operands contained in the storage with the ones provided in
400  /// 'operands'.
401  void setOperands(Operation *owner, ValueRange operands);
402 
403  /// Erase an operand held by the storage.
404  void eraseOperand(unsigned index);
405 
406  /// Get the operation operands held by the storage.
407  MutableArrayRef<OpOperand> getOperands() {
408  return {getRawOperands(), size()};
409  }
410 
411  /// Return the number of operands held in the storage.
412  unsigned size() const { return numOperands; }
413 
414  /// Returns the additional size necessary for allocating this object.
415  static size_t additionalAllocSize(unsigned numOperands, bool resizable) {
416  return additionalSizeToAlloc<ResizableStorage, OpOperand>(resizable ? 1 : 0,
417  numOperands);
418  }
419 
420  /// Returns if this storage is resizable.
421  bool isResizable() const { return resizable; }
422 
423 private:
424  /// Clear the storage and destroy the current operands held by the storage.
425  void clear() { numOperands = 0; }
426 
427  /// Returns the current pointer for the raw operands array.
428  OpOperand *getRawOperands() {
429  return resizable ? getResizableStorage().getPointer()
430  : getTrailingObjects<OpOperand>();
431  }
432 
433  /// Returns the resizable operand utility class.
434  ResizableStorage &getResizableStorage() {
435  assert(resizable);
436  return *getTrailingObjects<ResizableStorage>();
437  }
438 
439  /// Grow the internal resizable operand storage.
440  void grow(ResizableStorage &resizeUtil, size_t minSize);
441 
442  /// The current number of operands, and the current max operand capacity.
443  unsigned numOperands : 31;
444 
445  /// Whether this storage is resizable or not.
446  bool resizable : 1;
447 
448  // This stuff is used by the TrailingObjects template.
449  friend llvm::TrailingObjects<OperandStorage, ResizableStorage, OpOperand>;
450  size_t numTrailingObjects(OverloadToken<ResizableStorage>) const {
451  return resizable ? 1 : 0;
452  }
453 };
