27 /// An IndexHandle is a simple wrapper around a ValueHandle.
28 /// IndexHandles are ubiquitous enough to justify a new type to allow simple
29 /// declarations without boilerplate such as:
30 ///
31 /// ```c++
32 /// IndexHandle i, j, k;
33 /// ```
34 struct IndexHandle : public ValueHandle {
35  explicit IndexHandle()
36  : ValueHandle(ScopedContext::getBuilder().getIndexType()) {}
37  explicit IndexHandle(index_type v) : ValueHandle(v) {}
38  explicit IndexHandle(Value v) : ValueHandle(v) {
39  assert(v.getType() == ScopedContext::getBuilder().getIndexType() &&
40  "Expected index type");
41  }
42  explicit IndexHandle(ValueHandle v) : ValueHandle(v) {
43  assert(v.getType() == ScopedContext::getBuilder().getIndexType() &&
44  "Expected index type");
45  }
46  IndexHandle &operator=(const ValueHandle &v) {
47  assert(v.getType() == ScopedContext::getBuilder().getIndexType() &&
48  "Expected index type");
49  /// Creating a new IndexHandle(v) and then std::swap rightly complains the
50  /// binding has already occurred and that we should use another name.
51  this->t = v.getType();
52  this->v = v.getValue();
53  return *this;
54  }
55 };

61 /// Entry point to build multiple ValueHandle* from a mutable list `ivs` of T.
62 template <typename T>
63 inline SmallVector<ValueHandle *, 8>
64 makeHandlePointers(MutableArrayRef<T> ivs) {
65  SmallVector<ValueHandle *, 8> pivs;
66  pivs.reserve(ivs.size());
67  for (auto &iv : ivs) {
68  pivs.push_back(&iv);
69  }
70  return pivs;
71 }

73 /// Returns a vector of the underlying Value from `ivs`.
74 inline SmallVector<Value, 8> extractValues(ArrayRef<IndexHandle> ivs) {
75  SmallVector<Value, 8> vals;
76  vals.reserve(ivs.size());
77  for (auto &iv : ivs) {
78  vals.push_back(iv.getValue());
79  }
80  return vals;

88 /// Helper structure to be used with ValueBuilder / OperationBuilder.
89 /// It serves the purpose of removing boilerplate specialization for the sole
90 /// purpose of implicitly converting ArrayRef<ValueHandle> -> ArrayRef<Value>.
91 class ValueHandleArray {
92 public:
93  ValueHandleArray(ArrayRef<ValueHandle> vals) {
94  values.append(vals.begin(), vals.end());
95  }
96  ValueHandleArray(ArrayRef<IndexHandle> vals) {
97  values.append(vals.begin(), vals.end());
98  }
99  ValueHandleArray(ArrayRef<index_type> vals) {
100  SmallVector<IndexHandle, 8> tmp(vals.begin(), vals.end());
101  values.append(tmp.begin(), tmp.end());
102  }
103  operator ArrayRef<Value>() { return values; }
104 
105 private:
106  ValueHandleArray() = default;
107  SmallVector<Value, 8> values;
108 };

118 /// Helper variadic abstraction to allow extending to any MLIR op without
119 /// boilerplate or Tablegen.
120 /// Arguably a builder is not a ValueHandle but in practice it is only used as
121 /// an alias to a notional ValueHandle<Op>.
122 /// Implementing it as a subclass allows it to compose all the way to Value.
123 /// Without subclassing, implicit conversion to Value would fail when composing
124 /// in patterns such as: `select(a, b, select(c, d, e))`.
125 template <typename Op> struct ValueBuilder : public ValueHandle {
126  // Builder-based
127  template <typename... Args>
128  ValueBuilder(Args... args)
129  : ValueHandle(ValueHandle::create<Op>(detail::unpack(args)...)) {}
130  ValueBuilder(ArrayRef<ValueHandle> vs)
131  : ValueBuilder(ValueBuilder::create<Op>(detail::unpack(vs))) {}
132  template <typename... Args>
133  ValueBuilder(ArrayRef<ValueHandle> vs, Args... args)
134  : ValueHandle(ValueHandle::create<Op>(detail::unpack(vs),
135  detail::unpack(args)...)) {}
136  template <typename T, typename... Args>
137  ValueBuilder(T t, ArrayRef<ValueHandle> vs, Args... args)
138  : ValueHandle(ValueHandle::create<Op>(
139  detail::unpack(t), detail::unpack(vs), detail::unpack(args)...)) {}
140  template <typename T1, typename T2, typename... Args>
141  ValueBuilder(T1 t1, T2 t2, ArrayRef<ValueHandle> vs, Args... args)
142  : ValueHandle(ValueHandle::create<Op>(
143  detail::unpack(t1), detail::unpack(t2), detail::unpack(vs),
144  detail::unpack(args)...)) {}
145 
146  /// Folder-based
147  template <typename... Args>
148  ValueBuilder(OperationFolder *folder, Args... args)
149  : ValueHandle(ValueHandle::create<Op>(folder, detail::unpack(args)...)) {}
150  ValueBuilder(OperationFolder *folder, ArrayRef<ValueHandle> vs)
151  : ValueBuilder(ValueBuilder::create<Op>(folder, detail::unpack(vs))) {}
152  template <typename... Args>
153  ValueBuilder(OperationFolder *folder, ArrayRef<ValueHandle> vs, Args... args)
154  : ValueHandle(ValueHandle::create<Op>(folder, detail::unpack(vs),
155  detail::unpack(args)...)) {}
156  template <typename T, typename... Args>
157  ValueBuilder(OperationFolder *folder, T t, ArrayRef<ValueHandle> vs,
158  Args... args)
159  : ValueHandle(ValueHandle::create<Op>(folder, detail::unpack(t),
160  detail::unpack(vs),
161  detail::unpack(args)...)) {}
162  template <typename T1, typename T2, typename... Args>
163  ValueBuilder(OperationFolder *folder, T1 t1, T2 t2, ArrayRef<ValueHandle> vs,
164  Args... args)
165  : ValueHandle(ValueHandle::create<Op>(
166  folder, detail::unpack(t1), detail::unpack(t2), detail::unpack(vs),
167  detail::unpack(args)...)) {}
168 
169  ValueBuilder() : ValueHandle(ValueHandle::create<Op>()) {}
170 };

226 /// Creates a new mlir::Block* and branches to it from the current block.
227 /// Argument types are specified by `operands`.
228 /// Captures the new block in `bh` and the actual `operands` in `captures`. To
229 /// insert the new mlir::Block*, a local ScopedContext is constructed and
230 /// released to the current block. The branch operation is then added to the
231 /// new block.
232 ///
233 /// Prerequisites:
234 /// `b` has not yet captured an mlir::Block*.
235 /// No `captures` have captured any mlir::Value.
236 /// All `operands` have already captured an mlir::Value
237 /// captures.size() == operands.size()
238 /// captures and operands are pairwise of the same type.
239 OperationHandle br(BlockHandle *bh, ArrayRef<ValueHandle *> captures,
240  ArrayRef<ValueHandle> operands);
