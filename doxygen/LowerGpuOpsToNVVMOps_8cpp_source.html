32 /// Derived type converter for GPU to NVVM lowering. The GPU dialect uses memory
33 /// space 5 for private memory attributions, but NVVM represents private
34 /// memory allocations as local `alloca`s in the default address space. This
35 /// converter drops the private memory space to support the use case above.
36 class NVVMTypeConverter : public LLVMTypeConverter {
37 public:
38  using LLVMTypeConverter::LLVMTypeConverter;

40  Type convertType(Type type) override {
41  auto memref = type.dyn_cast<MemRefType>();
42  if (memref &&
43  memref.getMemorySpace() == gpu::GPUDialect::getPrivateAddressSpace()) {
44  type = MemRefType::Builder(memref).setMemorySpace(0);
45  }
46 
47  return LLVMTypeConverter::convertType(type);
48  }

52 struct GPUAllReduceOpLowering : public LLVMOpLowering {
53  using AccumulatorFactory =
54  std::function<Value(Location, Value, Value, ConversionPatternRewriter &)>;

56  explicit GPUAllReduceOpLowering(LLVMTypeConverter &lowering_)
57  : LLVMOpLowering(gpu::AllReduceOp::getOperationName(),
58  lowering_.getDialect()->getContext(), lowering_),
59  int32Type(LLVM::LLVMType::getInt32Ty(lowering_.getDialect())) {}

61  PatternMatchResult
62  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
63  ConversionPatternRewriter &rewriter) const override {
64  Location loc = op->getLoc();
65  Value operand = operands.front();
66 
67  // TODO(csigg): Generalize to other types of accumulation.
68  assert(op->getOperand(0).getType().isIntOrFloat());
69 
70  // Create the reduction using an accumulator factory.
71  AccumulatorFactory factory =
72  getFactory(cast<gpu::AllReduceOp>(op), operand);
73  assert(factory && "failed to create accumulator factory");
74  Value result = createBlockReduce(loc, operand, factory, rewriter);
75 
76  rewriter.replaceOp(op, {result});
77  return matchSuccess();
78  }

80 private:
81  /// Returns an accumulator factory using either the op attribute or the body
82  /// region.
83  AccumulatorFactory getFactory(gpu::AllReduceOp allReduce,
84  Value operand) const {
85  if (!allReduce.body().empty()) {
86  return getFactory(allReduce.body());
87  }
88  if (allReduce.op()) {
89  auto type = operand.getType().cast<LLVM::LLVMType>();
90  return getFactory(*allReduce.op(), type.getUnderlyingType());
91  }
92  return AccumulatorFactory();
93  }

95  /// Returns an accumulator factory that clones the body. The body's entry
96  /// block is expected to have 2 arguments. The gpu.yield return the
97  /// accumulated value of the same type.
98  AccumulatorFactory getFactory(Region &body) const {
99  return AccumulatorFactory([&](Location loc, Value lhs, Value rhs,
100  ConversionPatternRewriter &rewriter) {
101  Block *block = rewriter.getInsertionBlock();
102  Block *split = rewriter.splitBlock(block, rewriter.getInsertionPoint());
103 
104  // Insert accumulator body between split block.
105  BlockAndValueMapping mapping;
106  mapping.map(body.front().getArgument(0), lhs);
107  mapping.map(body.front().getArgument(1), rhs);
108  rewriter.cloneRegionBefore(body, *split->getParent(),
109  split->getIterator(), mapping);
110 
111  // Add branch before inserted body, into body.
112  block = block->getNextNode();
113  rewriter.create<LLVM::BrOp>(loc, ArrayRef<Value>{},
114  llvm::makeArrayRef(block), ValueRange());
115 
116  // Replace all gpu.yield ops with branch out of body.
117  for (; block != split; block = block->getNextNode()) {
118  Operation *terminator = block->getTerminator();
119  if (!llvm::isa<gpu::YieldOp>(terminator))
120  continue;
121  rewriter.setInsertionPointToEnd(block);
122  rewriter.replaceOpWithNewOp<LLVM::BrOp>(
123  terminator, ArrayRef<Value>{}, llvm::makeArrayRef(split),
124  ValueRange(terminator->getOperand(0)));
125  }
126 
127  // Return accumulator result.
128  rewriter.setInsertionPointToStart(split);
129  return split->addArgument(lhs.getType());
130  });
131  }

133  /// Returns an accumulator factory that creates an op specified by opName.
134  AccumulatorFactory getFactory(StringRef opName, llvm::Type *type) const {
135  if (type->isVectorTy() || type->isArrayTy())
136  return getFactory(opName, type->getSequentialElementType());
137 
138  bool isFloatingPoint = type->isFloatingPointTy();
139 
140  if (opName == "add") {
141  return isFloatingPoint ? getFactory<LLVM::FAddOp>()
142  : getFactory<LLVM::AddOp>();
143  }
144  if (opName == "mul") {
145  return isFloatingPoint ? getFactory<LLVM::FMulOp>()
146  : getFactory<LLVM::MulOp>();
147  }
148 
149  return AccumulatorFactory();
150  }

153  template <typename T> AccumulatorFactory getFactory() const {
154  return [](Location loc, Value lhs, Value rhs,

160  /// Creates an all_reduce across the block.
161  ///
162  /// First reduce the elements within a warp. The first thread of each warp
163  /// writes the intermediate result to shared memory. After synchronizing the
164  /// block, the first warp reduces the values from shared memory. The result
165  /// is broadcasted to all threads through shared memory.
166  ///
167  /// %warp_reduce = `createWarpReduce(%operand)`
168  /// %shared_mem_ptr = llvm.mlir.addressof @reduce_buffer
169  /// %zero = llvm.mlir.constant(0 : i32) : !llvm.i32
170  /// %lane_id = nvvm.read.ptx.sreg.laneid : !llvm.i32
171  /// %is_first_lane = llvm.icmp "eq" %lane_id, %zero : !llvm.i1
172  /// %thread_idx = `getLinearThreadIndex()` : !llvm.i32
173  /// llvm.cond_br %is_first_lane, ^then1, ^continue1
174  /// ^then1:
175  /// %warp_id = `getWarpId()`
176  /// %store_dst = llvm.getelementptr %shared_mem_ptr[%zero, %warp_id]
177  /// llvm.store %store_dst, %warp_reduce
178  /// llvm.br ^continue1
179  /// ^continue1:
180  /// nvvm.barrier0
181  /// %num_warps = `getNumWarps()` : !llvm.i32
182  /// %is_valid_warp = llvm.icmp "slt" %thread_idx, %num_warps
183  /// %result_ptr = llvm.getelementptr %shared_mem_ptr[%zero, %zero]
184  /// llvm.cond_br %is_first_lane, ^then2, ^continue2
185  /// ^then2:
186  /// %load_src = llvm.getelementptr %shared_mem_ptr[%zero, %thread_idx]
187  /// %value = llvm.load %load_src
188  /// %result = `createWarpReduce(%value)`
189  /// llvm.store %result_ptr, %result
190  /// llvm.br ^continue2
191  /// ^continue2:
192  /// nvvm.barrier0
193  /// %result = llvm.load %result_ptr
194  /// return %result
195  ///
196  Value createBlockReduce(Location loc, Value operand,
197  AccumulatorFactory &accumFactory,
198  ConversionPatternRewriter &rewriter) const {
199  auto type = operand.getType().cast<LLVM::LLVMType>();
200 
201  // Create shared memory array to store the warp reduction.
202  auto module = operand.getDefiningOp()->getParentOfType<gpu::GPUModuleOp>();
203  assert(module && "op must belong to a module");
204  Value sharedMemPtr =
205  createSharedMemoryArray(loc, module, type, kWarpSize, rewriter);
206 
207  Value zero = rewriter.create<LLVM::ConstantOp>(
208  loc, int32Type, rewriter.getI32IntegerAttr(0u));
209  Value laneId = rewriter.create<NVVM::LaneIdOp>(loc, int32Type);
210  Value isFirstLane = rewriter.create<LLVM::ICmpOp>(
211  loc, LLVM::ICmpPredicate::eq, laneId, zero);
212  Value threadIdx = getLinearThreadIndex(loc, rewriter);
213  Value blockSize = getBlockSize(loc, rewriter);
214  Value activeWidth = getActiveWidth(loc, threadIdx, blockSize, rewriter);
215 
216  // Reduce elements within each warp to produce the intermediate results.
217  Value warpReduce = createWarpReduce(loc, activeWidth, laneId, operand,
218  accumFactory, rewriter);
219 
220  // Write the intermediate results to shared memory, using the first lane of
221  // each warp.
222  createPredicatedBlock(loc, rewriter, isFirstLane, [&] {
223  Value warpId = getDivideByWarpSize(threadIdx, rewriter);
224  Value storeDst = rewriter.create<LLVM::GEPOp>(
225  loc, type, sharedMemPtr, ArrayRef<Value>({zero, warpId}));
226  rewriter.create<LLVM::StoreOp>(loc, warpReduce, storeDst);
227  });
228  rewriter.create<NVVM::Barrier0Op>(loc);
229 
230  Value numWarps = getNumWarps(loc, blockSize, rewriter);
231  Value isValidWarp = rewriter.create<LLVM::ICmpOp>(
232  loc, LLVM::ICmpPredicate::slt, threadIdx, numWarps);
233  Value resultPtr = rewriter.create<LLVM::GEPOp>(
234  loc, type, sharedMemPtr, ArrayRef<Value>({zero, zero}));
235 
236  // Use the first numWarps threads to reduce the intermediate results from
237  // shared memory. The final result is written to shared memory again.
238  createPredicatedBlock(loc, rewriter, isValidWarp, [&] {
239  Value loadSrc = rewriter.create<LLVM::GEPOp>(
240  loc, type, sharedMemPtr, ArrayRef<Value>({zero, threadIdx}));
241  Value value = rewriter.create<LLVM::LoadOp>(loc, type, loadSrc);
242  Value result = createWarpReduce(loc, numWarps, laneId, value,
243  accumFactory, rewriter);
244  rewriter.create<LLVM::StoreOp>(loc, result, resultPtr);
245  });
246  rewriter.create<NVVM::Barrier0Op>(loc);
247 
248  // Load and return result from shared memory.
249  Value result = rewriter.create<LLVM::LoadOp>(loc, type, resultPtr);
250  return result;
251  }

253  /// Creates an if-block skeleton and calls the two factories to generate the
254  /// ops in the `then` and `else` block..
255  ///
256  /// llvm.cond_br %condition, ^then, ^continue
257  /// ^then:
258  /// %then_operands = `thenOpsFactory()`
259  /// llvm.br ^continue(%then_operands)
260  /// ^else:
261  /// %else_operands = `elseOpsFactory()`
262  /// llvm.br ^continue(%else_operands)
263  /// ^continue(%block_operands):
264  ///
265  template <typename ThenOpsFactory, typename ElseOpsFactory>
266  void createIf(Location loc, ConversionPatternRewriter &rewriter,
267  Value condition, ThenOpsFactory &&thenOpsFactory,
268  ElseOpsFactory &&elseOpsFactory) const {
269  Block *currentBlock = rewriter.getInsertionBlock();
270  auto currentPoint = rewriter.getInsertionPoint();
271 
272  Block *thenBlock = rewriter.splitBlock(currentBlock, currentPoint);
273  Block *elseBlock = rewriter.splitBlock(thenBlock, thenBlock->begin());
274  Block *continueBlock = rewriter.splitBlock(elseBlock, elseBlock->begin());
275 
276  rewriter.setInsertionPointToEnd(currentBlock);
277  rewriter.create<LLVM::CondBrOp>(loc, llvm::makeArrayRef(condition),
278  ArrayRef<Block *>{thenBlock, elseBlock});
279 
280  auto addBranch = [&](ValueRange operands) {
281  rewriter.create<LLVM::BrOp>(loc, ArrayRef<Value>{},
282  llvm::makeArrayRef(continueBlock),
283  llvm::makeArrayRef(operands));
284  };
285 
286  rewriter.setInsertionPointToStart(thenBlock);
287  auto thenOperands = thenOpsFactory();
288  addBranch(thenOperands);
289 
290  rewriter.setInsertionPointToStart(elseBlock);
291  auto elseOperands = elseOpsFactory();
292  addBranch(elseOperands);
293 
294  assert(thenOperands.size() == elseOperands.size());
295  rewriter.setInsertionPointToStart(continueBlock);
296  for (auto operand : thenOperands)
297  continueBlock->addArgument(operand.getType());
298  }

301  template <typename Factory>

314  /// Creates a reduction across the first activeWidth lanes of a warp.
315  /// The first lane returns the result, all others return values are undefined.
316  Value createWarpReduce(Location loc, Value activeWidth, Value laneId,
317  Value operand, AccumulatorFactory accumFactory,
318  ConversionPatternRewriter &rewriter) const {
319  Value warpSize = rewriter.create<LLVM::ConstantOp>(
320  loc, int32Type, rewriter.getI32IntegerAttr(kWarpSize));
321  Value isPartialWarp = rewriter.create<LLVM::ICmpOp>(
322  loc, LLVM::ICmpPredicate::slt, activeWidth, warpSize);
323  auto type = operand.getType().cast<LLVM::LLVMType>();
324 
325  createIf(
326  loc, rewriter, isPartialWarp,
327  // Generate reduction over a (potentially) partial warp.
328  [&] {
329  Value value = operand;
330  Value one = rewriter.create<LLVM::ConstantOp>(
331  loc, int32Type, rewriter.getI32IntegerAttr(1));
332  // Bit mask of active lanes: `(1 << activeWidth) - 1`.
333  Value activeMask = rewriter.create<LLVM::SubOp>(
334  loc, int32Type,
335  rewriter.create<LLVM::ShlOp>(loc, int32Type, one, activeWidth),
336  one);
337  // Clamp lane: `activeWidth - 1`
338  Value maskAndClamp =
339  rewriter.create<LLVM::SubOp>(loc, int32Type, activeWidth, one);
340  auto dialect = lowering.getDialect();
341  auto predTy = LLVM::LLVMType::getInt1Ty(dialect);
342  auto shflTy = LLVM::LLVMType::getStructTy(dialect, {type, predTy});
343  auto returnValueAndIsValidAttr = rewriter.getUnitAttr();
344 
345  // Repeatedly shuffle value from 'laneId ^ i' and accumulate if source
346  // lane is within the active range. All lanes contain the final
347  // result, but only the first lane's result is used.
348  for (int i = 1; i < kWarpSize; i <<= 1) {
349  Value offset = rewriter.create<LLVM::ConstantOp>(
350  loc, int32Type, rewriter.getI32IntegerAttr(i));
351  Value shfl = rewriter.create<NVVM::ShflBflyOp>(
352  loc, shflTy, activeMask, value, offset, maskAndClamp,
353  returnValueAndIsValidAttr);
354  Value isActiveSrcLane = rewriter.create<LLVM::ExtractValueOp>(
355  loc, predTy, shfl, rewriter.getIndexArrayAttr(1));
356  // Skip the accumulation if the shuffle op read from a lane outside
357  // of the active range.
358  createIf(
359  loc, rewriter, isActiveSrcLane,
360  [&] {
361  Value shflValue = rewriter.create<LLVM::ExtractValueOp>(
362  loc, type, shfl, rewriter.getIndexArrayAttr(0));
363  return SmallVector<Value, 1>{
364  accumFactory(loc, value, shflValue, rewriter)};
365  },
366  [&] { return llvm::makeArrayRef(value); });
367  value = rewriter.getInsertionBlock()->getArgument(0);
368  }
369  return SmallVector<Value, 1>{value};
370  },
371  // Generate a reduction over the entire warp. This is a specialization
372  // of the above reduction with unconditional accumulation.
373  [&] {
374  Value value = operand;
375  Value activeMask = rewriter.create<LLVM::ConstantOp>(
376  loc, int32Type, rewriter.getI32IntegerAttr(~0u));
377  Value maskAndClamp = rewriter.create<LLVM::ConstantOp>(
378  loc, int32Type, rewriter.getI32IntegerAttr(kWarpSize - 1));
379  for (int i = 1; i < kWarpSize; i <<= 1) {
380  Value offset = rewriter.create<LLVM::ConstantOp>(
381  loc, int32Type, rewriter.getI32IntegerAttr(i));
382  Value shflValue = rewriter.create<NVVM::ShflBflyOp>(
383  loc, type, activeMask, value, offset, maskAndClamp,
384  /*return_value_and_is_valid=*/UnitAttr());
385  value = accumFactory(loc, value, shflValue, rewriter);
386  }
387  return SmallVector<Value, 1>{value};
388  });
389  return rewriter.getInsertionBlock()->getArgument(0);
390  }

408  /// Returns the index of the thread within the block.
409  Value getLinearThreadIndex(Location loc,

411  Value dimX = rewriter.create<NVVM::BlockDimXOp>(loc, int32Type);
412  Value dimY = rewriter.create<NVVM::BlockDimYOp>(loc, int32Type);
413  Value idX = rewriter.create<NVVM::ThreadIdXOp>(loc, int32Type);
414  Value idY = rewriter.create<NVVM::ThreadIdYOp>(loc, int32Type);

417  Value tmp2 = rewriter.create<LLVM::AddOp>(loc, int32Type, tmp1, idY);

423  Value getBlockSize(Location loc, ConversionPatternRewriter &rewriter) const {
424  Value dimX = rewriter.create<NVVM::BlockDimXOp>(loc, int32Type);
425  Value dimY = rewriter.create<NVVM::BlockDimYOp>(loc, int32Type);

431  /// Returns the number of warps in the block.
432  Value getNumWarps(Location loc, Value blockSize,
433  ConversionPatternRewriter &rewriter) const {
434  auto warpSizeMinusOne = rewriter.create<LLVM::ConstantOp>(
435  loc, int32Type, rewriter.getI32IntegerAttr(kWarpSize - 1));
436  auto biasedBlockSize = rewriter.create<LLVM::AddOp>(
437  loc, int32Type, blockSize, warpSizeMinusOne);
438  return getDivideByWarpSize(biasedBlockSize, rewriter);
439  }
440 
441  /// Returns the number of active threads in the warp, not clamped to 32.
442  Value getActiveWidth(Location loc, Value threadIdx, Value blockSize,
443  ConversionPatternRewriter &rewriter) const {
444  Value threadIdxMask = rewriter.create<LLVM::ConstantOp>(
445  loc, int32Type, rewriter.getI32IntegerAttr(~(kWarpSize - 1)));
446  Value numThreadsWithSmallerWarpId =
447  rewriter.create<LLVM::AndOp>(loc, threadIdx, threadIdxMask);
448  return rewriter.create<LLVM::SubOp>(loc, blockSize,
449  numThreadsWithSmallerWarpId);
450  }
451 
452  /// Returns value divided by the warp size (i.e. 32).
453  Value getDivideByWarpSize(Value value,
454  ConversionPatternRewriter &rewriter) const {
455  auto loc = value.getLoc();
456  auto warpSize = rewriter.create<LLVM::ConstantOp>(
457  loc, int32Type, rewriter.getI32IntegerAttr(kWarpSize));
458  return rewriter.create<LLVM::SDivOp>(loc, int32Type, value, warpSize);
459  }

471  /// Lowers a shuffle to the corresponding NVVM op.
472  ///
473  /// Convert the `width` argument into an activeMask (a bitmask which specifies
474  /// which threads participate in the shuffle) and a maskAndClamp (specifying
475  /// the highest lane which participates in the shuffle).
476  ///
477  /// %one = llvm.constant(1 : i32) : !llvm.i32
478  /// %shl = llvm.shl %one, %width : !llvm.i32
479  /// %active_mask = llvm.sub %shl, %one : !llvm.i32
480  /// %mask_and_clamp = llvm.sub %width, %one : !llvm.i32
481  /// %shfl = nvvm.shfl.sync.bfly %active_mask, %value, %offset,
482  /// %mask_and_clamp : !llvm<"{ float, i1 }">
483  /// %shfl_value = llvm.extractvalue %shfl[0 : index] :
484  /// !llvm<"{ float, i1 }">
485  /// %shfl_pred = llvm.extractvalue %shfl[1 : index] :
486  /// !llvm<"{ float, i1 }">
487  PatternMatchResult
488  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
489  ConversionPatternRewriter &rewriter) const override {
490  Location loc = op->getLoc();
491  gpu::ShuffleOpOperandAdaptor adaptor(operands);
492 
493  auto dialect = lowering.getDialect();
494  auto valueTy = adaptor.value().getType().cast<LLVM::LLVMType>();
495  auto int32Type = LLVM::LLVMType::getInt32Ty(dialect);
496  auto predTy = LLVM::LLVMType::getInt1Ty(dialect);
497  auto resultTy = LLVM::LLVMType::getStructTy(dialect, {valueTy, predTy});
498 
499  Value one = rewriter.create<LLVM::ConstantOp>(
500  loc, int32Type, rewriter.getI32IntegerAttr(1));
501  // Bit mask of active lanes: `(1 << activeWidth) - 1`.
502  Value activeMask = rewriter.create<LLVM::SubOp>(
503  loc, int32Type,
504  rewriter.create<LLVM::ShlOp>(loc, int32Type, one, adaptor.width()),
505  one);
506  // Clamp lane: `activeWidth - 1`
507  Value maskAndClamp =
508  rewriter.create<LLVM::SubOp>(loc, int32Type, adaptor.width(), one);
509 
510  auto returnValueAndIsValidAttr = rewriter.getUnitAttr();
511  Value shfl = rewriter.create<NVVM::ShflBflyOp>(
512  loc, resultTy, activeMask, adaptor.value(), adaptor.offset(),
513  maskAndClamp, returnValueAndIsValidAttr);
514  Value shflValue = rewriter.create<LLVM::ExtractValueOp>(
515  loc, valueTy, shfl, rewriter.getIndexArrayAttr(0));
516  Value isActiveSrcLane = rewriter.create<LLVM::ExtractValueOp>(
517  loc, predTy, shfl, rewriter.getIndexArrayAttr(1));
518 
519  rewriter.replaceOp(op, {shflValue, isActiveSrcLane});
520  return matchSuccess();
521  }

531  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
532  ConversionPatternRewriter &rewriter) const override {
533  assert(operands.empty() && "func op is not expected to have operands");
534  auto gpuFuncOp = cast<gpu::GPUFuncOp>(op);
535  Location loc = gpuFuncOp.getLoc();
536 
537  SmallVector<LLVM::GlobalOp, 3> workgroupBuffers;
538  workgroupBuffers.reserve(gpuFuncOp.getNumWorkgroupAttributions());
539  for (auto en : llvm::enumerate(gpuFuncOp.getWorkgroupAttributions())) {
540  Value attribution = en.value();
541 
542  auto type = attribution.getType().dyn_cast<MemRefType>();
543  assert(type && type.hasStaticShape() && "unexpected type in attribution");
544 
545  uint64_t numElements = type.getNumElements();
546 
547  auto elementType =
548  lowering.convertType(type.getElementType()).cast<LLVM::LLVMType>();
549  auto arrayType = LLVM::LLVMType::getArrayTy(elementType, numElements);
550  std::string name = std::string(
551  llvm::formatv("__wg_{0}_{1}", gpuFuncOp.getName(), en.index()));
552  auto globalOp = rewriter.create<LLVM::GlobalOp>(
553  gpuFuncOp.getLoc(), arrayType, /*isConstant=*/false,
554  LLVM::Linkage::Internal, name, /*value=*/Attribute(),
555  gpu::GPUDialect::getWorkgroupAddressSpace());
556  workgroupBuffers.push_back(globalOp);
557  }
558 
559  // Rewrite the original GPU function to an LLVM function.
560  auto funcType = lowering.convertType(gpuFuncOp.getType())
561  .cast<LLVM::LLVMType>()
562  .getPointerElementTy();
563 
564  // Remap proper input types.
565  TypeConverter::SignatureConversion signatureConversion(
566  gpuFuncOp.front().getNumArguments());
567  for (unsigned i = 0, e = funcType.getFunctionNumParams(); i < e; ++i)
568  signatureConversion.addInputs(i, funcType.getFunctionParamType(i));
569 
570  // Create the new function operation. Only copy those attributes that are
571  // not specific to function modeling.
572  SmallVector<NamedAttribute, 4> attributes;
573  for (const auto &attr : gpuFuncOp.getAttrs()) {
574  if (attr.first.is(SymbolTable::getSymbolAttrName()) ||
575  attr.first.is(impl::getTypeAttrName()) ||
576  attr.first.is(gpu::GPUFuncOp::getNumWorkgroupAttributionsAttrName()))
577  continue;
578  attributes.push_back(attr);
579  }
580  auto llvmFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
581  gpuFuncOp.getLoc(), gpuFuncOp.getName(), funcType,
582  LLVM::Linkage::External, attributes);
583 
584  {
585  // Insert operations that correspond to converted workgroup and private
586  // memory attributions to the body of the function. This must operate on
587  // the original function, before the body region is inlined in the new
588  // function to maintain the relation between block arguments and the
589  // parent operation that assigns their semantics.
590  OpBuilder::InsertionGuard guard(rewriter);
591 
592  // Rewrite workgroup memory attributions to addresses of global buffers.
593  rewriter.setInsertionPointToStart(&gpuFuncOp.front());
594  unsigned numProperArguments = gpuFuncOp.getNumArguments();
595  auto i32Type = LLVM::LLVMType::getInt32Ty(lowering.getDialect());
596 
597  Value zero = nullptr;
598  if (!workgroupBuffers.empty())
599  zero = rewriter.create<LLVM::ConstantOp>(loc, i32Type,
600  rewriter.getI32IntegerAttr(0));
601  for (auto en : llvm::enumerate(workgroupBuffers)) {
602  LLVM::GlobalOp global = en.value();
603  Value address = rewriter.create<LLVM::AddressOfOp>(loc, global);
604  auto elementType = global.getType().getArrayElementType();
605  Value memory = rewriter.create<LLVM::GEPOp>(
606  loc, elementType.getPointerTo(global.addr_space().getZExtValue()),
607  address, ArrayRef<Value>{zero, zero});
608 
609  // Build a memref descriptor pointing to the buffer to plug with the
610  // existing memref infrastructure. This may use more registers than
611  // otherwise necessary given that memref sizes are fixed, but we can try
612  // and canonicalize that away later.
613  Value attribution = gpuFuncOp.getWorkgroupAttributions()[en.index()];
614  auto type = attribution.getType().cast<MemRefType>();
615  auto descr = MemRefDescriptor::fromStaticShape(rewriter, loc, lowering,
616  type, memory);
617  signatureConversion.remapInput(numProperArguments + en.index(), descr);
618  }
619 
620  // Rewrite private memory attributions to alloca'ed buffers.
621  unsigned numWorkgroupAttributions =
622  gpuFuncOp.getNumWorkgroupAttributions();
623  auto int64Ty = LLVM::LLVMType::getInt64Ty(lowering.getDialect());
624  for (auto en : llvm::enumerate(gpuFuncOp.getPrivateAttributions())) {
625  Value attribution = en.value();
626  auto type = attribution.getType().cast<MemRefType>();
627  assert(type && type.hasStaticShape() &&
628  "unexpected type in attribution");
629 
630  // Explicitly drop memory space when lowering private memory
631  // attributions since NVVM models it as `alloca`s in the default
632  // memory space and does not support `alloca`s with addrspace(5).
633  auto ptrType = lowering.convertType(type.getElementType())
634  .cast<LLVM::LLVMType>()
635  .getPointerTo();
636  Value numElements = rewriter.create<LLVM::ConstantOp>(
637  gpuFuncOp.getLoc(), int64Ty,
638  rewriter.getI64IntegerAttr(type.getNumElements()));
639  Value allocated = rewriter.create<LLVM::AllocaOp>(
640  gpuFuncOp.getLoc(), ptrType, numElements, /*alignment=*/0);
641  auto descr = MemRefDescriptor::fromStaticShape(rewriter, loc, lowering,
642  type, allocated);
643  signatureConversion.remapInput(
644  numProperArguments + numWorkgroupAttributions + en.index(), descr);
645  }
646  }
647 
648  // Move the region to the new function, update the entry block signature.
649  rewriter.inlineRegionBefore(gpuFuncOp.getBody(), llvmFuncOp.getBody(),
650  llvmFuncOp.end());
651  rewriter.applySignatureConversion(&llvmFuncOp.getBody(),
652  signatureConversion);
653 
654  {
655  // For memref-typed arguments, insert the relevant loads in the beginning
656  // of the block to comply with the LLVM dialect calling convention. This
657  // needs to be done after signature conversion to get the right types.
658  OpBuilder::InsertionGuard guard(rewriter);
659  Block &block = llvmFuncOp.front();
660  rewriter.setInsertionPointToStart(&block);
661 
662  for (auto en : llvm::enumerate(gpuFuncOp.getType().getInputs())) {
663  if (!en.value().isa<MemRefType>() &&
664  !en.value().isa<UnrankedMemRefType>())
665  continue;
666 
667  BlockArgument arg = block.getArgument(en.index());
668  Value loaded = rewriter.create<LLVM::LoadOp>(loc, arg);
669  rewriter.replaceUsesOfBlockArgument(arg, loaded);
670  }
671  }
672 
673  rewriter.eraseOp(gpuFuncOp);
674  return matchSuccess();
675  }
676 };

719  if (failed(applyPartialConversion(m, target, patterns, &converter)))

721  }
