25 /// Recursively collect all of the operations to convert from within 'region'.
26 /// If 'target' is nonnull, operations that are recursively legal have their
27 /// regions pre-filtered to avoid considering them for legalization.
28 static LogicalResult
29 computeConversionSet(iterator_range<Region::iterator> region,
30  Location regionLoc, std::vector<Operation *> &toConvert,
31  ConversionTarget *target = nullptr) {
32  if (llvm::empty(region))
33  return success();
34 
35  // Traverse starting from the entry block.
36  SmallVector<Block *, 16> worklist(1, &*region.begin());
37  DenseSet<Block *> visitedBlocks;
38  visitedBlocks.insert(worklist.front());
39  while (!worklist.empty()) {
40  Block *block = worklist.pop_back_val();
41 
42  // Compute the conversion set of each of the nested operations.
43  for (Operation &op : *block) {
44  toConvert.emplace_back(&op);
45 
46  // Don't check this operation's children for conversion if the operation
47  // is recursively legal.
48  auto legalityInfo = target ? target->isLegal(&op)
49  : Optional<ConversionTarget::LegalOpDetails>();
50  if (legalityInfo && legalityInfo->isRecursivelyLegal)
51  continue;
52  for (auto &region : op.getRegions())
53  computeConversionSet(region.getBlocks(), region.getLoc(), toConvert,
54  target);
55  }
56 
57  // Recurse to children that haven't been visited.
58  for (Block *succ : block->getSuccessors())
59  if (visitedBlocks.insert(succ).second)
60  worklist.push_back(succ);
61  }
62 
63  // Check that all blocks in the region were visited.
64  if (llvm::any_of(llvm::drop_begin(region, 1),
65  [&](Block &block) { return !visitedBlocks.count(&block); }))
66  return emitError(regionLoc, "unreachable blocks were not converted");
67  return success();
68 }

94 /// Lookup a mapped value within the map. If a mapping for the provided value
95 /// does not exist then return the provided value.
96 Value ConversionValueMapping::lookupOrDefault(Value from) const {
97  // If this value had a valid mapping, unmap that value as well in the case
98  // that it was also replaced.
99  while (auto mappedValue = mapping.lookupOrNull(from))
100  from = mappedValue;
101  return from;

214 void ArgConverter::notifyOpRemoved(Operation *op) {
215  for (Region &region : op->getRegions()) {
216  for (Block &block : region) {
217  // Drop any rewrites from within.
218  for (Operation &nestedOp : block)
219  if (nestedOp.getNumRegions())
220  notifyOpRemoved(&nestedOp);
221 
222  // Check if this block was converted.
223  auto it = conversionInfo.find(&block);
224  if (it == conversionInfo.end())
225  return;
226 
227  // Drop all uses of the original arguments and delete the original block.
228  Block *origBlock = it->second.origBlock;
229  for (BlockArgument arg : origBlock->getArguments())
230  arg.dropAllUses();
231  conversionInfo.erase(it);
232  }
233  }
234 }

236 void ArgConverter::discardRewrites(Block *block) {
237  auto it = conversionInfo.find(block);
238  if (it == conversionInfo.end())
239  return;
240  Block *origBlock = it->second.origBlock;
241 
242  // Drop all uses of the new block arguments and replace uses of the new block.
243  for (int i = block->getNumArguments() - 1; i >= 0; --i)
244  block->getArgument(i).dropAllUses();
245  block->replaceAllUsesWith(origBlock);
246 
247  // Move the operations back the original block and the delete the new block.
248  origBlock->getOperations().splice(origBlock->end(), block->getOperations());
249  origBlock->moveBefore(block);
250  block->erase();
251 
252  conversionInfo.erase(it);
253 }

255 void ArgConverter::applyRewrites(ConversionValueMapping &mapping) {
256  for (auto &info : conversionInfo) {
257  Block *newBlock = info.first;
258  ConvertedBlockInfo &blockInfo = info.second;
259  Block *origBlock = blockInfo.origBlock;
260 
261  // Process the remapping for each of the original arguments.
262  for (unsigned i = 0, e = origBlock->getNumArguments(); i != e; ++i) {
263  Optional<ConvertedArgInfo> &argInfo = blockInfo.argInfo[i];
264  BlockArgument origArg = origBlock->getArgument(i);
265 
266  // Handle the case of a 1->0 value mapping.
267  if (!argInfo) {
268  // If a replacement value was given for this argument, use that to
269  // replace all uses.
270  auto argReplacementValue = mapping.lookupOrDefault(origArg);
271  if (argReplacementValue != origArg) {
272  origArg.replaceAllUsesWith(argReplacementValue);
273  continue;
274  }
275  // If there are any dangling uses then replace the argument with one
276  // generated by the type converter. This is necessary as the cast must
277  // persist in the IR after conversion.
278  if (!origArg.use_empty()) {
279  rewriter.setInsertionPointToStart(newBlock);
280  auto *newOp = typeConverter->materializeConversion(
281  rewriter, origArg.getType(), llvm::None, loc);
282  origArg.replaceAllUsesWith(newOp->getResult(0));
283  }
284  continue;
285  }
286 
287  // If mapping is 1-1, replace the remaining uses and drop the cast
288  // operation.
289  // FIXME(riverriddle) This should check that the result type and operand
290  // type are the same, otherwise it should force a conversion to be
291  // materialized.
292  if (argInfo->newArgSize == 1) {
293  origArg.replaceAllUsesWith(
294  mapping.lookupOrDefault(newBlock->getArgument(argInfo->newArgIdx)));
295  continue;
296  }
297 
298  // Otherwise this is a 1->N value mapping.
299  Value castValue = argInfo->castValue;
300  assert(argInfo->newArgSize > 1 && castValue && "expected 1->N mapping");
301 
302  // If the argument is still used, replace it with the generated cast.
303  if (!origArg.use_empty())
304  origArg.replaceAllUsesWith(mapping.lookupOrDefault(castValue));
305 
306  // If all users of the cast were removed, we can drop it. Otherwise, keep
307  // the operation alive and let the user handle any remaining usages.
308  if (castValue.use_empty())
309  castValue.getDefiningOp()->erase();
310  }
311  }

324 Block *ArgConverter::applySignatureConversion(
325  Block *block, TypeConverter::SignatureConversion &signatureConversion,
326  ConversionValueMapping &mapping) {
327  // If no arguments are being changed or added, there is nothing to do.
328  unsigned origArgCount = block->getNumArguments();
329  auto convertedTypes = signatureConversion.getConvertedTypes();
330  if (origArgCount == 0 && convertedTypes.empty())
331  return block;
332 
333  // Split the block at the beginning to get a new block to use for the updated
334  // signature.
335  Block *newBlock = block->splitBlock(block->begin());
336  block->replaceAllUsesWith(newBlock);
337 
338  SmallVector<Value, 4> newArgRange(newBlock->addArguments(convertedTypes));
339  ArrayRef<Value> newArgs(newArgRange);
340 
341  // Remap each of the original arguments as determined by the signature
342  // conversion.
343  ConvertedBlockInfo info(block);
344  info.argInfo.resize(origArgCount);
345 
346  OpBuilder::InsertionGuard guard(rewriter);
347  rewriter.setInsertionPointToStart(newBlock);
348  for (unsigned i = 0; i != origArgCount; ++i) {
349  auto inputMap = signatureConversion.getInputMapping(i);
350  if (!inputMap)
351  continue;
352  BlockArgument origArg = block->getArgument(i);
353 
354  // If inputMap->replacementValue is not nullptr, then the argument is
355  // dropped and a replacement value is provided to be the remappedValue.
356  if (inputMap->replacementValue) {
357  assert(inputMap->size == 0 &&
358  "invalid to provide a replacement value when the argument isn't "
359  "dropped");
360  mapping.map(origArg, inputMap->replacementValue);
361  continue;
362  }
363 
364  // If this is a 1->1 mapping, then map the argument directly.
365  if (inputMap->size == 1) {
366  mapping.map(origArg, newArgs[inputMap->inputNo]);
367  info.argInfo[i] = ConvertedArgInfo(inputMap->inputNo, inputMap->size);
368  continue;
369  }
370 
371  // Otherwise, this is a 1->N mapping. Call into the provided type converter
372  // to pack the new values.
373  auto replArgs = newArgs.slice(inputMap->inputNo, inputMap->size);
374  Operation *cast = typeConverter->materializeConversion(
375  rewriter, origArg.getType(), replArgs, loc);
376  assert(cast->getNumResults() == 1 &&
377  cast->getNumOperands() == replArgs.size());
378  mapping.map(origArg, cast->getResult(0));
379  info.argInfo[i] =
380  ConvertedArgInfo(inputMap->inputNo, inputMap->size, cast->getResult(0));
381  }
382 
383  // Remove the original block from the region and return the new one.
384  insertConversion(newBlock, std::move(info));
385  return newBlock;
386 }

388 void ArgConverter::insertConversion(Block *newBlock,
389  ConvertedBlockInfo &&info) {
390  // Get a region to insert the old block.
391  Region *region = newBlock->getParent();
392  std::unique_ptr<Region> &mappedRegion = regionMapping[region];
393  if (!mappedRegion)
394  mappedRegion = std::make_unique<Region>(region->getParentOp());
395 
396  // Move the original block to the mapped region and emplace the conversion.
397  mappedRegion->getBlocks().splice(mappedRegion->end(), region->getBlocks(),
398  info.origBlock->getIterator());
399  conversionInfo.insert({newBlock, std::move(info)});
400 }

627 void ConversionPatternRewriterImpl::resetState(RewriterState state) {
628  // Reset any operations that were updated in place.
629  for (unsigned i = state.numRootUpdates, e = rootUpdates.size(); i != e; ++i)
630  rootUpdates[i].resetOperation();
631  rootUpdates.resize(state.numRootUpdates);
632 
633  // Undo any block actions.
634  undoBlockActions(state.numBlockActions);
635 
636  // Reset any replaced operations and undo any saved mappings.
637  for (auto &repl : llvm::drop_begin(replacements, state.numReplacements))
638  for (auto result : repl.op->getResults())
639  mapping.erase(result);
640  replacements.resize(state.numReplacements);
641 
642  // Pop all of the newly created operations.
643  while (createdOps.size() != state.numCreatedOps) {
644  createdOps.back()->erase();
645  createdOps.pop_back();
646  }
647 
648  // Pop all of the recorded ignored operations that are no longer valid.
649  while (ignoredOps.size() != state.numIgnoredOperations)
650  ignoredOps.pop_back();
651 }

653 void ConversionPatternRewriterImpl::undoBlockActions(
654  unsigned numActionsToKeep) {
655  for (auto &action :
656  llvm::reverse(llvm::drop_begin(blockActions, numActionsToKeep))) {
657  switch (action.kind) {
658  // Delete the created block.
659  case BlockActionKind::Create: {
660  // Unlink all of the operations within this block, they will be deleted
661  // separately.
662  auto &blockOps = action.block->getOperations();
663  while (!blockOps.empty())
664  blockOps.remove(blockOps.begin());
665  action.block->dropAllDefinedValueUses();
666  action.block->erase();
667  break;
668  }
669  // Move the block back to its original position.
670  case BlockActionKind::Move: {
671  Region *originalRegion = action.originalPosition.region;
672  originalRegion->getBlocks().splice(
673  std::next(originalRegion->begin(), action.originalPosition.position),
674  action.block->getParent()->getBlocks(), action.block);
675  break;
676  }
677  // Merge back the block that was split out.
678  case BlockActionKind::Split: {
679  action.originalBlock->getOperations().splice(
680  action.originalBlock->end(), action.block->getOperations());
681  action.block->dropAllDefinedValueUses();
682  action.block->erase();
683  break;
684  }
685  // Undo the type conversion.
686  case BlockActionKind::TypeConversion: {
687  argConverter.discardRewrites(action.block);
688  break;
689  }
690  }
691  }
692  blockActions.resize(numActionsToKeep);
693 }

695 void ConversionPatternRewriterImpl::discardRewrites() {
696  // Reset any operations that were updated in place.
697  for (auto &state : rootUpdates)
698  state.resetOperation();
699 
700  undoBlockActions();
701 
702  // Remove any newly created ops.
703  for (auto *op : llvm::reverse(createdOps))
704  op->erase();
705 }

707 void ConversionPatternRewriterImpl::applyRewrites() {
708  // Apply all of the rewrites replacements requested during conversion.
709  for (auto &repl : replacements) {
710  for (unsigned i = 0, e = repl.newValues.size(); i != e; ++i) {
711  if (auto newValue = repl.newValues[i])
712  repl.op->getResult(i).replaceAllUsesWith(
713  mapping.lookupOrDefault(newValue));
714  }
715 
716  // If this operation defines any regions, drop any pending argument
717  // rewrites.
718  if (argConverter.typeConverter && repl.op->getNumRegions())
719  argConverter.notifyOpRemoved(repl.op);
720  }
721 
722  // In a second pass, erase all of the replaced operations in reverse. This
723  // allows processing nested operations before their parent region is
724  // destroyed.
725  for (auto &repl : llvm::reverse(replacements))
726  repl.op->erase();
727 
728  argConverter.applyRewrites(mapping);
729 }

966 /// Attempt to match and rewrite the IR root at the specified operation.
967 PatternMatchResult
968 ConversionPattern::matchAndRewrite(Operation *op,
969  PatternRewriter &rewriter) const {
970  SmallVector<Value, 4> operands;
971  auto &dialectRewriter = static_cast<ConversionPatternRewriter &>(rewriter);
972  dialectRewriter.getImpl().remapValues(op->getOperands(), operands);
973 
974  // If this operation has no successors, invoke the rewrite directly.
975  if (op->getNumSuccessors() == 0)
976  return matchAndRewrite(op, operands, dialectRewriter);
977 
978  // Otherwise, we need to remap the successors.
979  SmallVector<Block *, 2> destinations;
980  destinations.reserve(op->getNumSuccessors());
981 
982  SmallVector<ArrayRef<Value>, 2> operandsPerDestination;
983  unsigned firstSuccessorOperand = op->getSuccessorOperandIndex(0);
984  for (unsigned i = 0, seen = 0, e = op->getNumSuccessors(); i < e; ++i) {
985  destinations.push_back(op->getSuccessor(i));
986 
987  // Lookup the successors operands.
988  unsigned n = op->getNumSuccessorOperands(i);
989  operandsPerDestination.push_back(
990  llvm::makeArrayRef(operands.data() + firstSuccessorOperand + seen, n));
991  seen += n;
992  }
993 
994  // Rewrite the operation.
995  return matchAndRewrite(
996  op,
997  llvm::makeArrayRef(operands.data(),
998  operands.data() + firstSuccessorOperand),
999  destinations, operandsPerDestination, dialectRewriter);
1000 }

1008 using LegalizationPatterns = SmallVector<RewritePattern *, 1>;

1010 /// This class defines a recursive operation legalizer.
1011 class OperationLegalizer {
1012 public:
1013  using LegalizationAction = ConversionTarget::LegalizationAction;
1014 
1015  OperationLegalizer(ConversionTarget &targetInfo,
1016  const OwningRewritePatternList &patterns)
1017  : target(targetInfo) {
1018  buildLegalizationGraph(patterns);
1019  computeLegalizationGraphBenefit();
1020  }
1021

1023  bool isIllegal(Operation *op) const;
1024 
1025  /// Attempt to legalize the given operation. Returns success if the operation
1026  /// was legalized, failure otherwise.
1027  LogicalResult legalize(Operation *op, ConversionPatternRewriter &rewriter);
1028 
1029  /// Returns the conversion target in use by the legalizer.
1030  ConversionTarget &getTarget() { return target; }
1031 
1032 private:
1033  /// Attempt to legalize the given operation by folding it.
1034  LogicalResult legalizeWithFold(Operation *op,
1035  ConversionPatternRewriter &rewriter);
1036

1038  /// Returns success if the operation was legalized, failure otherwise.
1039  LogicalResult legalizePattern(Operation *op, RewritePattern *pattern,
1040  ConversionPatternRewriter &rewriter);
1041 
1042  /// Build an optimistic legalization graph given the provided patterns. This
1043  /// function populates 'legalizerPatterns' with the operations that are not
1044  /// directly legal, but may be transitively legal for the current target given
1045  /// the provided patterns.
1046  void buildLegalizationGraph(const OwningRewritePatternList &patterns);
1047 
1048  /// Compute the benefit of each node within the computed legalization graph.
1049  /// This orders the patterns within 'legalizerPatterns' based upon two
1050  /// criteria:
1051  /// 1) Prefer patterns that have the lowest legalization depth, i.e.
1052  /// represent the more direct mapping to the target.
1053  /// 2) When comparing patterns with the same legalization depth, prefer the
1054  /// pattern with the highest PatternBenefit. This allows for users to
1055  /// prefer specific legalizations over others.
1056  void computeLegalizationGraphBenefit();
1057 
1058  /// The current set of patterns that have been applied.
1059  SmallPtrSet<RewritePattern *, 8> appliedPatterns;
1060 
1061  /// The set of legality information for operations transitively supported by
1062  /// the target.
1063  DenseMap<OperationName, LegalizationPatterns> legalizerPatterns;
1064 
1065  /// The legalization information provided by the target.
1066  ConversionTarget &target;
1067 };

1075 LogicalResult
1076 OperationLegalizer::legalize(Operation *op,
1077  ConversionPatternRewriter &rewriter) {
1078  LLVM_DEBUG(llvm::dbgs() << "Legalizing operation : " << op->getName()
1079  << "\n");
1080 
1081  // Check if this operation is legal on the target.
1082  if (auto legalityInfo = target.isLegal(op)) {
1083  LLVM_DEBUG(llvm::dbgs()
1084  << "-- Success : Operation marked legal by the target\n");
1085  // If this operation is recursively legal, mark its children as ignored so
1086  // that we don't consider them for legalization.
1087  if (legalityInfo->isRecursivelyLegal) {
1088  LLVM_DEBUG(llvm::dbgs() << "-- Success : Operation is recursively legal; "
1089  "Skipping internals\n");
1090  rewriter.getImpl().markNestedOpsIgnored(op);
1091  }
1092  return success();
1093  }
1094 
1095  // Check to see if the operation is ignored and doesn't need to be converted.
1096  if (rewriter.getImpl().isOpIgnored(op)) {
1097  LLVM_DEBUG(llvm::dbgs()
1098  << "-- Success : Operation marked ignored during conversion\n");
1099  return success();
1100  }
1101 
1102  // If the operation isn't legal, try to fold it in-place.
1103  // TODO(riverriddle) Should we always try to do this, even if the op is
1104  // already legal?
1105  if (succeeded(legalizeWithFold(op, rewriter))) {
1106  LLVM_DEBUG(llvm::dbgs() << "-- Success : Operation was folded\n");
1107  return success();
1108  }
1109 
1110  // Otherwise, we need to apply a legalization pattern to this operation.
1111  auto it = legalizerPatterns.find(op->getName());
1112  if (it == legalizerPatterns.end()) {
1113  LLVM_DEBUG(llvm::dbgs() << "-- FAIL : no known legalization path.\n");
1114  return failure();
1115  }
1116 
1117  // The patterns are sorted by expected benefit, so try to apply each in-order.
1118  for (auto *pattern : it->second)
1119  if (succeeded(legalizePattern(op, pattern, rewriter)))
1120  return success();
1121 
1122  LLVM_DEBUG(llvm::dbgs() << "-- FAIL : no matched legalization pattern.\n");
1123  return failure();
1124 }

1126 LogicalResult
1127 OperationLegalizer::legalizeWithFold(Operation *op,
1128  ConversionPatternRewriter &rewriter) {
1129  auto &rewriterImpl = rewriter.getImpl();
1130  RewriterState curState = rewriterImpl.getCurrentState();
1131 
1132  // Try to fold the operation.
1133  SmallVector<Value, 2> replacementValues;
1134  rewriter.setInsertionPoint(op);
1135  if (failed(rewriter.tryFold(op, replacementValues)))
1136  return failure();
1137 
1138  // Insert a replacement for 'op' with the folded replacement values.
1139  rewriter.replaceOp(op, replacementValues);
1140 
1141  // Recursively legalize any new constant operations.
1142  for (unsigned i = curState.numCreatedOps, e = rewriterImpl.createdOps.size();
1143  i != e; ++i) {
1144  Operation *cstOp = rewriterImpl.createdOps[i];
1145  if (failed(legalize(cstOp, rewriter))) {
1146  LLVM_DEBUG(llvm::dbgs() << "-- FAIL: Generated folding constant '"
1147  << cstOp->getName() << "' was illegal.\n");
1148  rewriterImpl.resetState(curState);
1149  return failure();
1150  }
1151  }
1152  return success();
1153 }

1155 LogicalResult
1156 OperationLegalizer::legalizePattern(Operation *op, RewritePattern *pattern,
1157  ConversionPatternRewriter &rewriter) {
1158  LLVM_DEBUG({
1159  llvm::dbgs() << "-* Applying rewrite pattern '" << op->getName() << " -> (";
1160  interleaveComma(pattern->getGeneratedOps(), llvm::dbgs());
1161  llvm::dbgs() << ")'.\n";
1162  });
1163 
1164  // Ensure that we don't cycle by not allowing the same pattern to be
1165  // applied twice in the same recursion stack.
1166  // TODO(riverriddle) We could eventually converge, but that requires more
1167  // complicated analysis.
1168  if (!appliedPatterns.insert(pattern).second) {
1169  LLVM_DEBUG(llvm::dbgs() << "-- FAIL: Pattern was already applied.\n");
1170  return failure();
1171  }
1172 
1173  auto &rewriterImpl = rewriter.getImpl();
1174  RewriterState curState = rewriterImpl.getCurrentState();
1175  auto cleanupFailure = [&] {
1176  // Reset the rewriter state and pop this pattern.
1177  rewriterImpl.resetState(curState);
1178  appliedPatterns.erase(pattern);
1179  return failure();
1180  };
1181 
1182  // Try to rewrite with the given pattern.
1183  rewriter.setInsertionPoint(op);
1184  auto matchedPattern = pattern->matchAndRewrite(op, rewriter);
1185 #ifndef NDEBUG
1186  assert(rewriterImpl.pendingRootUpdates.empty() && "dangling root updates");
1187 #endif
1188 
1189  if (!matchedPattern) {
1190  LLVM_DEBUG(llvm::dbgs() << "-- FAIL: Pattern failed to match.\n");
1191  return cleanupFailure();
1192  }
1193 
1194  // If the pattern moved or created any blocks, try to legalize their types.
1195  // This ensures that the types of the block arguments are legal for the region
1196  // they were moved into.
1197  for (unsigned i = curState.numBlockActions,
1198  e = rewriterImpl.blockActions.size();
1199  i != e; ++i) {
1200  auto &action = rewriterImpl.blockActions[i];
1201  if (action.kind ==
1202  ConversionPatternRewriterImpl::BlockActionKind::TypeConversion)
1203  continue;
1204 
1205  // Convert the block signature.
1206  if (failed(rewriterImpl.convertBlockSignature(action.block))) {
1207  LLVM_DEBUG(llvm::dbgs()
1208  << "-- FAIL: failed to convert types of moved block.\n");
1209  return cleanupFailure();
1210  }
1211  }
1212 
1213  // Check all of the replacements to ensure that the pattern actually replaced
1214  // the root operation. We also mark any other replaced ops as 'dead' so that
1215  // we don't try to legalize them later.
1216  bool replacedRoot = false;
1217  for (unsigned i = curState.numReplacements,
1218  e = rewriterImpl.replacements.size();
1219  i != e; ++i) {
1220  Operation *replacedOp = rewriterImpl.replacements[i].op;
1221  if (replacedOp == op)
1222  replacedRoot = true;
1223  else
1224  rewriterImpl.ignoredOps.insert(replacedOp);
1225  }
1226 
1227  // Check that the root was either updated or replace.
1228  auto updatedRootInPlace = [&] {
1229  return llvm::any_of(
1230  llvm::drop_begin(rewriterImpl.rootUpdates, curState.numRootUpdates),
1231  [op](auto &state) { return state.getOperation() == op; });
1232  };
1233  (void)replacedRoot;
1234  (void)updatedRootInPlace;
1235  assert((replacedRoot || updatedRootInPlace()) &&
1236  "expected pattern to replace the root operation");
1237 
1238  // Recursively legalize each of the operations updated in place.
1239  for (unsigned i = curState.numRootUpdates,
1240  e = rewriterImpl.rootUpdates.size();
1241  i != e; ++i) {
1242  auto &state = rewriterImpl.rootUpdates[i];
1243  if (failed(legalize(state.getOperation(), rewriter))) {
1244  LLVM_DEBUG(llvm::dbgs() << "-- FAIL: Operation updated in-place '"
1245  << op->getName() << "' was illegal.\n");
1246  return cleanupFailure();
1247  }
1248  }
1249 
1250  // Recursively legalize each of the new operations.
1251  for (unsigned i = curState.numCreatedOps, e = rewriterImpl.createdOps.size();
1252  i != e; ++i) {
1253  Operation *op = rewriterImpl.createdOps[i];
1254  if (failed(legalize(op, rewriter))) {
1255  LLVM_DEBUG(llvm::dbgs() << "-- FAIL: Generated operation '"
1256  << op->getName() << "' was illegal.\n");
1257  return cleanupFailure();
1258  }
1259  }
1260 
1261  appliedPatterns.erase(pattern);
1262  return success();
1263 }
1264 
1265 void OperationLegalizer::buildLegalizationGraph(
1266  const OwningRewritePatternList &patterns) {
1267  // A mapping between an operation and a set of operations that can be used to
1268  // generate it.
1269  DenseMap<OperationName, SmallPtrSet<OperationName, 2>> parentOps;
1270  // A mapping between an operation and any currently invalid patterns it has.
1271  DenseMap<OperationName, SmallPtrSet<RewritePattern *, 2>> invalidPatterns;
1272  // A worklist of patterns to consider for legality.
1273  llvm::SetVector<RewritePattern *> patternWorklist;
1274 
1275  // Build the mapping from operations to the parent ops that may generate them.
1276  for (auto &pattern : patterns) {
1277  auto root = pattern->getRootKind();
1278 
1279  // Skip operations that are always known to be legal.
1280  if (target.getOpAction(root) == LegalizationAction::Legal)
1281  continue;
1282 
1283  // Add this pattern to the invalid set for the root op and record this root
1284  // as a parent for any generated operations.
1285  invalidPatterns[root].insert(pattern.get());
1286  for (auto op : pattern->getGeneratedOps())
1287  parentOps[op].insert(root);
1288 
1289  // Add this pattern to the worklist.
1290  patternWorklist.insert(pattern.get());
1291  }

1294  auto *pattern = patternWorklist.pop_back_val();
1295 
1296  // Check to see if any of the generated operations are invalid.
1297  if (llvm::any_of(pattern->getGeneratedOps(), [&](OperationName op) {
1298  Optional<LegalizationAction> action = target.getOpAction(op);
1299  return !legalizerPatterns.count(op) &&
1300  (!action || action == LegalizationAction::Illegal);
1301  }))
1302  continue;
1303 
1304  // Otherwise, if all of the generated operation are valid, this op is now
1305  // legal so add all of the child patterns to the worklist.
1306  legalizerPatterns[pattern->getRootKind()].push_back(pattern);
1307  invalidPatterns[pattern->getRootKind()].erase(pattern);
1308 
1309  // Add any invalid patterns of the parent operations to see if they have now
1310  // become legal.
1311  for (auto op : parentOps[pattern->getRootKind()])
1312  patternWorklist.set_union(invalidPatterns[op]);
1313  }

1316 void OperationLegalizer::computeLegalizationGraphBenefit() {
1317  // The smallest pattern depth, when legalizing an operation.
1318  DenseMap<OperationName, unsigned> minPatternDepth;
1319 
1320  // Compute the minimum legalization depth for a given operation.
1321  std::function<unsigned(OperationName)> computeDepth = [&](OperationName op) {
1322  // Check for existing depth.
1323  auto depthIt = minPatternDepth.find(op);
1324  if (depthIt != minPatternDepth.end())
1325  return depthIt->second;
1326 
1327  // If a mapping for this operation does not exist, then this operation
1328  // is always legal. Return 0 as the depth for a directly legal operation.
1329  auto opPatternsIt = legalizerPatterns.find(op);
1330  if (opPatternsIt == legalizerPatterns.end() || opPatternsIt->second.empty())
1331  return 0u;
1332 
1333  // Initialize the depth to the maximum value.
1334  unsigned minDepth = std::numeric_limits<unsigned>::max();
1335 
1336  // Record this initial depth in case we encounter this op again when
1337  // recursively computing the depth.
1338  minPatternDepth.try_emplace(op, minDepth);
1339 
1340  // Compute the depth for each pattern used to legalize this operation.
1341  SmallVector<std::pair<RewritePattern *, unsigned>, 4> patternsByDepth;
1342  patternsByDepth.reserve(opPatternsIt->second.size());
1343  for (RewritePattern *pattern : opPatternsIt->second) {
1344  unsigned depth = 0;
1345  for (auto generatedOp : pattern->getGeneratedOps())
1346  depth = std::max(depth, computeDepth(generatedOp) + 1);
1347  patternsByDepth.emplace_back(pattern, depth);
1348 
1349  // Update the min depth for this operation.
1350  minDepth = std::min(minDepth, depth);
1351  }
1352 
1353  // Update the pattern depth.
1354  minPatternDepth[op] = minDepth;
1355 
1356  // If the operation only has one legalization pattern, there is no need to
1357  // sort them.
1358  if (patternsByDepth.size() == 1)
1359  return minDepth;
1360 
1361  // Sort the patterns by those likely to be the most beneficial.
1362  llvm::array_pod_sort(
1363  patternsByDepth.begin(), patternsByDepth.end(),
1364  [](const std::pair<RewritePattern *, unsigned> *lhs,
1365  const std::pair<RewritePattern *, unsigned> *rhs) {
1366  // First sort by the smaller pattern legalization depth.
1367  if (lhs->second != rhs->second)
1368  return llvm::array_pod_sort_comparator<unsigned>(&lhs->second,
1369  &rhs->second);
1370 
1371  // Then sort by the larger pattern benefit.
1372  auto lhsBenefit = lhs->first->getBenefit();
1373  auto rhsBenefit = rhs->first->getBenefit();
1374  return llvm::array_pod_sort_comparator<PatternBenefit>(&rhsBenefit,
1375  &lhsBenefit);
1376  });
1377 
1378  // Update the legalization pattern to use the new sorted list.
1379  opPatternsIt->second.clear();
1380  for (auto &patternIt : patternsByDepth)
1381  opPatternsIt->second.push_back(patternIt.first);
1382 
1383  return minDepth;
1384  };
1385 
1386  // For each operation that is transitively legal, compute a cost for it.
1387  for (auto &opIt : legalizerPatterns)
1388  if (!minPatternDepth.count(opIt.first))
1389  computeDepth(opIt.first);
1390 }
