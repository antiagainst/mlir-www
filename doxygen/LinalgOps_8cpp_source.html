35 /// Determines whether it is possible to fold it away in the parent Linalg op:
36 ///
37 /// ```mlir
38 /// %1 = memref_cast %0 : memref<8x16xf32> to memref<?x?xf32>
39 /// %2 = linalg.slice %1 ... : memref<?x?xf32> ...
40 /// // or
41 /// %1 = memref_cast %0 : memref<8x16xf32, affine_map<(i, j)->(16 * i + j)>>
42 /// to memref<?x?xf32>
43 /// linalg.generic(%1 ...) : memref<?x?xf32> ...
44 /// ```
45 ///
46 /// into
47 ///
48 /// ```mlir
49 /// %2 = linalg.slice %0 ... : memref<8x16xf32> ...
50 /// // or
51 /// linalg.generic(%0 ... : memref<8x16xf32, affine_map<(i, j)->(16 * i + j)>>
52 /// ```
53 ///
54 static bool canFold(MemRefCastOp castOp) {
55  MemRefType sourceType = castOp.source().getType().dyn_cast<MemRefType>();
56  MemRefType resultType = castOp.getType().dyn_cast<MemRefType>();
57 
58  // If we don't have MemRefType as source and destination, bail out.
59  if (!sourceType || !resultType)
60  return false;
61 
62  // If resultType has a map, it needs to be the same as the source type to
63  // canonicalize.
64  if (!resultType.getAffineMaps().empty() &&
65  sourceType.getAffineMaps() != resultType.getAffineMaps())
66  return false;
67 
68  // Ensure that:
69  // 1. source is static
70  // 2. source and target have the same rank (will be extended when needed)
71  // 3. if result is partially static, ensure sizes match.
72  if (!sourceType.hasStaticShape() ||
73  sourceType.getRank() != resultType.getRank())
74  return false;
75 
76  for (auto it : llvm::zip(sourceType.getShape(), resultType.getShape())) {
77  auto sourceSize = std::get<0>(it);
78  auto resultSize = std::get<1>(it);
79  if (ShapedType::isDynamic(resultSize))
80  continue;
81  if (sourceSize != resultSize)
82  return false;
83  }
84 
85  // If source has a map, it can only canonicalize if it is the canonical
86  // strided layout map.
87  if (sourceType.getAffineMaps().empty())
88  return true;
89 
90  int64_t offset;
91  SmallVector<int64_t, 4> strides;
92  auto res = getStridesAndOffset(sourceType, strides, offset);
93  (void)res;
94  assert(succeeded(res));
95  auto stridedMap =
96  makeStridedLinearLayoutMap(strides, offset, castOp.getContext());
97  AffineMap sourceMap = sourceType.getAffineMaps().front();
98  return sourceMap == stridedMap;
99 }

101 /// This is a common class used for patterns of the form
102 /// ```
103 /// someop(memrefcast) -> someop
104 /// ```
105 /// It folds the source of any memref_cast into the root operation directly.
106 static LogicalResult foldMemRefCast(Operation *op) {
107  bool folded = false;
108  for (OpOperand &operand : op->getOpOperands()) {
109  auto castOp = dyn_cast_or_null<MemRefCastOp>(operand.get().getDefiningOp());
110  if (castOp && canFold(castOp)) {
111  operand.set(castOp.getOperand());
112  folded = true;
113  }
114  }
115  return success(folded);
116 }

155 static ParseResult parseGenericOp(OpAsmParser &parser, OperationState &result) {
156  SmallVector<OpAsmParser::OperandType, 8> operandsInfo, regionOperandsInfo;
157  DictionaryAttr dictAttr;
158  // Parse the core linalg traits that must check into a dictAttr.
159  // The name is unimportant as we will overwrite result.attributes.
160  // The core linalg traits must contain the information necessary to pass the
161  // verifier.
162  if (parser.parseAttribute(dictAttr, "_", result.attributes) ||
163  parser.parseOperandList(operandsInfo))
164  return failure();
165  result.attributes.assign(dictAttr.getValue().begin(),
166  dictAttr.getValue().end());
167 
168  Region &region = *result.addRegion();
169  SmallVector<Type, 8> operandTypes, regionTypes;
170  // Optional attributes may be added.
171  // Either Optional getFunAttrName() attribute or region must be specified.
172  if (!dictAttr.get(getFunAttrName()) &&
173  parser.parseOptionalRegion(region, regionOperandsInfo, regionTypes))
174  return failure();
175  if (parser.parseOptionalAttrDict(result.attributes) ||
176  parser.parseColonTypeList(operandTypes))
177  return failure();
178  // Generic ops may specify that a subset of its outputs are tensors. Such
179  // outputs are specified in the result type.
180  SmallVector<Type, 8> tensorResultTypes;
181  if (parser.parseOptionalArrowTypeList(tensorResultTypes))
182  return failure();
183  if (!tensorResultTypes.empty())
184  result.addTypes(tensorResultTypes);
185  return parser.resolveOperands(operandsInfo, operandTypes,
186  parser.getCurrentLocation(), result.operands);

314 template <typename GenericOpType>
315 static LogicalResult verifyGenericOp(GenericOpType op) {
316  auto nInputViews = op.getNumInputs();
317  auto nLoops = op.getNumLoops();
318  auto nInputsAndOutputBuffers = op.getNumInputsAndOutputBuffers();
319  if (nInputsAndOutputBuffers != llvm::size(op.views()))
320  return op.emitOpError("expected exactly ")
321  << nInputsAndOutputBuffers
322  << " inputs (tensor or buffer) and output buffer operands";
323 
324  auto &region = op.region();
325  auto funOp = op.getFunction();
326  auto funType = funOp ? funOp.getType() : FunctionType();
327  if (!region.empty()) {
328  if (region.getBlocks().size() != 1)
329  return op.emitOpError("expected region with 1 block");
330  if (failed(verifyBlockArgs(op, region.getBlocks().front())))
331  return failure();
332  } else {
333  if (!funOp || !funOp.getType())
334  return op.emitOpError(
335  "expected function attribute to refer to a defined symbol");
336  if (failed(verifyFuncArgsGeneric(op, funType)))
337  return failure();
338  }
339 
340  SmallVector<AffineMap, 4> indexingMaps;
341  indexingMaps.reserve(op.indexing_maps().size());
342  for (auto en : llvm::enumerate(op.indexing_maps())) {
343  auto idx = en.index();
344  auto m = en.value().template cast<AffineMapAttr>().getValue();
345  indexingMaps.push_back(m); // Save reference to map for further checks.
346  auto view = (idx < nInputViews) ? op.getInputShapedType(idx)
347  : op.getOutputShapedType(idx - nInputViews);
348 
349  if (m.getNumSymbols() != 0)
350  return op.emitOpError("expected indexing_map #")
351  << idx << " to have no symbols";
352 
353  if (m.getNumDims() != nLoops)
354  return op.emitOpError("expected indexing_map #")
355  << idx << " to have " << nLoops
356  << " dim(s) to match the number of loops";
357 
358  if (m.getNumResults() == 1 && view.getRank() == 0) {
359  auto cst = m.getResult(0).template dyn_cast<AffineConstantExpr>();
360  if (!cst || cst.getValue() != 0)
361  return op.emitOpError("expected indexing_map #")
362  << idx << " to be 0 to match 0-D view: " << view;
363  }
364 
365  if (m.getNumResults() != view.getRank())
366  return op.emitOpError("expected indexing_map #")
367  << idx << " results to match view rank: " << view;
368  }
369 
370  auto concatMap = concatAffineMaps(indexingMaps);
371  auto aggregateMap = inversePermutation(concatMap);
372  if (!aggregateMap)
373  return op.emitOpError("expected the concatenation of maps in indexing_map "
374  "to be invertible");
375 
376  return success();
377 }

386 /// Return true if the reassociation specification is valid, false otherwise.
387 /// When false, the `invalidIndex` integer pointer is optionally filled with the
388 /// index of the offending reassociation map.
389 static bool isReassociationValid(ArrayRef<AffineMap> reassociation,
390  int *invalidIndex = nullptr) {
391  if (reassociation.empty())
392  return true;
393  unsigned nDims = reassociation[0].getNumDims();
394  unsigned nextExpectedDim = 0;
395  for (auto it : llvm::enumerate(reassociation)) {
396  auto m = it.value();
397  if (m.getNumDims() != nDims || m.getNumSymbols() != 0) {
398  if (invalidIndex)
399  *invalidIndex = it.index();
400  return false;
401  }
402  for (auto e : m.getResults()) {
403  auto d = e.dyn_cast<AffineDimExpr>();
404  if (!d || d.getPosition() != nextExpectedDim++) {
405  if (invalidIndex)
406  *invalidIndex = it.index();
407  return false;
408  }
409  }
410  }
411  if (nextExpectedDim != nDims) {
412  if (invalidIndex)
413  *invalidIndex = reassociation.size() - 1;
414  return false;
415  }
416  return true;
417 }

419 /// Detect whether memref dims [dim, dim + extent) can be reshaped without
420 /// copies.
421 static bool isReshapableDimBand(unsigned dim, unsigned extent,
422  ArrayRef<int64_t> sizes,
423  ArrayRef<AffineExpr> strides) {
424  assert(sizes.size() == strides.size() && "mismatched ranks");
425  // off by 1 indexing to avoid out of bounds
426  // V
427  for (auto idx = dim, e = dim + extent; idx + 1 < e; ++idx) {
428  // Only bands of static shapes are reshapable. This is due to the fact that
429  // there is no relation between dynamic sizes and dynamic strides: we do not
430  // have enough information to know whether a "-1" size corresponds to the
431  // proper symbol in the AffineExpr of a stride.
432  if (ShapedType::isDynamic(sizes[dim + 1]))
433  return false;
434  // TODO(ntv) Refine this by passing the proper nDims and nSymbols so we can
435  // simplify on the fly and catch more reshapable cases.
436  if (strides[idx] != strides[idx + 1] * sizes[idx + 1])
437  return false;
438  }
439  return true;
440 }

442 /// Compute the MemRefType obtained by applying the `reassociation` (which is
443 /// expected to be valid) to `type`.
444 /// If `type` is Contiguous MemRefType, this always produce a contiguous
445 /// MemRefType.
446 static MemRefType
447 computeReshapeCollapsedType(MemRefType type,
448  ArrayRef<AffineMap> reassociation) {
449  auto sizes = type.getShape();
450  AffineExpr offset;
451  SmallVector<AffineExpr, 4> strides;
452  auto status = getStridesAndOffset(type, strides, offset);
453  (void)status;
454  assert(succeeded(status) && "expected strided memref");
455 
456  SmallVector<int64_t, 4> newSizes;
457  newSizes.reserve(reassociation.size());
458  SmallVector<AffineExpr, 4> newStrides;
459  newStrides.reserve(reassociation.size());
460 
461  // Use the fact that reassociation is valid to simplify the logic: only use
462  // each map's rank.
463  assert(isReassociationValid(reassociation) && "invalid reassociation");
464  unsigned currentDim = 0;
465  for (AffineMap m : reassociation) {
466  unsigned dim = m.getNumResults();
467  int64_t size = 1;
468  AffineExpr stride = strides[currentDim + dim - 1];
469  if (!isReshapableDimBand(currentDim, dim, sizes, strides)) {
470  size = ShapedType::kDynamicSize;
471  stride = AffineExpr();
472  } else {
473  for (unsigned d = 0; d < dim; ++d)
474  size *= sizes[currentDim + d];
475  }
476  newSizes.push_back(size);
477  newStrides.push_back(stride);
478  currentDim += dim;
479  }
480 
481  // Early-exit: if `type` is contiguous, the result must be contiguous.
482  if (canonicalizeStridedLayout(type).getAffineMaps().empty())
483  return MemRefType::Builder(type).setShape(newSizes).setAffineMaps({});
484 
485  // Convert back to int64_t because we don't have enough information to create
486  // new strided layouts from AffineExpr only. This corresponds to a case where
487  // copies may be necessary.
488  int64_t intOffset = ShapedType::kDynamicStrideOrOffset;
489  if (auto o = offset.dyn_cast<AffineConstantExpr>())
490  intOffset = o.getValue();
491  SmallVector<int64_t, 4> intStrides;
492  intStrides.reserve(strides.size());
493  for (auto stride : newStrides) {
494  if (auto cst = stride.dyn_cast_or_null<AffineConstantExpr>())
495  intStrides.push_back(cst.getValue());
496  else
497  intStrides.push_back(ShapedType::kDynamicStrideOrOffset);
498  }
499  auto layout =
500  makeStridedLinearLayoutMap(intStrides, intOffset, type.getContext());
501  return canonicalizeStridedLayout(
502  MemRefType::Builder(type).setShape(newSizes).setAffineMaps({layout}));
503 }

562 static LogicalResult verify(ReshapeOp op) {
563  MemRefType expandedType = op.getViewType();
564  MemRefType collapsedType = op.getResult().getType().cast<MemRefType>();
565  unsigned expandedRank = expandedType.getRank();
566  unsigned collapsedRank = collapsedType.getRank();
567  bool isCollapse = expandedRank > collapsedRank;
568  if (!isCollapse) {
569  std::swap(expandedRank, collapsedRank);
570  std::swap(expandedType, collapsedType);
571  }
572  if (expandedRank == 0 || collapsedRank == 0)
573  return op.emitOpError("expected non-zero memref ranks");
574  if (expandedRank == collapsedRank)
575  return op.emitOpError("expected to collapse or expand dims");
576 
577  if (collapsedRank != op.reassociation().size())
578  return op.emitOpError("expected rank of the collapsed view(")
579  << collapsedRank << ") to be the number of reassociation maps("
580  << op.reassociation().size() << ")";

592  if (collapsedType != expectedType)

924 // Note: both functions below would completely disappear with a simple tensor
925 // kernel language.
926 //
927 // Ideally this should all be Tablegen'd but there is no good story for
928 // AffineMap for now.
929 SmallVector<AffineMap, 4> mlir::linalg::loopToOperandRangesMaps(Operation *op) {
930  MLIRContext *context = op->getContext();
931  if (auto copyOp = dyn_cast<CopyOp>(op)) {
932  // I(input_perm(ivs)) -> O(output_perm(ivs))
933  auto maybeInputMap = copyOp.inputPermutation();
934  auto maybeOutputMap = copyOp.outputPermutation();
935  unsigned inputRank = copyOp.getInputShapedType(0).getRank();
936  unsigned outputRank = copyOp.getOutputShapedType(0).getRank();
937  return SmallVector<AffineMap, 4>{
938  extractOrIdentityMap(maybeInputMap, inputRank, context),
939  extractOrIdentityMap(maybeOutputMap, outputRank, context)};
940  }
941  if (auto fillOp = dyn_cast<FillOp>(op)) {
942  // filling_value -> O(ivs)
943  unsigned rank = fillOp.getNumParallelLoops();
944  return SmallVector<AffineMap, 4>{
945  extractOrIdentityMap(llvm::None, rank, context)};
946  }
947  auto i = getAffineDimExpr(0, context);
948  auto j = getAffineDimExpr(1, context);
949  auto k = getAffineDimExpr(2, context);
950  if (isa<DotOp>(op))
951  // A(r_i) * B(r_i) -> C()
952  return SmallVector<AffineMap, 4>{AffineMap::get(1, 0, {i}),
953  AffineMap::get(1, 0, {i}), AffineMap()};
954  if (isa<MatvecOp>(op))
955  // A(i, r_j) * B(r_j) -> C(i)
956  return SmallVector<AffineMap, 4>{AffineMap::get(2, 0, {i, j}),
957  AffineMap::get(2, 0, {j}),
958  AffineMap::get(2, 0, {i})};
959  if (isa<MatmulOp>(op))
960  // A(i, r_k) * B(r_k, j) -> C(i, j)
961  return SmallVector<AffineMap, 4>{AffineMap::get(3, 0, {i, k}),
962  AffineMap::get(3, 0, {k, j}),
963  AffineMap::get(3, 0, {i, j})};
964  if (auto convOp = dyn_cast<ConvOp>(op)) {
965  // F(z0, ..., zN-1, q, k) * I(b, x0 + z0, ..., xN-1 + zN-1, q) ->
966  // O(b, x0, ..., xN-1, k)
967  // for N equal to `nWindow`.
968  auto nWin = convOp.getNumWindowLoops();
969  assert(nWin > 0 && "expected at least one window dimension");
970  unsigned idx = 0;
971  // In the following, AffineDimExprs are indexed in loop order:
972  // [ b, xs, k, q, zs]
973  // parallels non-window reductions windows
974  //
975  // Parallel dims are exactly the dimensions indexing `output`:
976  // output[b, x[0], ..., x[N-1], k]; i.e.
977  // * batch dimensions (bs with #bs = 1 for now)
978  // * "image" dimensions (xs with #xs = #zs = output_rank - #bs - #ks)
979  // * output filter dimensions (ks with #ks = 1 for now)
980  auto bs = makeAffineDimExprs(convOp.getNumBatchDimensions(), idx, context);
981  auto xs = makeAffineDimExprs(nWin, idx, context);
982  auto ks = makeAffineDimExprs(convOp.getNumOutputFeatureDimensions(), idx,
983  context);
984  // Non-window reduction dim: sum_{z[0], ..., z[N-1], q}
985  auto qs =
986  makeAffineDimExprs(convOp.getNumInputFeatureDimensions(), idx, context);
987  // Window reduction dims: sum_{z[0], ..., z[N-1], q}
988  auto zs = makeAffineDimExprs(nWin, idx, context);
989  // Construct the weighedSum expression.
990  auto ws = weightedConvInputIndex(convOp, xs, zs);
991  return SmallVector<AffineMap, 4>{
992  // filter[z[0], ..., z[N-1], q, k]
993  AffineMap::get(idx, 0, concat(concat(zs, qs), ks)),
994  // input[b,
995  // x[0]*s[0] + d[0]*z[0], ..., x[N-1]*s[N-1] + d[N-1]*z[N-1],
996  // q]
997  AffineMap::get(idx, 0, concat(concat(bs, ws), qs)),
998  // output[b, x[0], ..., x[N-1], k]
999  AffineMap::get(idx, 0, concat(concat(bs, xs), ks))};
1000  } else if (auto genericOp = dyn_cast<GenericOp>(op)) {
1001  SmallVector<AffineMap, 4> res;
1002  unsigned nViews = genericOp.getNumInputsAndOutputs();
1003  res.reserve(nViews);
1004  for (unsigned i = 0, e = nViews; i < e; ++i) {
1005  res.push_back(genericOp.getIndexingMap(i));
1006  }
1007  return res;
1008  } else if (auto indexedGenericOp = dyn_cast<IndexedGenericOp>(op)) {
1009  SmallVector<AffineMap, 4> res;
1010  unsigned nViews = indexedGenericOp.getNumInputsAndOutputs();
1011  res.reserve(nViews);
1012  for (unsigned i = 0, e = nViews; i < e; ++i)
1013  res.push_back(indexedGenericOp.getIndexingMap(i));
1014  return res;
1015  }
1016  llvm_unreachable("Missing loopToOperandRangesMaps for op");
1017 }
