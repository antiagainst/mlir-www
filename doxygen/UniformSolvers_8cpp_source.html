17 bool UniformParamsFromMinMaxSolver::compute() {
18  // Compute adjMin, adjMax, clamping to ensure that they straddle zero.
19  if (boundingMin > 0 && boundingMax >= boundingMin) {
20  // Lop-sided to the positive.
21  adjMin = 0;
22  adjMax = boundingMax;
23  } else if (boundingMax < 0 && boundingMin <= boundingMax) {
24  // Lop-sided to the negative.
25  adjMin = boundingMin;
26  adjMax = 0;
27  } else if (boundingMin <= 0 && boundingMax >= 0) {
28  adjMin = boundingMin;
29  adjMax = boundingMax;
30  } else {
31  // Illegal bounds.
32  return satisfied = false;
33  }
34 
35  const double origMinAdj = adjMin;
36  const double origMaxAdj = adjMax;
37  const double numLevelsDouble = storageParams.numLevels;
38 
39  struct fns {
40  static std::pair<double, double>
41  computeMinMax(double boundingMin, double numLevels, double delta) {
42  double adjMin = delta * std::floor(boundingMin / delta);
43  return std::make_pair(adjMin, adjMin + numLevels * delta);
44  }
45  static double overshoot(double boundingMin, double boundingMax,
46  double numLevels, double delta) {
47  auto adjMinMax = computeMinMax(boundingMin, numLevels, delta);
48  double maxOvershoot = adjMinMax.second - boundingMax;
49  double minOvershoot = boundingMin - adjMinMax.first;
50  // If undershooting on the min or max end, return that because it is
51  // to be unconditionally avoided. Otherwise return the end with the
52  // greatest magnitude of overshoot.
53  if (maxOvershoot < 0)
54  return maxOvershoot;
55  if (minOvershoot < 0)
56  return minOvershoot;
57  return std::max(maxOvershoot, minOvershoot);
58  }
59  };
60 
61  // Bisect to find a suitable delta, starting with bounds of deltaInit
62  // and deltaMax.
63  double deltaInit = (adjMax - adjMin) / numLevelsDouble;
64  double deltaMax =
65  ((numLevelsDouble * deltaInit) + 2 * deltaInit) / numLevelsDouble;
66  double deltaMid;
67  double prevDeltaMid = 0.0;
68  for (stepCount = 0; stepCount < 60; ++stepCount) {
69  deltaMid = (deltaInit + deltaMax) / 2.0;
70  auto fInit =
71  fns::overshoot(origMinAdj, origMaxAdj, numLevelsDouble, deltaInit);
72  auto fMid =
73  fns::overshoot(origMinAdj, origMaxAdj, numLevelsDouble, deltaMid);
74  if (fMid == 0 || (fMid > 0 && std::fabs(deltaMid - prevDeltaMid) < 1e-15)) {
75  // Solution found (or step size is infinitesimal and an overshoot).
76  // Empirically, this seems to terminate around 30-50 steps or so.
77  // This will find a zero point for exactly representable ranges and
78  // will terminate on a small step size for inexact, biasing towards
79  // overshooting.
80  delta = deltaMid;
81  break;
82  }
83  bool signMid = fMid > 0;
84  bool signInit = fInit > 0;
85  if (signMid == signInit) {
86  deltaInit = deltaMid;
87  } else {
88  deltaMax = deltaMid;
89  }
90  prevDeltaMid = deltaMid;
91  }
92  delta = deltaMid;
93 
94  // Recalculate adjMin/adjMax based on new delta.
95  auto adjMinMax = fns::computeMinMax(origMinAdj, numLevelsDouble, delta);
96  adjMin = adjMinMax.first;
97  adjMax = adjMinMax.second;
98 
99  satisfied = false;
100  zp = 0;
101 
102  if (!std::isnan(delta) && !std::isnan(adjMin) && !std::isnan(adjMax)) {
103  satisfied = true;
104  // Finally, scale and zeroPoint. Since it casts to integer, only valid
105  // if the inputs are valid.
106  zp = std::round(storageParams.minValue - adjMin / delta);
107  }
108 
109  return satisfied;
110 }

113  int64_t xq = std::round(x / delta + zp);
114  return std::max<int64_t>(0, std::min<int64_t>(storageParams.numLevels, xq));
