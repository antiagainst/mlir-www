9 // This file implements loop unroll and jam. Unroll and jam is a transformation
10 // that improves locality, in particular, register reuse, while also improving
11 // operation level parallelism. The example below shows what it does in nearly
12 // the general case. Loop unroll and jam currently works if the bounds of the
13 // loops inner to the loop being unroll-jammed do not depend on the latter.
14 //
15 // Before After unroll and jam of i by factor 2:
16 //
17 // for i, step = 2
18 // for i S1(i);
19 // S1; S2(i);
20 // S2; S1(i+1);
21 // for j S2(i+1);
22 // S3; for j
23 // S4; S3(i, j);
24 // S5; S4(i, j);
25 // S6; S3(i+1, j)
26 // S4(i+1, j)
27 // S5(i);
28 // S6(i);
29 // S5(i+1);
30 // S6(i+1);
31 //
32 // Note: 'if/else' blocks are not jammed. So, if there are loops inside if
33 // op's, bodies of those loops will not be jammed.
34 //===----------------------------------------------------------------------===//

82 void LoopUnrollAndJam::runOnFunction() {
83  // Currently, just the outermost loop from the first loop nest is
84  // unroll-and-jammed by this pass. However, runOnAffineForOp can be called on
85  // any for operation.
86  auto &entryBlock = getFunction().front();
87  if (auto forOp = dyn_cast<AffineForOp>(entryBlock.front()))
88  runOnAffineForOp(forOp);
89 }

91 /// Unroll and jam a 'affine.for' op. Default unroll jam factor is
92 /// kDefaultUnrollJamFactor. Return failure if nothing was done.
93 LogicalResult LoopUnrollAndJam::runOnAffineForOp(AffineForOp forOp) {
94  // Unroll and jam by the factor that was passed if any.
95  if (unrollJamFactor.hasValue())
96  return loopUnrollJamByFactor(forOp, unrollJamFactor.getValue());
97  // Otherwise, unroll jam by the command-line factor if one was specified.
98  if (clUnrollJamFactor.getNumOccurrences() > 0)
99  return loopUnrollJamByFactor(forOp, clUnrollJamFactor);
100 
101  // Unroll and jam by four otherwise.
102  return loopUnrollJamByFactor(forOp, kDefaultUnrollJamFactor);
103 }

115 /// Unrolls and jams this loop by the specified factor.
116 LogicalResult mlir::loopUnrollJamByFactor(AffineForOp forOp,
117  uint64_t unrollJamFactor) {
118  // Gathers all maximal sub-blocks of operations that do not themselves
119  // include a for op (a operation could have a descendant for op though
120  // in its tree). Ignore the block terminators.
121  struct JamBlockGatherer {
122  // Store iterators to the first and last op of each sub-block found.
123  std::vector<std::pair<Block::iterator, Block::iterator>> subBlocks;
124 
125  // This is a linear time walk.
126  void walk(Operation *op) {
127  for (auto &region : op->getRegions())
128  for (auto &block : region)
129  walk(block);
130  }
131  void walk(Block &block) {
132  for (auto it = block.begin(), e = std::prev(block.end()); it != e;) {
133  auto subBlockStart = it;
134  while (it != e && !isa<AffineForOp>(&*it))
135  ++it;
136  if (it != subBlockStart)
137  subBlocks.push_back({subBlockStart, std::prev(it)});
138  // Process all for insts that appear next.
139  while (it != e && isa<AffineForOp>(&*it))
140  walk(&*it++);
141  }
142  }
143  };
144 
145  assert(unrollJamFactor >= 1 && "unroll jam factor should be >= 1");
146 
147  if (unrollJamFactor == 1)
148  return promoteIfSingleIteration(forOp);
149 
150  if (forOp.getBody()->empty() ||
151  forOp.getBody()->begin() == std::prev(forOp.getBody()->end()))
152  return failure();
153 
154  // Loops where both lower and upper bounds are multi-result maps won't be
155  // unrolled (since the trip can't be expressed as an affine function in
156  // general).
157  // TODO(mlir-team): this may not be common, but we could support the case
158  // where the lower bound is a multi-result map and the ub is a single result
159  // one.
160  if (forOp.getLowerBoundMap().getNumResults() != 1)
161  return failure();
162 
163  Optional<uint64_t> mayBeConstantTripCount = getConstantTripCount(forOp);
164  // If the trip count is lower than the unroll jam factor, no unroll jam.
165  if (mayBeConstantTripCount.hasValue() &&
166  mayBeConstantTripCount.getValue() < unrollJamFactor)
167  return failure();
168 
169  auto *forInst = forOp.getOperation();
170 
171  // Gather all sub-blocks to jam upon the loop being unrolled.
172  JamBlockGatherer jbg;
173  jbg.walk(forInst);
174  auto &subBlocks = jbg.subBlocks;
175 
176  // Generate the cleanup loop if trip count isn't a multiple of
177  // unrollJamFactor.
178  if (getLargestDivisorOfTripCount(forOp) % unrollJamFactor != 0) {
179  // Insert the cleanup loop right after 'forOp'.
180  OpBuilder builder(forInst->getBlock(), std::next(Block::iterator(forInst)));
181  auto cleanupAffineForOp = cast<AffineForOp>(builder.clone(*forInst));
182  // Adjust the lower bound of the cleanup loop; its upper bound is the same
183  // as the original loop's upper bound.
184  AffineMap cleanupMap;
185  SmallVector<Value, 4> cleanupOperands;
186  getCleanupLoopLowerBound(forOp, unrollJamFactor, &cleanupMap,
187  &cleanupOperands, builder);
188  cleanupAffineForOp.setLowerBound(cleanupOperands, cleanupMap);
189 
190  // Promote the cleanup loop if it has turned into a single iteration loop.
191  promoteIfSingleIteration(cleanupAffineForOp);
192 
193  // Adjust the upper bound of the original loop - it will be the same as the
194  // cleanup loop's lower bound. Its lower bound remains unchanged.
195  forOp.setUpperBound(cleanupOperands, cleanupMap);
196  }
197 
198  // Scale the step of loop being unroll-jammed by the unroll-jam factor.
199  int64_t step = forOp.getStep();
200  forOp.setStep(step * unrollJamFactor);
201 
202  auto forOpIV = forOp.getInductionVar();
203  // Unroll and jam (appends unrollJamFactor - 1 additional copies).
204  for (unsigned i = unrollJamFactor - 1; i >= 1; --i) {
205  // Operand map persists across all sub-blocks.
206  BlockAndValueMapping operandMapping;
207  for (auto &subBlock : subBlocks) {
208  // Builder to insert unroll-jammed bodies. Insert right at the end of
209  // sub-block.
210  OpBuilder builder(subBlock.first->getBlock(), std::next(subBlock.second));
211 
212  // If the induction variable is used, create a remapping to the value for
213  // this unrolled instance.
214  if (!forOpIV.use_empty()) {
215  // iv' = iv + i, i = 1 to unrollJamFactor-1.
216  auto d0 = builder.getAffineDimExpr(0);
217  auto bumpMap = AffineMap::get(1, 0, {d0 + i * step});
218  auto ivUnroll =
219  builder.create<AffineApplyOp>(forInst->getLoc(), bumpMap, forOpIV);
220  operandMapping.map(forOpIV, ivUnroll);
221  }
222  // Clone the sub-block being unroll-jammed.
223  for (auto it = subBlock.first; it != std::next(subBlock.second); ++it) {
224  builder.clone(*it, operandMapping);
225  }
226  }
227  }
228 
229  // Promote the loop body up if this has turned into a single iteration loop.
230  promoteIfSingleIteration(forOp);
231  return success();
232 }
