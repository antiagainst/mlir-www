43 // <operation> ::= `rocdl.XYZ` : type
44 static ParseResult parseROCDLOp(OpAsmParser &parser, OperationState &result) {
45  Type type;
46  return failure(parser.parseOptionalAttrDict(result.attributes) ||
47  parser.parseColonType(type) ||
48  parser.addTypeToList(type, result.types));
49 }
