33 /// Analyzes the `transfer` to find an access dimension along the fastest remote
34 /// MemRef dimension. If such a dimension with coalescing properties is found,
35 /// `pivs` and `vectorView` are swapped so that the invocation of
36 /// LoopNestBuilder captures it in the innermost loop.
37 template <typename TransferOpTy>
38 static void coalesceCopy(TransferOpTy transfer,
39  SmallVectorImpl<edsc::ValueHandle *> *pivs,
40  edsc::VectorView *vectorView) {
41  // rank of the remote memory access, coalescing behavior occurs on the
42  // innermost memory dimension.
43  auto remoteRank = transfer.getMemRefType().getRank();
44  // Iterate over the results expressions of the permutation map to determine
45  // the loop order for creating pointwise copies between remote and local
46  // memories.
47  int coalescedIdx = -1;
48  auto exprs = transfer.permutation_map().getResults();
49  for (auto en : llvm::enumerate(exprs)) {
50  auto dim = en.value().template dyn_cast<AffineDimExpr>();
51  if (!dim) {
52  continue;
53  }
54  auto memRefDim = dim.getPosition();
55  if (memRefDim == remoteRank - 1) {
56  // memRefDim has coalescing properties, it should be swapped in the last
57  // position.
58  assert(coalescedIdx == -1 && "Unexpected > 1 coalesced indices");
59  coalescedIdx = en.index();
60  }
61  }
62  if (coalescedIdx >= 0) {
63  std::swap(pivs->back(), (*pivs)[coalescedIdx]);
64  vectorView->swapRanges(pivs->size() - 1, coalescedIdx);
65  }
66 }

68 /// Emits remote memory accesses that are clipped to the boundaries of the
69 /// MemRef.
70 template <typename TransferOpTy>
71 static SmallVector<edsc::ValueHandle, 8> clip(TransferOpTy transfer,
72  edsc::MemRefView &view,
73  ArrayRef<edsc::IndexHandle> ivs) {
74  using namespace mlir::edsc;
75  using namespace edsc::op;
76  using edsc::intrinsics::select;
77 
78  IndexHandle zero(index_type(0)), one(index_type(1));
79  SmallVector<edsc::ValueHandle, 8> memRefAccess(transfer.indices());
80  SmallVector<edsc::ValueHandle, 8> clippedScalarAccessExprs(
81  memRefAccess.size(), edsc::IndexHandle());
82 
83  // Indices accessing to remote memory are clipped and their expressions are
84  // returned in clippedScalarAccessExprs.
85  for (unsigned memRefDim = 0; memRefDim < clippedScalarAccessExprs.size();
86  ++memRefDim) {
87  // Linear search on a small number of entries.
88  int loopIndex = -1;
89  auto exprs = transfer.permutation_map().getResults();
90  for (auto en : llvm::enumerate(exprs)) {
91  auto expr = en.value();
92  auto dim = expr.template dyn_cast<AffineDimExpr>();
93  // Sanity check.
94  assert(
95  (dim || expr.template cast<AffineConstantExpr>().getValue() == 0) &&
96  "Expected dim or 0 in permutationMap");
97  if (dim && memRefDim == dim.getPosition()) {
98  loopIndex = en.index();
99  break;
100  }
101  }
102 
103  // We cannot distinguish atm between unrolled dimensions that implement
104  // the "always full" tile abstraction and need clipping from the other
105  // ones. So we conservatively clip everything.
106  auto N = view.ub(memRefDim);
107  auto i = memRefAccess[memRefDim];
108  if (loopIndex < 0) {
109  auto N_minus_1 = N - one;
110  auto select_1 = select(i < N, i, N_minus_1);
111  clippedScalarAccessExprs[memRefDim] = select(i < zero, zero, select_1);
112  } else {
113  auto ii = ivs[loopIndex];
114  auto i_plus_ii = i + ii;
115  auto N_minus_1 = N - one;
116  auto select_1 = select(i_plus_ii < N, i_plus_ii, N_minus_1);
117  clippedScalarAccessExprs[memRefDim] =
118  select(i_plus_ii < zero, zero, select_1);
119  }
120  }
121 
122  return clippedScalarAccessExprs;
123 }

129 /// Implements lowering of TransferReadOp and TransferWriteOp to a
130 /// proper abstraction for the hardware.
131 ///
132 /// For now, we only emit a simple loop nest that performs clipped pointwise
133 /// copies from a remote to a locally allocated memory.
134 ///
135 /// Consider the case:
136 ///
137 /// ```mlir
138 /// // Read the slice `%A[%i0, %i1:%i1+256, %i2:%i2+32]` into
139 /// // vector<32x256xf32> and pad with %f0 to handle the boundary case:
140 /// %f0 = constant 0.0f : f32
141 /// loop.for %i0 = 0 to %0 {
142 /// loop.for %i1 = 0 to %1 step %c256 {
143 /// loop.for %i2 = 0 to %2 step %c32 {
144 /// %v = vector.transfer_read %A[%i0, %i1, %i2], %f0
145 /// {permutation_map: (d0, d1, d2) -> (d2, d1)} :
146 /// memref<?x?x?xf32>, vector<32x256xf32>
147 /// }}}
148 /// ```
149 ///
150 /// The rewriters construct loop and indices that access MemRef A in a pattern
151 /// resembling the following (while guaranteeing an always full-tile
152 /// abstraction):
153 ///
154 /// ```mlir
155 /// loop.for %d2 = 0 to %c256 {
156 /// loop.for %d1 = 0 to %c32 {
157 /// %s = %A[%i0, %i1 + %d1, %i2 + %d2] : f32
158 /// %tmp[%d2, %d1] = %s
159 /// }
160 /// }
161 /// ```
162 ///
163 /// In the current state, only a clipping transfer is implemented by `clip`,
164 /// which creates individual indexing expressions of the form:
165 ///
166 /// ```mlir-dsc
167 /// auto condMax = i + ii < N;
168 /// auto max = select(condMax, i + ii, N - one)
169 /// auto cond = i + ii < zero;
170 /// select(cond, zero, max);
171 /// ```
172 ///
173 /// In the future, clipping should not be the only way and instead we should
174 /// load vectors + mask them. Similarly on the write side, load/mask/store for
175 /// implementing RMW behavior.
176 ///
177 /// Lowers TransferOp into a combination of:
178 /// 1. local memory allocation;
179 /// 2. perfect loop nest over:
180 /// a. scalar load/stores from local buffers (viewed as a scalar memref);
181 /// a. scalar store/load to original memref (with clipping).
182 /// 3. vector_load/store
183 /// 4. local memory deallocation.
184 /// Minor variations occur depending on whether a TransferReadOp or
185 /// a TransferWriteOp is rewritten.
186 template <typename TransferOpTy>
187 struct VectorTransferRewriter : public RewritePattern {
188  explicit VectorTransferRewriter(MLIRContext *context)
189  : RewritePattern(TransferOpTy::getOperationName(), 1, context) {}
190 
191  /// Used for staging the transfer in a local scalar buffer.
192  MemRefType tmpMemRefType(TransferOpTy transfer) const {
193  auto vectorType = transfer.getVectorType();
194  return MemRefType::get(vectorType.getShape(), vectorType.getElementType(),
195  {}, 0);
196  }
197 
198  /// Performs the rewrite.
199  PatternMatchResult matchAndRewrite(Operation *op,
200  PatternRewriter &rewriter) const override;
201 };

203 /// Lowers TransferReadOp into a combination of:
204 /// 1. local memory allocation;
205 /// 2. perfect loop nest over:
206 /// a. scalar load from local buffers (viewed as a scalar memref);
207 /// a. scalar store to original memref (with clipping).
208 /// 3. vector_load from local buffer (viewed as a memref<1 x vector>);
209 /// 4. local memory deallocation.
210 ///
211 /// Lowers the data transfer part of a TransferReadOp while ensuring no
212 /// out-of-bounds accesses are possible. Out-of-bounds behavior is handled by
213 /// clipping. This means that a given value in memory can be read multiple
214 /// times and concurrently.
215 ///
216 /// Important notes about clipping and "full-tiles only" abstraction:
217 /// =================================================================
218 /// When using clipping for dealing with boundary conditions, the same edge
219 /// value will appear multiple times (a.k.a edge padding). This is fine if the
220 /// subsequent vector operations are all data-parallel but **is generally
221 /// incorrect** in the presence of reductions or extract operations.
222 ///
223 /// More generally, clipping is a scalar abstraction that is expected to work
224 /// fine as a baseline for CPUs and GPUs but not for vector_load and DMAs.
225 /// To deal with real vector_load and DMAs, a "padded allocation + view"
226 /// abstraction with the ability to read out-of-memref-bounds (but still within
227 /// the allocated region) is necessary.
228 ///
229 /// Whether using scalar loops or vector_load/DMAs to perform the transfer,
230 /// junk values will be materialized in the vectors and generally need to be
231 /// filtered out and replaced by the "neutral element". This neutral element is
232 /// op-dependent so, in the future, we expect to create a vector filter and
233 /// apply it to a splatted constant vector with the proper neutral element at
234 /// each ssa-use. This filtering is not necessary for pure data-parallel
235 /// operations.
236 ///
237 /// In the case of vector_store/DMAs, Read-Modify-Write will be required, which
238 /// also have concurrency implications. Note that by using clipped scalar stores
239 /// in the presence of data-parallel only operations, we generate code that
240 /// writes the same value multiple time on the edge locations.
241 ///
242 /// TODO(ntv): implement alternatives to clipping.
243 /// TODO(ntv): support non-data-parallel operations.
244 
245 /// Performs the rewrite.
246 template <>
247 PatternMatchResult VectorTransferRewriter<TransferReadOp>::matchAndRewrite(
248  Operation *op, PatternRewriter &rewriter) const {
249  using namespace mlir::edsc;
250  using namespace mlir::edsc::op;
251  using namespace mlir::edsc::intrinsics;
252  using IndexedValue =
253  TemplatedIndexedValue<intrinsics::std_load, intrinsics::std_store>;
254 
255  TransferReadOp transfer = cast<TransferReadOp>(op);
256 
257  // 1. Setup all the captures.
258  ScopedContext scope(rewriter, transfer.getLoc());
259  IndexedValue remote(transfer.memref());
260  MemRefView view(transfer.memref());
261  VectorView vectorView(transfer.vector());
262  SmallVector<IndexHandle, 8> ivs = makeIndexHandles(vectorView.rank());
263  SmallVector<ValueHandle *, 8> pivs =
264  makeHandlePointers(MutableArrayRef<IndexHandle>(ivs));
265  coalesceCopy(transfer, &pivs, &vectorView);
266 
267  auto lbs = vectorView.getLbs();
268  auto ubs = vectorView.getUbs();
269  SmallVector<ValueHandle, 8> steps;
270  steps.reserve(vectorView.getSteps().size());
271  for (auto step : vectorView.getSteps())
272  steps.push_back(constant_index(step));
273 
274  // 2. Emit alloc-copy-load-dealloc.
275  ValueHandle tmp = alloc(tmpMemRefType(transfer));
276  IndexedValue local(tmp);
277  ValueHandle vec = vector_type_cast(tmp);
278  LoopNestBuilder(pivs, lbs, ubs, steps)([&] {
279  // Computes clippedScalarAccessExprs in the loop nest scope (ivs exist).
280  local(ivs) = remote(clip(transfer, view, ivs));
281  });
282  ValueHandle vectorValue = std_load(vec);
283  (dealloc(tmp)); // vexing parse
284 
285  // 3. Propagate.
286  rewriter.replaceOp(op, vectorValue.getValue());
287  return matchSuccess();
288 }

290 /// Lowers TransferWriteOp into a combination of:
291 /// 1. local memory allocation;
292 /// 2. vector_store to local buffer (viewed as a memref<1 x vector>);
293 /// 3. perfect loop nest over:
294 /// a. scalar load from local buffers (viewed as a scalar memref);
295 /// a. scalar store to original memref (with clipping).
296 /// 4. local memory deallocation.
297 ///
298 /// More specifically, lowers the data transfer part while ensuring no
299 /// out-of-bounds accesses are possible. Out-of-bounds behavior is handled by
300 /// clipping. This means that a given value in memory can be written to multiple
301 /// times and concurrently.
302 ///
303 /// See `Important notes about clipping and full-tiles only abstraction` in the
304 /// description of `readClipped` above.
305 ///
306 /// TODO(ntv): implement alternatives to clipping.
307 /// TODO(ntv): support non-data-parallel operations.
308 template <>
309 PatternMatchResult VectorTransferRewriter<TransferWriteOp>::matchAndRewrite(
310  Operation *op, PatternRewriter &rewriter) const {
311  using namespace mlir::edsc;
312  using namespace mlir::edsc::op;
313  using namespace mlir::edsc::intrinsics;
314  using IndexedValue =
315  TemplatedIndexedValue<intrinsics::std_load, intrinsics::std_store>;
316 
317  TransferWriteOp transfer = cast<TransferWriteOp>(op);
318 
319  // 1. Setup all the captures.
320  ScopedContext scope(rewriter, transfer.getLoc());
321  IndexedValue remote(transfer.memref());
322  MemRefView view(transfer.memref());
323  ValueHandle vectorValue(transfer.vector());
324  VectorView vectorView(transfer.vector());
325  SmallVector<IndexHandle, 8> ivs = makeIndexHandles(vectorView.rank());
326  SmallVector<ValueHandle *, 8> pivs =
327  makeHandlePointers(MutableArrayRef<IndexHandle>(ivs));
328  coalesceCopy(transfer, &pivs, &vectorView);
329 
330  auto lbs = vectorView.getLbs();
331  auto ubs = vectorView.getUbs();
332  SmallVector<ValueHandle, 8> steps;
333  steps.reserve(vectorView.getSteps().size());
334  for (auto step : vectorView.getSteps())
335  steps.push_back(constant_index(step));
336 
337  // 2. Emit alloc-store-copy-dealloc.
338  ValueHandle tmp = alloc(tmpMemRefType(transfer));
339  IndexedValue local(tmp);
340  ValueHandle vec = vector_type_cast(tmp);
341  std_store(vectorValue, vec);
342  LoopNestBuilder(pivs, lbs, ubs, steps)([&] {
343  // Computes clippedScalarAccessExprs in the loop nest scope (ivs exist).
344  remote(clip(transfer, view, ivs)) = local(ivs);
345  });
346  (dealloc(tmp)); // vexing parse...
347 
348  rewriter.eraseOp(op);
349  return matchSuccess();
350 }
