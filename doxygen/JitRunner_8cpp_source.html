137 // JIT-compile the given module and run "entryPoint" with "args" as arguments.
138 static Error
139 compileAndExecute(ModuleOp module, StringRef entryPoint,
140  std::function<llvm::Error(llvm::Module *)> transformer,
141  void **args) {
142  Optional<llvm::CodeGenOpt::Level> jitCodeGenOptLevel;
143  if (auto clOptLevel = getCommandLineOptLevel())
144  jitCodeGenOptLevel =
145  static_cast<llvm::CodeGenOpt::Level>(clOptLevel.getValue());
146  SmallVector<StringRef, 4> libs(clSharedLibs.begin(), clSharedLibs.end());
147  auto expectedEngine = mlir::ExecutionEngine::create(module, transformer,
148  jitCodeGenOptLevel, libs);
149  if (!expectedEngine)
150  return expectedEngine.takeError();
151 
152  auto engine = std::move(*expectedEngine);
153  auto expectedFPtr = engine->lookup(entryPoint);
154  if (!expectedFPtr)
155  return expectedFPtr.takeError();
156 
157  if (dumpObjectFile)
158  engine->dumpToObjectFile(objectFilename.empty() ? inputFilename + ".o"
159  : objectFilename);
160 
161  void (*fptr)(void **) = *expectedFPtr;
162  (*fptr)(args);
163 
164  return Error::success();
165 }

167 static Error compileAndExecuteVoidFunction(
168  ModuleOp module, StringRef entryPoint,
169  std::function<llvm::Error(llvm::Module *)> transformer) {
170  auto mainFunction = module.lookupSymbol<LLVM::LLVMFuncOp>(entryPoint);
171  if (!mainFunction || mainFunction.getBlocks().empty())
172  return make_string_error("entry point not found");
173  void *empty = nullptr;
174  return compileAndExecute(module, entryPoint, transformer, &empty);
175 }

177 static Error compileAndExecuteSingleFloatReturnFunction(
178  ModuleOp module, StringRef entryPoint,
179  std::function<llvm::Error(llvm::Module *)> transformer) {
180  auto mainFunction = module.lookupSymbol<LLVM::LLVMFuncOp>(entryPoint);
181  if (!mainFunction || mainFunction.isExternal())
182  return make_string_error("entry point not found");
183 
184  if (mainFunction.getType().getFunctionNumParams() != 0)
185  return make_string_error("function inputs not supported");
186 
187  if (!mainFunction.getType().getFunctionResultType().isFloatTy())
188  return make_string_error("only single llvm.f32 function result supported");
189 
190  float res;
191  struct {
192  void *data;
193  } data;
194  data.data = &res;
195  if (auto error =
196  compileAndExecute(module, entryPoint, transformer, (void **)&data))
197  return error;
198 
199  // Intentional printing of the output so we can test.
200  llvm::outs() << res << '\n';
201 
202  return Error::success();
203 }

205 // Entry point for all CPU runners. Expects the common argc/argv arguments for
206 // standard C++ main functions and an mlirTransformer.
207 // The latter is applied after parsing the input into MLIR IR and before passing
208 // the MLIR module to the ExecutionEngine.
209 int mlir::JitRunnerMain(
210  int argc, char **argv,
211  function_ref<LogicalResult(mlir::ModuleOp)> mlirTransformer) {
212  llvm::InitLLVM y(argc, argv);
213 
214  initializeLLVM();
215  mlir::initializeLLVMPasses();
216 
217  llvm::cl::ParseCommandLineOptions(argc, argv, "MLIR CPU execution driver\n");
218 
219  Optional<unsigned> optLevel = getCommandLineOptLevel();
220  SmallVector<std::reference_wrapper<llvm::cl::opt<bool>>, 4> optFlags{
221  optO0, optO1, optO2, optO3};
222  unsigned optCLIPosition = 0;
223  // Determine if there is an optimization flag present, and its CLI position
224  // (optCLIPosition).
225  for (unsigned j = 0; j < 4; ++j) {
226  auto &flag = optFlags[j].get();
227  if (flag) {
228  optCLIPosition = flag.getPosition();
229  break;
230  }
231  }
232  // Generate vector of pass information, plus the index at which we should
233  // insert any optimization passes in that vector (optPosition).
234  SmallVector<const llvm::PassInfo *, 4> passes;
235  unsigned optPosition = 0;
236  for (unsigned i = 0, e = llvmPasses.size(); i < e; ++i) {
237  passes.push_back(llvmPasses[i]);
238  if (optCLIPosition < llvmPasses.getPosition(i)) {
239  optPosition = i;
240  optCLIPosition = UINT_MAX; // To ensure we never insert again
241  }
242  }
243 
244  MLIRContext context;
245  auto m = parseMLIRInput(inputFilename, &context);
246  if (!m) {
247  llvm::errs() << "could not parse the input IR\n";
248  return 1;
249  }
250 
251  if (mlirTransformer)
252  if (failed(mlirTransformer(m.get())))
253  return EXIT_FAILURE;
254 
255  auto tmBuilderOrError = llvm::orc::JITTargetMachineBuilder::detectHost();
256  if (!tmBuilderOrError) {
257  llvm::errs() << "Failed to create a JITTargetMachineBuilder for the host\n";
258  return EXIT_FAILURE;
259  }
260  auto tmOrError = tmBuilderOrError->createTargetMachine();
261  if (!tmOrError) {
262  llvm::errs() << "Failed to create a TargetMachine for the host\n";
263  return EXIT_FAILURE;
264  }
265 
266  auto transformer = mlir::makeLLVMPassesTransformer(
267  passes, optLevel, /*targetMachine=*/tmOrError->get(), optPosition);
268 
269  // Get the function used to compile and execute the module.
270  using CompileAndExecuteFnT = Error (*)(
271  ModuleOp, StringRef, std::function<llvm::Error(llvm::Module *)>);
272  auto compileAndExecuteFn =
273  llvm::StringSwitch<CompileAndExecuteFnT>(mainFuncType.getValue())
274  .Case("f32", compileAndExecuteSingleFloatReturnFunction)
275  .Case("void", compileAndExecuteVoidFunction)
276  .Default(nullptr);
277 
278  Error error =
279  compileAndExecuteFn
280  ? compileAndExecuteFn(m.get(), mainFuncName.getValue(), transformer)
281  : make_string_error("unsupported function type");
282 
283  int exitCode = EXIT_SUCCESS;
284  llvm::handleAllErrors(std::move(error),
285  [&exitCode](const llvm::ErrorInfoBase &info) {
286  llvm::errs() << "Error: ";
287  info.log(llvm::errs());
288  llvm::errs() << '\n';
289  exitCode = EXIT_FAILURE;
290  });
291 
292  return exitCode;
293 }
