36 static QuantizedType solveUniformType(SolverContext &solverContext,
37  const ClusteredFacts &clusteredFacts,
38  const CandidateQuantizedType &ct,
39  Type originalElementType, Location loc) {
40  switch (ct.scheme) {
41  default:
42  emitError(loc, "unsupported scheme for uniform type conversion");
43  return nullptr;
44 
45  case CandidateQuantizedType::Scheme::UniformPerLayer: {
46  if (!clusteredFacts.requiredRange.hasValue()) {
47  // TODO: Issue some kind of diagnostic. This is not an error.
48  return nullptr;
49  }
50 
51  uint64_t numLevels = ct.quantizedType.getStorageTypeMax() -
52  ct.quantizedType.getStorageTypeMin();
53  UniformStorageParams params{numLevels,
54  ct.quantizedType.getStorageTypeMin()};
55  UniformParamsFromMinMaxSolver solver(
56  params, clusteredFacts.requiredRange.getValue().first,
57  clusteredFacts.requiredRange.getValue().second);
58  if (!solver.compute()) {
59  emitWarning(loc) << "unable to solve uniform type with "
60  << "UniformParamsFromMinMaxSolver";
61  return nullptr;
62  }
63 
64  return UniformQuantizedType::getChecked(
65  ct.quantizedType.getFlags(), ct.quantizedType.getStorageType(),
66  originalElementType, solver.getScale(), solver.getZp(),
67  ct.quantizedType.getStorageTypeMin(),
68  ct.quantizedType.getStorageTypeMax(), loc);
69  }
70  case CandidateQuantizedType::Scheme::UniformExplicitFixedPointScale: {
71  if (!clusteredFacts.explicitScaleZeroPoint.hasValue()) {
72  emitRemark(loc)
73  << "unable to solve uniform type with UniformExplicitFixedPointScale "
74  << "(no explicitScaleZeroPoint)";
75  return nullptr;
76  }
77 
78  const auto &scaleZp = clusteredFacts.explicitScaleZeroPoint.getValue();
79  assert(scaleZp.value && "optional value not set on fact");
80 
81  if (scaleZp.conflict) {
82  emitWarning(loc)
83  << "conflicting explicit scale/zeroPoint on node cluster: "
84  << "an arbitrary scale/zeroPoint will be used";
85  }
86 
87  return UniformQuantizedType::getChecked(
88  ct.quantizedType.getFlags(), ct.quantizedType.getStorageType(),
89  originalElementType,
90  scaleZp.value->first, // scale
91  0, // zeroPoint (fixed point solutions only for this scheme)
92  ct.quantizedType.getStorageTypeMin(),
93  ct.quantizedType.getStorageTypeMax(), loc);
94 
95  return nullptr;
96  }
97  }
98 }

102 class PropagateExplicitScale : public CAGConstraintNode {
103 public:
104  PropagateExplicitScale()
105  : CAGConstraintNode(Kind::UniformPropagateExplicitScale) {}
106  static bool classof(const CAGNode *n) {
107  return n->getKind() == Kind::Constraint ||
108  n->getKind() == Kind::UniformPropagateExplicitScale;
109  }
110 
111 private:
112  void printLabel(raw_ostream &os) const override {
113  os << "PropagateExplicitScale";
114  }
115  void propagate(SolverContext &solverContext,
116  const TargetConfiguration &config) override {
117  DiscreteScaleZeroPointFact scaleZp;
118 
119  // Get scale/zp from all parents.
120  for (auto it = incoming_begin(), e = incoming_end(); it != e; ++it) {
121  auto parentAnchor = cast<CAGAnchorNode>(*it);
122  auto selectedType = parentAnchor->getUniformMetadata().selectedType;
123  if (auto uqType = selectedType.dyn_cast_or_null<UniformQuantizedType>()) {
124  scaleZp.assertValue(
125  CAGUniformMetadata::SalienceRequired,
126  std::make_pair(uqType.getScale(), static_cast<int64_t>(0)));
127  }
128  }
129 
130  // Propagate to children.
131  if (scaleZp.hasValue()) {
132  for (auto it = begin(), e = end(); it != e; ++it) {
133  auto childAnchor = cast<CAGAnchorNode>(*it);
134  if (modified(childAnchor->getUniformMetadata()
135  .explicitScaleZeroPoint.mergeFrom(scaleZp))) {
136  childAnchor->markDirty();
137  }
138  }
139  }
140  }
141 };

143 /// A constraint node which will solve uniform quantization for all parents
144 /// of the constraint, assuming that they are coupled.
145 class SolveUniformConstraintNode : public CAGConstraintNode {
146 public:
147  SolveUniformConstraintNode()
148  : CAGConstraintNode(Kind::SolveUniformConstraint) {
149  markDirty();
150  }
151  static bool classof(const CAGNode *n) {
152  return n->getKind() == Kind::Constraint ||
153  n->getKind() == Kind::SolveUniformConstraint;
154  }
155 
156 private:
157  void printLabel(raw_ostream &os) const override { os << "SolveUniform"; }
158 
159  void propagate(SolverContext &solverContext,
160  const TargetConfiguration &config) override {
161  // First determine the required min/max range and type constraints.
162  Location fusedLoc = UnknownLoc::get(&solverContext.getMlirContext());
163  llvm::SmallBitVector enabledCandidateTypesMask(
164  config.getAllCandidateTypesMask());
165  ClusteredFacts clusteredFacts;
166  Type originalElementType;
167  for (auto it = incoming_begin(), e = incoming_end(); it != e; ++it) {
168  auto parentAnchor = cast<CAGAnchorNode>(*it);
169  auto metadata = parentAnchor->getUniformMetadata();
170  // TODO: Possibly use a location that fuses all involved parents.
171  fusedLoc = parentAnchor->getOp()->getLoc();
172 
173  // Shared element type.
174  auto parentOriginalElementType =
175  getElementOrPrimitiveType(parentAnchor->getOriginalType());
176  if (!originalElementType) {
177  originalElementType = parentOriginalElementType;
178  } else {
179  if (originalElementType != parentOriginalElementType) {
180  parentAnchor->getOp()->emitError()
181  << "cannot compute uniform type: parent element types mismatch";
182  return;
183  }
184  }
185  // Range.
186  clusteredFacts.requiredRange.mergeFrom(metadata.requiredRange);
187 
188  // Explicit scale and zero point.
189  clusteredFacts.explicitScaleZeroPoint.mergeFrom(
190  metadata.explicitScaleZeroPoint);
191 
192  // Shared candidate types.
193  enabledCandidateTypesMask.reset(metadata.disabledCandidateTypes);
194  }
195 
196  // Find the first enabled candidate type.
197  const CandidateQuantizedType *bestCandidateType = nullptr;
198  for (auto &ct : config.getCandidateTypes()) {
199  if (enabledCandidateTypesMask.test(ct.ordinal)) {
200  bestCandidateType = &ct;
201  break;
202  }
203  }
204 
205  if (!bestCandidateType || !originalElementType) {
206  emitRemark(fusedLoc)
207  << "not solving uniform type (no viable candidate type)";
208  return;
209  }
210 
211  // Solve for the type.
212  QuantizedType selectedType =
213  solveUniformType(solverContext, clusteredFacts, *bestCandidateType,
214  originalElementType, fusedLoc);
215 
216  // Apply it to all parents.
217  for (auto it = incoming_begin(), e = incoming_end(); it != e; ++it) {
218  auto parentAnchor = cast<CAGAnchorNode>(*it);
219  auto &metadata = parentAnchor->getUniformMetadata();
220  if (metadata.selectedType != selectedType) {
221  metadata.selectedType = selectedType;
222  // And mark all children of the parent dirty (except us).
223  for (auto child : *parentAnchor) {
224  if (child != this) {
225  child->markDirty();
226  }
227  }
228  }
229  }
230  }
231 };
