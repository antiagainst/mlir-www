42 PatternMatchResult
43 FuncOpConversion::matchAndRewrite(FuncOp funcOp, ArrayRef<Value> operands,
44  ConversionPatternRewriter &rewriter) const {
45  auto fnType = funcOp.getType();
46  if (fnType.getNumResults()) {
47  return matchFailure();
48  }
49 
50  TypeConverter::SignatureConversion signatureConverter(fnType.getNumInputs());
51  {
52  for (auto argType : enumerate(funcOp.getType().getInputs())) {
53  auto convertedType = typeConverter.convertType(argType.value());
54  if (!convertedType) {
55  return matchFailure();
56  }
57  signatureConverter.addInputs(argType.index(), convertedType);
58  }
59  }
60 
61  rewriter.updateRootInPlace(funcOp, [&] {
62  funcOp.setType(rewriter.getFunctionType(
63  signatureConverter.getConvertedTypes(), llvm::None));
64  rewriter.applySignatureConversion(&funcOp.getBody(), signatureConverter);
65  });
66  return matchSuccess();
67 }

69 void ConvertStandardToSPIRVPass::runOnModule() {
70  MLIRContext *context = &getContext();
71  ModuleOp module = getModule();
72 
73  SPIRVTypeConverter typeConverter;
74  OwningRewritePatternList patterns;
75  populateStandardToSPIRVPatterns(context, typeConverter, patterns);
76  patterns.insert<FuncOpConversion>(context, typeConverter);
77 
78  std::unique_ptr<ConversionTarget> target = spirv::SPIRVConversionTarget::get(
79  spirv::lookupTargetEnvOrDefault(module), context);
80  target->addDynamicallyLegalOp<FuncOp>(
81  [&](FuncOp op) { return typeConverter.isSignatureLegal(op.getType()); });
82 
83  if (failed(applyPartialConversion(module, *target, patterns))) {
84  return signalPassFailure();
85  }
86 }
