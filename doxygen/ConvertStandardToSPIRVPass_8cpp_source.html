30 void ConvertStandardToSPIRVPass::runOnModule() {
31  MLIRContext *context = &getContext();
32  ModuleOp module = getModule();
33 
34  SPIRVTypeConverter typeConverter;
35  OwningRewritePatternList patterns;
36  populateStandardToSPIRVPatterns(context, typeConverter, patterns);
37  populateBuiltinFuncToSPIRVPatterns(context, typeConverter, patterns);
38 
39  std::unique_ptr<ConversionTarget> target = spirv::SPIRVConversionTarget::get(
40  spirv::lookupTargetEnvOrDefault(module), context);
41  target->addDynamicallyLegalOp<FuncOp>(
42  [&](FuncOp op) { return typeConverter.isSignatureLegal(op.getType()); });
43 
44  if (failed(applyPartialConversion(module, *target, patterns))) {
45  return signalPassFailure();
46  }
47 }
