38 TranslationParser::TranslationParser(llvm::cl::Option &opt)
39  : llvm::cl::parser<const TranslateFunction *>(opt) {
40  const auto &toMLIRRegistry = getTranslationToMLIRRegistry();
41  const auto &fromMLIRRegistry = getTranslationFromMLIRRegistry();
42  const auto &fileToFileRegistry = getTranslationRegistry();
43 
44  // Reserve the required capacity upfront so that pointers are not
45  // invalidated on reallocation.
46  wrapperStorage.reserve(toMLIRRegistry.size() + fromMLIRRegistry.size() +
47  fileToFileRegistry.size());
48  for (const auto &kv : toMLIRRegistry) {
49  TranslateSourceMgrToMLIRFunction function = kv.second;
50  TranslateFunction wrapper = [function](llvm::SourceMgr &sourceMgr,
51  raw_ostream &output,
52  MLIRContext *context) {
53  OwningModuleRef module = function(sourceMgr, context);
54  if (!module)
55  return failure();
56  return printMLIROutput(*module, output);
57  };
58  wrapperStorage.emplace_back(std::move(wrapper));
59 
60  addLiteralOption(kv.first(), &wrapperStorage.back(), kv.first());
61  }
62 
63  for (const auto &kv : fromMLIRRegistry) {
64  TranslateFromMLIRFunction function = kv.second;
65  TranslateFunction wrapper = [function](llvm::SourceMgr &sourceMgr,
66  raw_ostream &output,
67  MLIRContext *context) {
68  auto module = OwningModuleRef(parseSourceFile(sourceMgr, context));
69  if (!module)
70  return failure();
71  return function(module.get(), output);
72  };
73  wrapperStorage.emplace_back(std::move(wrapper));
74 
75  addLiteralOption(kv.first(), &wrapperStorage.back(), kv.first());
76  }
77  for (const auto &kv : fileToFileRegistry) {
78  wrapperStorage.emplace_back(kv.second);
79  addLiteralOption(kv.first(), &wrapperStorage.back(), kv.first());
80  }
81 }
