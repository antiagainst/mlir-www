94 // Check the structure of the loop nest:
95 // - there are enough loops to map to numDims;
96 // - the loops are perfectly nested;
97 // - the loop bounds can be computed above the outermost loop.
98 // This roughly corresponds to the "matcher" part of the pattern-based
99 // rewriting infrastructure.
100 template <typename OpTy>
101 static LogicalResult checkLoopNestMappableImpl(OpTy forOp, unsigned numDims) {
102  Region &limit = forOp.region();
103  for (unsigned i = 0, e = numDims; i < e; ++i) {
104  Operation *nested = &forOp.getBody()->front();
105  if (!areValuesDefinedAbove(getLowerBoundOperands(forOp), limit) ||
106  !areValuesDefinedAbove(getUpperBoundOperands(forOp), limit))
107  return forOp.emitError(
108  "loops with bounds depending on other mapped loops "
109  "are not supported");
110 
111  // The innermost loop can have an arbitrary body, skip the perfect nesting
112  // check for it.
113  if (i == e - 1)
114  break;
115 
116  auto begin = forOp.getBody()->begin(), end = forOp.getBody()->end();
117  if (forOp.getBody()->empty() || std::next(begin, 2) != end)
118  return forOp.emitError("expected perfectly nested loops in the body");
119 
120  if (!(forOp = dyn_cast<OpTy>(nested)))
121  return nested->emitError("expected a nested loop");
122  }
123  return success();
124 }

126 template <typename OpTy>
127 static LogicalResult checkLoopNestMappable(OpTy forOp, unsigned numBlockDims,
128  unsigned numThreadDims) {
129  if (numBlockDims < 1 || numThreadDims < 1) {
130  LLVM_DEBUG(llvm::dbgs() << "nothing to map");
131  return success();
132  }
133 
134  OpBuilder builder(forOp.getOperation());
135  if (numBlockDims > 3) {
136  return forOp.emitError("cannot map to more than 3 block dimensions");
137  }
138  if (numThreadDims > 3) {
139  return forOp.emitError("cannot map to more than 3 thread dimensions");
140  }
141  return checkLoopNestMappableImpl(forOp, numBlockDims + numThreadDims);
142 }

144 template <typename OpTy>
145 static LogicalResult checkLoopOpMappable(OpTy forOp, unsigned numBlockDims,
146  unsigned numThreadDims) {
147  if (numBlockDims < 1 || numThreadDims < 1) {
148  LLVM_DEBUG(llvm::dbgs() << "nothing to map");
149  return success();
150  }
151 
152  if (numBlockDims > 3) {
153  return forOp.emitError("cannot map to more than 3 block dimensions");
154  }
155  if (numThreadDims > 3) {
156  return forOp.emitError("cannot map to more than 3 thread dimensions");
157  }
158  if (numBlockDims != numThreadDims) {
159  // TODO(ravishankarm) : This can probably be relaxed by having a one-trip
160  // loop for the missing dimension, but there is not reason to handle this
161  // case for now.
162  return forOp.emitError(
163  "mismatch in block dimensions and thread dimensions");
164  }
165 
166  // Check that the forOp contains perfectly nested loops for numBlockDims
167  if (failed(checkLoopNestMappableImpl(forOp, numBlockDims))) {
168  return failure();
169  }
170 
171  // Get to the innermost loop.
172  for (auto i : seq<unsigned>(0, numBlockDims - 1)) {
173  forOp = cast<OpTy>(&forOp.getBody()->front());
174  (void)i;
175  }
176 
177  // The forOp now points to the body of the innermost loop mapped to blocks.
178  for (Operation &op : *forOp.getBody()) {
179  // If the operation is a loop, check that it is mappable to workItems.
180  if (auto innerLoop = dyn_cast<OpTy>(&op)) {
181  if (failed(checkLoopNestMappableImpl(innerLoop, numThreadDims))) {
182  return failure();
183  }
184  continue;
185  }
186  // TODO(ravishankarm) : If it is not a loop op, it is assumed that the
187  // statement is executed by all threads. It might be a collective operation,
188  // or some non-side effect instruction. Have to decide on "allowable"
189  // statements and check for those here.
190  }
191  return success();
192 }

223 // Collect ranges, bounds, steps and induction variables in preparation for
224 // mapping a loop nest of depth "numLoops" rooted at "forOp" to a GPU kernel.
225 // This may fail if the IR for computing loop bounds cannot be constructed, for
226 // example if an affine loop uses semi-affine maps. Return the last loop to be
227 // mapped on success, llvm::None on failure.
228 template <typename OpTy>
229 Optional<OpTy> LoopToGpuConverter::collectBounds(OpTy forOp,
230  unsigned numLoops) {
231  OpBuilder builder(forOp.getOperation());
232  dims.reserve(numLoops);
233  lbs.reserve(numLoops);
234  ivs.reserve(numLoops);
235  steps.reserve(numLoops);
236  OpTy currentLoop = forOp;
237  for (unsigned i = 0; i < numLoops; ++i) {
238  Value lowerBound = getOrEmitLowerBound(currentLoop, builder);
239  Value upperBound = getOrEmitUpperBound(currentLoop, builder);
240  if (!lowerBound || !upperBound) {
241  return llvm::None;
242  }
243 
244  Value range =
245  builder.create<SubIOp>(currentLoop.getLoc(), upperBound, lowerBound);
246  Value step = getOrCreateStep(currentLoop, builder);
247  if (!isConstantOne(step))
248  range = builder.create<SignedDivIOp>(currentLoop.getLoc(), range, step);
249  dims.push_back(range);
250 
251  lbs.push_back(lowerBound);
252  ivs.push_back(currentLoop.getInductionVar());
253  steps.push_back(step);
254 
255  if (i != numLoops - 1)
256  currentLoop = cast<OpTy>(&currentLoop.getBody()->front());
257  }
258  return currentLoop;
259 }

261 /// Given `nDims` perfectly nested loops rooted as `rootForOp`, convert them o
262 /// be partitioned across workgroups or workitems. The values for the
263 /// workgroup/workitem id along each dimension is passed in with `ids`. The
264 /// number of workgroups/workitems along each dimension are passed in with
265 /// `nids`. The innermost loop is mapped to the x-dimension, followed by the
266 /// next innermost loop to y-dimension, followed by z-dimension.
267 template <typename OpTy>
268 static OpTy createGPULaunchLoops(OpTy rootForOp, ArrayRef<Value> ids,
269  ArrayRef<Value> nids) {
270  auto nDims = ids.size();
271  assert(nDims == nids.size());
272  for (auto dim : llvm::seq<unsigned>(0, nDims)) {
273  // TODO(ravishankarm): Don't always need to generate a loop here. If nids >=
274  // number of iterations of the original loop, this becomes a if
275  // condition. Though that does rely on how the workgroup/workitem sizes are
276  // specified to begin with.
277  mapLoopToProcessorIds(rootForOp, ids[dim], nids[dim]);
278  if (dim != nDims - 1) {
279  rootForOp = cast<OpTy>(rootForOp.getBody()->front());
280  }
281  }
282  return rootForOp;
283 }

288 static void packIdAndNumId(gpu::KernelDim3 kernelIds,
289  gpu::KernelDim3 kernelNids, unsigned nDims,
290  SmallVectorImpl<Value> &ids,
291  SmallVectorImpl<Value> &nids) {
292  assert(nDims <= 3 && "invalid number of launch dimensions");

294  SmallVector<Value, 3> allNids = {kernelNids.z, kernelNids.y, kernelNids.x};
295  ids.clear();
296  ids.append(std::next(allIds.begin(), allIds.size() - nDims), allIds.end());

302 /// Generate the body of the launch operation.
303 template <typename OpTy>
304 static LogicalResult
305 createLaunchBody(OpBuilder &builder, OpTy rootForOp, gpu::LaunchOp launchOp,
306  unsigned numBlockDims, unsigned numThreadDims) {
307  OpBuilder::InsertionGuard bodyInsertionGuard(builder);
308  builder.setInsertionPointToEnd(&launchOp.body().front());
309  auto returnOp = builder.create<gpu::ReturnOp>(launchOp.getLoc());
310 
311  rootForOp.getOperation()->moveBefore(returnOp);
312  SmallVector<Value, 3> workgroupID, numWorkGroups;
313  packIdAndNumId(launchOp.getBlockIds(), launchOp.getGridSize(), numBlockDims,
314  workgroupID, numWorkGroups);
315 
316  // Partition the loop for mapping to workgroups.
317  auto loopOp = createGPULaunchLoops(rootForOp, workgroupID, numWorkGroups);
318 
319  // Iterate over the body of the loopOp and get the loops to partition for
320  // thread blocks.
321  SmallVector<OpTy, 1> threadRootForOps;
322  for (Operation &op : *loopOp.getBody()) {
323  if (auto threadRootForOp = dyn_cast<OpTy>(&op)) {
324  threadRootForOps.push_back(threadRootForOp);
325  }
326  }
327 
328  SmallVector<Value, 3> workItemID, workGroupSize;
329  packIdAndNumId(launchOp.getThreadIds(), launchOp.getBlockSize(),
330  numThreadDims, workItemID, workGroupSize);
331  for (auto &loopOp : threadRootForOps) {
332  builder.setInsertionPoint(loopOp);
333  createGPULaunchLoops(loopOp, workItemID, workGroupSize);
334  }
335  return success();
336 }

338 // Convert the computation rooted at the `rootForOp`, into a GPU kernel with the
339 // given workgroup size and number of workgroups.
340 template <typename OpTy>
341 static LogicalResult createLaunchFromOp(OpTy rootForOp,
342  ArrayRef<Value> numWorkGroups,
343  ArrayRef<Value> workGroupSizes) {
344  OpBuilder builder(rootForOp.getOperation());
345  if (numWorkGroups.size() > 3) {
346  return rootForOp.emitError("invalid ")
347  << numWorkGroups.size() << "-D workgroup specification";
348  }
349  auto loc = rootForOp.getLoc();
350  Value one = builder.create<ConstantOp>(
351  loc, builder.getIntegerAttr(builder.getIndexType(), 1));
352  SmallVector<Value, 3> numWorkGroups3D(3, one), workGroupSize3D(3, one);
353  for (auto numWorkGroup : enumerate(numWorkGroups)) {
354  numWorkGroups3D[numWorkGroup.index()] = numWorkGroup.value();
355  }
356  for (auto workGroupSize : enumerate(workGroupSizes)) {
357  workGroupSize3D[workGroupSize.index()] = workGroupSize.value();
358  }
359 
360  // Get the values used within the region of the rootForOp but defined above
361  // it.
362  llvm::SetVector<Value> valuesToForwardSet;
363  getUsedValuesDefinedAbove(rootForOp.region(), rootForOp.region(),
364  valuesToForwardSet);
365  // Also add the values used for the lb, ub, and step of the rootForOp.
366  valuesToForwardSet.insert(rootForOp.getOperands().begin(),
367  rootForOp.getOperands().end());
368  auto valuesToForward = valuesToForwardSet.takeVector();
369  auto launchOp = builder.create<gpu::LaunchOp>(
370  rootForOp.getLoc(), numWorkGroups3D[0], numWorkGroups3D[1],
371  numWorkGroups3D[2], workGroupSize3D[0], workGroupSize3D[1],
372  workGroupSize3D[2], valuesToForward);
373  if (failed(createLaunchBody(builder, rootForOp, launchOp,
374  numWorkGroups.size(), workGroupSizes.size()))) {
375  return failure();
376  }
377 
378  // Replace values that are used within the region of the launchOp but are
379  // defined outside. They all are replaced with kernel arguments.
380  for (auto pair :
381  llvm::zip_first(valuesToForward, launchOp.getKernelArguments())) {
382  Value from = std::get<0>(pair);
383  Value to = std::get<1>(pair);
384  replaceAllUsesInRegionWith(from, to, launchOp.body());
385  }
386  return success();
387 }

389 // Replace the rooted at "rootForOp" with a GPU launch operation. This expects
390 // "innermostForOp" to point to the last loop to be transformed to the kernel,
391 // and to have (numBlockDims + numThreadDims) perfectly nested loops between
392 // "rootForOp" and "innermostForOp".
393 // TODO(ravishankarm) : This method can be modified to use the
394 // createLaunchFromOp method, since that is a strict generalization of this
395 // method.
396 template <typename OpTy>
397 void LoopToGpuConverter::createLaunch(OpTy rootForOp, OpTy innermostForOp,
398  unsigned numBlockDims,
399  unsigned numThreadDims) {
400  OpBuilder builder(rootForOp.getOperation());
401  // Prepare the grid and block sizes for the launch operation. If there is
402  // no loop mapped to a specific dimension, use constant "1" as its size.
403  Value constOne = (numBlockDims < 3 || numThreadDims < 3)
404  ? builder.create<ConstantIndexOp>(rootForOp.getLoc(), 1)
405  : nullptr;
406  Value gridSizeX = dims[0];
407  Value gridSizeY = numBlockDims > 1 ? dims[1] : constOne;
408  Value gridSizeZ = numBlockDims > 2 ? dims[2] : constOne;
409  Value blockSizeX = dims[numBlockDims];
410  Value blockSizeY = numThreadDims > 1 ? dims[numBlockDims + 1] : constOne;
411  Value blockSizeZ = numThreadDims > 2 ? dims[numBlockDims + 2] : constOne;
412 
413  // Create a launch op and move the body region of the innermost loop to the
414  // launch op. Pass the values defined outside the outermost loop and used
415  // inside the innermost loop and loop lower bounds as kernel data arguments.
416  // Still assuming perfect nesting so there are no values other than induction
417  // variables that are defined in one loop and used in deeper loops.
418  llvm::SetVector<Value> valuesToForwardSet;
419  getUsedValuesDefinedAbove(innermostForOp.region(), rootForOp.region(),
420  valuesToForwardSet);
421  auto valuesToForward = valuesToForwardSet.takeVector();
422  auto originallyForwardedValues = valuesToForward.size();
423  valuesToForward.insert(valuesToForward.end(), lbs.begin(), lbs.end());
424  valuesToForward.insert(valuesToForward.end(), steps.begin(), steps.end());
425  auto launchOp = builder.create<gpu::LaunchOp>(
426  rootForOp.getLoc(), gridSizeX, gridSizeY, gridSizeZ, blockSizeX,
427  blockSizeY, blockSizeZ, valuesToForward);
428  valuesToForward.resize(originallyForwardedValues);
429 
430  // Replace the loop terminator (loops contain only a single block) with the
431  // gpu return and move the operations from the loop body block to the gpu
432  // launch body block. Do not move the entire block because of the difference
433  // in block arguments.
434  Operation &terminator = innermostForOp.getBody()->back();
435  Location terminatorLoc = terminator.getLoc();
436  terminator.erase();
437  builder.setInsertionPointToEnd(innermostForOp.getBody());
438  builder.create<gpu::ReturnOp>(terminatorLoc);
439  launchOp.body().front().getOperations().splice(
440  launchOp.body().front().begin(),
441  innermostForOp.getBody()->getOperations());
442 
443  // Remap the loop iterators to use block/thread identifiers instead. Loops
444  // may iterate from LB with step S whereas GPU thread/block ids always iterate
445  // from 0 to N with step 1. Therefore, loop induction variables are replaced
446  // with (gpu-thread/block-id * S) + LB.
447  builder.setInsertionPointToStart(&launchOp.body().front());
448  auto lbArgumentIt = std::next(launchOp.getKernelArguments().begin(),
449  originallyForwardedValues);
450  auto stepArgumentIt = std::next(lbArgumentIt, lbs.size());
451  for (auto en : llvm::enumerate(ivs)) {
452  Value id =
453  en.index() < numBlockDims
454  ? getDim3Value(launchOp.getBlockIds(), en.index())
455  : getDim3Value(launchOp.getThreadIds(), en.index() - numBlockDims);
456  Value step = steps[en.index()];
457  if (!isConstantOne(step))
458  id = builder.create<MulIOp>(rootForOp.getLoc(), step, id);
459 
460  Value ivReplacement =
461  builder.create<AddIOp>(rootForOp.getLoc(), *lbArgumentIt, id);
462  en.value().replaceAllUsesWith(ivReplacement);
463  replaceAllUsesInRegionWith(steps[en.index()], *stepArgumentIt,
464  launchOp.body());
465  std::advance(lbArgumentIt, 1);
466  std::advance(stepArgumentIt, 1);
467  }
468 
469  // Remap the values defined outside the body to use kernel arguments instead.
470  // The list of kernel arguments also contains the lower bounds for loops at
471  // trailing positions, make sure we don't touch those.
472  for (auto pair :
473  llvm::zip_first(valuesToForward, launchOp.getKernelArguments())) {
474  Value from = std::get<0>(pair);
475  Value to = std::get<1>(pair);
476  replaceAllUsesInRegionWith(from, to, launchOp.body());
477  }
478 
479  // We are done and can erase the original outermost loop.
480  rootForOp.erase();
481 }

483 // Generic loop to GPU kernel conversion function.
484 template <typename OpTy>
485 static LogicalResult convertLoopNestToGPULaunch(OpTy forOp,
486  unsigned numBlockDims,
487  unsigned numThreadDims) {
488  if (failed(checkLoopNestMappable(forOp, numBlockDims, numThreadDims)))
489  return failure();
490 
491  LoopToGpuConverter converter;
492  auto maybeInnerLoop =
493  converter.collectBounds(forOp, numBlockDims + numThreadDims);
494  if (!maybeInnerLoop)
495  return failure();
496  converter.createLaunch(forOp, *maybeInnerLoop, numBlockDims, numThreadDims);
497 
498  return success();
499 }
