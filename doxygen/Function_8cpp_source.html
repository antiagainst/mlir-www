88 LogicalResult FuncOp::verify() {
89  // If this function is external there is nothing to do.
90  if (isExternal())
91  return success();
92 
93  // Verify that the argument list of the function and the arg list of the entry
94  // block line up. The trait already verified that the number of arguments is
95  // the same between the signature and the block.
96  auto fnInputTypes = getType().getInputs();
97  Block &entryBlock = front();
98  for (unsigned i = 0, e = entryBlock.getNumArguments(); i != e; ++i)
99  if (fnInputTypes[i] != entryBlock.getArgument(i).getType())
100  return emitOpError("type of entry block argument #")
101  << i << '(' << entryBlock.getArgument(i).getType()
102  << ") must match the type of the corresponding argument in "
103  << "function signature(" << fnInputTypes[i] << ')';
104 
105  return success();
106 }

108 void FuncOp::eraseArguments(ArrayRef<unsigned> argIndices) {
109  auto oldType = getType();
110  int originalNumArgs = oldType.getNumInputs();
111  llvm::BitVector eraseIndices(originalNumArgs);
112  for (auto index : argIndices)
113  eraseIndices.set(index);
114  auto shouldEraseArg = [&](int i) { return eraseIndices.test(i); };
115 
116  // There are 3 things that need to be updated:
117  // - Function type.
118  // - Arg attrs.
119  // - Block arguments of entry block.
120 
121  // Update the function type and arg attrs.
122  SmallVector<Type, 4> newInputTypes;
123  SmallVector<NamedAttributeList, 4> newArgAttrs;
124  for (int i = 0; i < originalNumArgs; i++) {
125  if (shouldEraseArg(i))
126  continue;
127  newInputTypes.emplace_back(oldType.getInput(i));
128  newArgAttrs.emplace_back(getArgAttrDict(i));
129  }
130  setType(FunctionType::get(newInputTypes, oldType.getResults(), getContext()));
131  setAllArgAttrs(newArgAttrs);
132 
133  // Update the entry block's arguments.
134  // We do this in reverse so that we erase later indices before earlier
135  // indices, to avoid shifting the later indices.
136  Block &entry = front();
137  for (int i = 0; i < originalNumArgs; i++)
138  if (shouldEraseArg(originalNumArgs - i - 1))
139  entry.eraseArgument(originalNumArgs - i - 1);
140 }

162 void FuncOp::cloneInto(FuncOp dest, BlockAndValueMapping &mapper) {
163  // Add the attributes of this function to dest.
164  llvm::MapVector<Identifier, Attribute> newAttrs;
165  for (auto &attr : dest.getAttrs())
166  newAttrs.insert(attr);
167  for (auto &attr : getAttrs())
168  newAttrs.insert(attr);
169  dest.getOperation()->setAttrs(
170  DictionaryAttr::get(newAttrs.takeVector(), getContext()));
171 
172  // Clone the body.
173  getBody().cloneInto(&dest.getBody(), mapper);

176 /// Create a deep copy of this function and all of its blocks, remapping
177 /// any operands that use values outside of the function using the map that is
178 /// provided (leaving them alone if no entry is present). Replaces references
179 /// to cloned sub-values with the corresponding value that is copied, and adds
180 /// those mappings to the mapper.
181 FuncOp FuncOp::clone(BlockAndValueMapping &mapper) {
182  FunctionType newType = getType();
183 
184  // If the function has a body, then the user might be deleting arguments to
185  // the function by specifying them in the mapper. If so, we don't add the
186  // argument to the input type vector.
187  bool isExternalFn = isExternal();
188  if (!isExternalFn) {
189  SmallVector<Type, 4> inputTypes;
190  inputTypes.reserve(newType.getNumInputs());
191  for (unsigned i = 0, e = getNumArguments(); i != e; ++i)
192  if (!mapper.contains(getArgument(i)))
193  inputTypes.push_back(newType.getInput(i));
194  newType = FunctionType::get(inputTypes, newType.getResults(), getContext());
195  }
196 
197  // Create the new function.
198  FuncOp newFunc = cast<FuncOp>(getOperation()->cloneWithoutRegions());
199  newFunc.setType(newType);
200 
201  /// Set the argument attributes for arguments that aren't being replaced.
202  for (unsigned i = 0, e = getNumArguments(), destI = 0; i != e; ++i)
203  if (isExternalFn || !mapper.contains(getArgument(i)))
204  newFunc.setArgAttrs(destI++, getArgAttrs(i));
205 
206  /// Clone the current function into the new one and return it.
207  cloneInto(newFunc, mapper);
208  return newFunc;
209 }
