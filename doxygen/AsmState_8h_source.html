25 /// This class provides management for the lifetime of the state used when
26 /// printing the IR. It allows for alleviating the cost of recomputing the
27 /// internal state of the asm printer.
28 ///
29 /// The IR should not be mutated in-between invocations using this state, and
30 /// the IR being printed must not be an parent of the IR originally used to
31 /// initialize this state. This means that if a child operation is provided, a
32 /// parent operation cannot reuse this state.
33 class AsmState {
34 public:
35  /// Initialize the asm state at the level of the given operation.
36  AsmState(Operation *op);
37  ~AsmState();
38 
39  /// Return an instance of the internal implementation. Returns nullptr if the
40  /// state has not been initialized.
41  detail::AsmStateImpl &getImpl() { return *impl; }
42 
43 private:
44  AsmState() = delete;
45 
46  /// A pointer to allocated storage for the impl state.
47  std::unique_ptr<detail::AsmStateImpl> impl;
48 };
