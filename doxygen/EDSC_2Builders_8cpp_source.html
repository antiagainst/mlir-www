121 BlockHandle mlir::edsc::BlockHandle::create(ArrayRef<Type> argTypes) {
122  auto &currentB = ScopedContext::getBuilder();
123  auto *ib = currentB.getInsertionBlock();
124  auto ip = currentB.getInsertionPoint();
125  BlockHandle res;
126  res.block = ScopedContext::getBuilder().createBlock(ib->getParent());
127  // createBlock sets the insertion point inside the block.

152 static Optional<ValueHandle> emitStaticFor(ArrayRef<ValueHandle> lbs,
153  ArrayRef<ValueHandle> ubs,
154  int64_t step) {
155  if (lbs.size() != 1 || ubs.size() != 1)
156  return Optional<ValueHandle>();
157 
158  auto *lbDef = lbs.front().getValue().getDefiningOp();
159  auto *ubDef = ubs.front().getValue().getDefiningOp();
160  if (!lbDef || !ubDef)
161  return Optional<ValueHandle>();
162 
163  auto lbConst = dyn_cast<ConstantIndexOp>(lbDef);
164  auto ubConst = dyn_cast<ConstantIndexOp>(ubDef);
165  if (!lbConst || !ubConst)
166  return Optional<ValueHandle>();
167 
168  return ValueHandle::create<AffineForOp>(lbConst.getValue(),
169  ubConst.getValue(), step);
170 }

172 mlir::edsc::LoopBuilder mlir::edsc::LoopBuilder::makeAffine(
173  ValueHandle *iv, ArrayRef<ValueHandle> lbHandles,
174  ArrayRef<ValueHandle> ubHandles, int64_t step) {
175  mlir::edsc::LoopBuilder result;
176  if (auto staticFor = emitStaticFor(lbHandles, ubHandles, step)) {
177  *iv = staticFor.getValue();
178  } else {
179  SmallVector<Value, 4> lbs(lbHandles.begin(), lbHandles.end());
180  SmallVector<Value, 4> ubs(ubHandles.begin(), ubHandles.end());
181  *iv = ValueHandle::create<AffineForOp>(
182  lbs, ScopedContext::getBuilder().getMultiDimIdentityMap(lbs.size()),
183  ubs, ScopedContext::getBuilder().getMultiDimIdentityMap(ubs.size()),
184  step);
185  }
186  auto *body = getForInductionVarOwner(iv->getValue()).getBody();
187  result.enter(body, /*prev=*/1);
188  return result;
189 }

204 void mlir::edsc::LoopBuilder::operator()(function_ref<void(void)> fun) {
205  // Call to `exit` must be explicit and asymmetric (cannot happen in the
206  // destructor) because of ordering wrt comma operator.
207  /// The particular use case concerns nested blocks:
208  ///
209  /// ```c++
210  /// For (&i, lb, ub, 1)({
211  /// /--- destructor for this `For` is not always called before ...
212  /// V
213  /// For (&j1, lb, ub, 1)({
214  /// some_op_1,
215  /// }),
216  /// /--- ... this scope is entered, resulting in improperly nested IR.
217  /// V
218  /// For (&j2, lb, ub, 1)({
219  /// some_op_2,
220  /// }),
221  /// });
222  /// ```
223  if (fun)
224  fun();
225  exit();
226 }

245 void mlir::edsc::AffineLoopNestBuilder::operator()(
246  function_ref<void(void)> fun) {
247  if (fun)
248  fun();
249  // Iterate on the calling operator() on all the loops in the nest.
250  // The iteration order is from innermost to outermost because enter/exit needs
251  // to be asymmetric (i.e. enter() occurs on LoopBuilder construction, exit()
252  // occurs on calling operator()). The asymmetry is required for properly
253  // nesting imperfectly nested regions (see LoopBuilder::operator()).
254  for (auto lit = loops.rbegin(), eit = loops.rend(); lit != eit; ++lit)
255  (*lit)();
256 }

271  assert(loops.size() == ivs.size() && "Mismatch loops vs ivs size");

336 static std::pair<AffineExpr, Value>
337 categorizeValueByAffineType(MLIRContext *context, Value val, unsigned &numDims,
338  unsigned &numSymbols) {
339  AffineExpr d;
340  Value resultVal = nullptr;
341  if (auto constant = dyn_cast_or_null<ConstantIndexOp>(val.getDefiningOp())) {
342  d = getAffineConstantExpr(constant.getValue(), context);
343  } else if (isValidSymbol(val) && !isValidDim(val)) {
344  d = getAffineSymbolExpr(numSymbols++, context);
345  resultVal = val;
346  } else {
347  d = getAffineDimExpr(numDims++, context);
348  resultVal = val;
349  }
350  return std::make_pair(d, resultVal);
351 }

353 static ValueHandle createBinaryIndexHandle(
354  ValueHandle lhs, ValueHandle rhs,
355  function_ref<AffineExpr(AffineExpr, AffineExpr)> affCombiner) {
356  MLIRContext *context = ScopedContext::getContext();
357  unsigned numDims = 0, numSymbols = 0;
358  AffineExpr d0, d1;
359  Value v0, v1;
360  std::tie(d0, v0) =
361  categorizeValueByAffineType(context, lhs.getValue(), numDims, numSymbols);
362  std::tie(d1, v1) =
363  categorizeValueByAffineType(context, rhs.getValue(), numDims, numSymbols);
364  SmallVector<Value, 2> operands;
365  if (v0) {
366  operands.push_back(v0);
367  }
368  if (v1) {
369  operands.push_back(v1);
370  }
371  auto map = AffineMap::get(numDims, numSymbols, {affCombiner(d0, d1)});
372  // TODO: createOrFold when available.
373  return ValueHandle::createComposedAffineApply(map, operands);
374 }

376 template <typename IOp, typename FOp>
377 static ValueHandle createBinaryHandle(
378  ValueHandle lhs, ValueHandle rhs,
379  function_ref<AffineExpr(AffineExpr, AffineExpr)> affCombiner) {
380  auto thisType = lhs.getValue().getType();
381  auto thatType = rhs.getValue().getType();
382  assert(thisType == thatType && "cannot mix types in operators");
383  (void)thisType;
384  (void)thatType;
385  if (thisType.isIndex()) {
386  return createBinaryIndexHandle(lhs, rhs, affCombiner);
387  } else if (thisType.isa<IntegerType>()) {
388  return createBinaryHandle<IOp>(lhs, rhs);
389  } else if (thisType.isa<FloatType>()) {
390  return createBinaryHandle<FOp>(lhs, rhs);
391  } else if (thisType.isa<VectorType>() || thisType.isa<TensorType>()) {
392  auto aggregateType = thisType.cast<ShapedType>();
393  if (aggregateType.getElementType().isa<IntegerType>())
394  return createBinaryHandle<IOp>(lhs, rhs);
395  else if (aggregateType.getElementType().isa<FloatType>())
396  return createBinaryHandle<FOp>(lhs, rhs);
397  }
398  llvm_unreachable("failed to create a ValueHandle");
399 }
