22 static void collectElementsStatisticsDim(ElementsAttr attr,
23  unsigned numElements,
24  ArrayRef<int64_t> shape,
25  SmallVectorImpl<uint64_t> &indices,
26  uint64_t dim,
27  TensorAxisStatistics &statistics) {
28  // Recursive terminating condition.
29  if (dim >= shape.size())
30  return;
31 
32  if (dim < (shape.size() - 1)) {
33  // Recurse past dim.
34  for (uint64_t i = 0, s = shape[dim]; i < s; ++i) {
35  indices[dim] = i;
36  collectElementsStatisticsDim(attr, numElements, shape, indices, dim + 1,
37  statistics);
38  }
39  return;
40  }
41 
42  // Collection dim.
43  for (uint64_t i = 0, s = shape[dim]; i < s; ++i) {
44  indices[dim] = i;
45  double value = attr.getValue<FloatAttr>(indices).getValueAsDouble();
46  statistics.minValue = std::min(statistics.minValue, value);
47  statistics.maxValue = std::max(statistics.maxValue, value);
48  statistics.mean += value / numElements;
49  // TODO: Calculate a running variance.
50  }
51 }

53 static bool getElementsStatistics(ElementsAttr attr,
54  TensorAxisStatistics &statistics) {
55  statistics.clear();
56  statistics.minValue = std::numeric_limits<double>::infinity();
57  statistics.maxValue = -std::numeric_limits<double>::infinity();
58 
59  ShapedType sType = attr.getType();
60  if (!sType.hasStaticShape())
61  return false;
62  Type elementTy = sType.getElementType();
63  if (!elementTy.isa<FloatType>())
64  return false;
65 
66  SmallVector<uint64_t, 4> indices;
67  indices.resize(sType.getRank());
68  ArrayRef<int64_t> shape = sType.getShape();
69 
70  auto numElements = sType.getNumElements();
71  collectElementsStatisticsDim(attr, numElements, shape, indices, 0,
72  statistics);
73  statistics.sampleSize = numElements;
74 
75  return true;
76 }
