73 /// Convert integer binary operations to SPIR-V operations. Cannot use
74 /// tablegen for this. If the integer operation is on variables of IndexType,
75 /// the type of the return value of the replacement operation differs from
76 /// that of the replaced operation. This is not handled in tablegen-based
77 /// pattern specification.
78 // TODO(ravishankarm) : This should be moved into DRR.

80 class IntegerOpConversion final : public SPIRVOpLowering<StdOp> {
81 public:
82  using SPIRVOpLowering<StdOp>::SPIRVOpLowering;
83 
84  PatternMatchResult
85  matchAndRewrite(StdOp operation, ArrayRef<Value> operands,
86  ConversionPatternRewriter &rewriter) const override {
87  auto resultType =
88  this->typeConverter.convertType(operation.getResult().getType());
89  rewriter.template replaceOpWithNewOp<SPIRVOp>(
90  operation, resultType, operands, ArrayRef<NamedAttribute>());
91  return this->matchSuccess();
92  }

148 /// Performs the index computation to get to the element pointed to by
149 /// `indices` using the layout map of `baseType`.
150 
151 // TODO(ravishankarm) : This method assumes that the `origBaseType` is a
152 // MemRefType with AffineMap that has static strides. Handle dynamic strides
153 static spirv::AccessChainOp getElementPtr(OpBuilder &builder,
154  SPIRVTypeConverter &typeConverter,
155  Location loc, MemRefType origBaseType,
156  Value basePtr,
157  ArrayRef<Value> indices) {
158  // Get base and offset of the MemRefType and verify they are static.
159  int64_t offset;
160  SmallVector<int64_t, 4> strides;
161  if (failed(getStridesAndOffset(origBaseType, strides, offset)) ||
162  llvm::is_contained(strides, MemRefType::getDynamicStrideOrOffset())) {
163  return nullptr;
164  }
165 
166  auto indexType = typeConverter.getIndexType(builder.getContext());
167 
168  Value ptrLoc = nullptr;
169  assert(indices.size() == strides.size());
170  for (auto index : enumerate(indices)) {
171  Value strideVal = builder.create<spirv::ConstantOp>(
172  loc, indexType, IntegerAttr::get(indexType, strides[index.index()]));
173  Value update = builder.create<spirv::IMulOp>(loc, strideVal, index.value());
174  ptrLoc =
175  (ptrLoc ? builder.create<spirv::IAddOp>(loc, ptrLoc, update).getResult()
176  : update);
177  }
178  SmallVector<Value, 2> linearizedIndices;
179  // Add a '0' at the start to index into the struct.
180  linearizedIndices.push_back(builder.create<spirv::ConstantOp>(
181  loc, indexType, IntegerAttr::get(indexType, 0)));
182  linearizedIndices.push_back(ptrLoc);
183  return builder.create<spirv::AccessChainOp>(loc, basePtr, linearizedIndices);
184 }

190 PatternMatchResult ConstantCompositeOpConversion::matchAndRewrite(
191  ConstantOp constCompositeOp, ArrayRef<Value> operands,
192  ConversionPatternRewriter &rewriter) const {
193  auto compositeType =
194  constCompositeOp.getResult().getType().dyn_cast<RankedTensorType>();
195  if (!compositeType)
196  return matchFailure();
197 
198  auto spirvCompositeType = typeConverter.convertType(compositeType);
199  if (!spirvCompositeType)
200  return matchFailure();
201 
202  auto linearizedElements =
203  constCompositeOp.value().dyn_cast<DenseElementsAttr>();
204  if (!linearizedElements)
205  return matchFailure();
206 
207  // If composite type has rank greater than one, then perform linearization.
208  if (compositeType.getRank() > 1) {
209  auto linearizedType = RankedTensorType::get(compositeType.getNumElements(),
210  compositeType.getElementType());
211  linearizedElements = linearizedElements.reshape(linearizedType);
212  }
213 
214  rewriter.replaceOpWithNewOp<spirv::ConstantOp>(
215  constCompositeOp, spirvCompositeType, linearizedElements);
216  return matchSuccess();
217 }

223 PatternMatchResult ConstantIndexOpConversion::matchAndRewrite(
224  ConstantOp constIndexOp, ArrayRef<Value> operands,
225  ConversionPatternRewriter &rewriter) const {
226  if (!constIndexOp.getResult().getType().isa<IndexType>()) {
227  return matchFailure();
228  }
229  // The attribute has index type which is not directly supported in
230  // SPIR-V. Get the integer value and create a new IntegerAttr.
231  auto constAttr = constIndexOp.value().dyn_cast<IntegerAttr>();
232  if (!constAttr) {
233  return matchFailure();
234  }
235 
236  // Use the bitwidth set in the value attribute to decide the result type
237  // of the SPIR-V constant operation since SPIR-V does not support index
238  // types.
239  auto constVal = constAttr.getValue();
240  auto constValType = constAttr.getType().dyn_cast<IndexType>();
241  if (!constValType) {
242  return matchFailure();
243  }
244  auto spirvConstType =
245  typeConverter.convertType(constIndexOp.getResult().getType());
246  auto spirvConstVal =
247  rewriter.getIntegerAttr(spirvConstType, constAttr.getInt());
248  rewriter.replaceOpWithNewOp<spirv::ConstantOp>(constIndexOp, spirvConstType,
249  spirvConstVal);
250  return matchSuccess();
251 }

257 PatternMatchResult
258 CmpFOpConversion::matchAndRewrite(CmpFOp cmpFOp, ArrayRef<Value> operands,
259  ConversionPatternRewriter &rewriter) const {
260  CmpFOpOperandAdaptor cmpFOpOperands(operands);
261 
262  switch (cmpFOp.getPredicate()) {
263 #define DISPATCH(cmpPredicate, spirvOp) \
264  case cmpPredicate: \
265  rewriter.replaceOpWithNewOp<spirvOp>(cmpFOp, cmpFOp.getResult().getType(), \
266  cmpFOpOperands.lhs(), \
267  cmpFOpOperands.rhs()); \
268  return matchSuccess();
269 
270  // Ordered.
271  DISPATCH(CmpFPredicate::OEQ, spirv::FOrdEqualOp);
272  DISPATCH(CmpFPredicate::OGT, spirv::FOrdGreaterThanOp);
273  DISPATCH(CmpFPredicate::OGE, spirv::FOrdGreaterThanEqualOp);
274  DISPATCH(CmpFPredicate::OLT, spirv::FOrdLessThanOp);
275  DISPATCH(CmpFPredicate::OLE, spirv::FOrdLessThanEqualOp);
276  DISPATCH(CmpFPredicate::ONE, spirv::FOrdNotEqualOp);
277  // Unordered.
278  DISPATCH(CmpFPredicate::UEQ, spirv::FUnordEqualOp);
279  DISPATCH(CmpFPredicate::UGT, spirv::FUnordGreaterThanOp);
280  DISPATCH(CmpFPredicate::UGE, spirv::FUnordGreaterThanEqualOp);
281  DISPATCH(CmpFPredicate::ULT, spirv::FUnordLessThanOp);
282  DISPATCH(CmpFPredicate::ULE, spirv::FUnordLessThanEqualOp);
283  DISPATCH(CmpFPredicate::UNE, spirv::FUnordNotEqualOp);
284 
285 #undef DISPATCH
286 
287  default:
288  break;
289  }
290  return matchFailure();
291 }
