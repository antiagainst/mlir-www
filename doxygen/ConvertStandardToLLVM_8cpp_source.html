110 // Function types are converted to LLVM Function types by recursively converting
111 // argument and result types. If MLIR Function has zero results, the LLVM
112 // Function has one VoidType result. If MLIR Function has more than one result,
113 // they are into an LLVM StructType in their order of appearance.
114 LLVM::LLVMType LLVMTypeConverter::convertFunctionSignature(
115  FunctionType type, bool isVariadic,
116  LLVMTypeConverter::SignatureConversion &result) {
117  // Convert argument types one by one and check for errors.
118  for (auto &en : llvm::enumerate(type.getInputs())) {
119  Type type = en.value();
120  auto converted = convertType(type).dyn_cast_or_null<LLVM::LLVMType>();
121  if (!converted)
122  return {};
123  if (type.isa<MemRefType>() || type.isa<UnrankedMemRefType>())
124  converted = converted.getPointerTo();
125  result.addInputs(en.index(), converted);
126  }
127 
128  SmallVector<LLVM::LLVMType, 8> argTypes;
129  argTypes.reserve(llvm::size(result.getConvertedTypes()));
130  for (Type type : result.getConvertedTypes())
131  argTypes.push_back(unwrap(type));
132 
133  // If function does not return anything, create the void result type,
134  // if it returns on element, convert it, otherwise pack the result types into
135  // a struct.
136  LLVM::LLVMType resultType =
137  type.getNumResults() == 0
138  ? LLVM::LLVMType::getVoidTy(llvmDialect)
139  : unwrap(packFunctionResults(type.getResults()));
140  if (!resultType)
141  return {};
142  return LLVM::LLVMType::getFunctionTy(resultType, argTypes, isVariadic);
143 }

145 // Convert a MemRef to an LLVM type. The result is a MemRef descriptor which
146 // contains:
147 // 1. the pointer to the data buffer, followed by
148 // 2. a lowered `index`-type integer containing the distance between the
149 // beginning of the buffer and the first element to be accessed through the
150 // view, followed by
151 // 3. an array containing as many `index`-type integers as the rank of the
152 // MemRef: the array represents the size, in number of elements, of the memref
153 // along the given dimension. For constant MemRef dimensions, the
154 // corresponding size entry is a constant whose runtime value must match the
155 // static value, followed by
156 // 4. a second array containing as many `index`-type integers as the rank of
157 // the MemRef: the second array represents the "stride" (in tensor abstraction
158 // sense), i.e. the number of consecutive elements of the underlying buffer.
159 // TODO(ntv, zinenko): add assertions for the static cases.
160 //
161 // template <typename Elem, size_t Rank>
162 // struct {
163 // Elem *allocatedPtr;
164 // Elem *alignedPtr;
165 // int64_t offset;
166 // int64_t sizes[Rank]; // omitted when rank == 0
167 // int64_t strides[Rank]; // omitted when rank == 0
168 // };
169 static constexpr unsigned kAllocatedPtrPosInMemRefDescriptor = 0;
170 static constexpr unsigned kAlignedPtrPosInMemRefDescriptor = 1;
171 static constexpr unsigned kOffsetPosInMemRefDescriptor = 2;
172 static constexpr unsigned kSizePosInMemRefDescriptor = 3;
173 static constexpr unsigned kStridePosInMemRefDescriptor = 4;
174 Type LLVMTypeConverter::convertMemRefType(MemRefType type) {
175  int64_t offset;
176  SmallVector<int64_t, 4> strides;
177  bool strideSuccess = succeeded(getStridesAndOffset(type, strides, offset));
178  assert(strideSuccess &&
179  "Non-strided layout maps must have been normalized away");
180  (void)strideSuccess;
181  LLVM::LLVMType elementType = unwrap(convertType(type.getElementType()));
182  if (!elementType)
183  return {};
184  auto ptrTy = elementType.getPointerTo(type.getMemorySpace());
185  auto indexTy = getIndexType();
186  auto rank = type.getRank();
187  if (rank > 0) {
188  auto arrayTy = LLVM::LLVMType::getArrayTy(indexTy, type.getRank());
189  return LLVM::LLVMType::getStructTy(ptrTy, ptrTy, indexTy, arrayTy, arrayTy);
190  }
191  return LLVM::LLVMType::getStructTy(ptrTy, ptrTy, indexTy);
192 }

210 // Convert an n-D vector type to an LLVM vector type via (n-1)-D array type when
211 // n > 1.
212 // For example, `vector<4 x f32>` converts to `!llvm.type<"<4 x float>">` and
213 // `vector<4 x 8 x 16 f32>` converts to `!llvm<"[4 x [8 x <16 x float>]]">`.
214 Type LLVMTypeConverter::convertVectorType(VectorType type) {
215  auto elementType = unwrap(convertType(type.getElementType()));
216  if (!elementType)
217  return {};
218  auto vectorType =
219  LLVM::LLVMType::getVectorTy(elementType, type.getShape().back());
220  auto shape = type.getShape();
221  for (int i = shape.size() - 2; i >= 0; --i)
222  vectorType = LLVM::LLVMType::getArrayTy(vectorType, shape[i]);
223  return vectorType;
224 }

226 // Dispatch based on the actual type. Return null type on error.
227 Type LLVMTypeConverter::convertStandardType(Type t) {
228  return TypeSwitch<Type, Type>(t)
229  .Case([&](FloatType type) { return convertFloatType(type); })
230  .Case([&](FunctionType type) { return convertFunctionType(type); })
231  .Case([&](IndexType type) { return convertIndexType(type); })
232  .Case([&](IntegerType type) { return convertIntegerType(type); })
233  .Case([&](MemRefType type) { return convertMemRefType(type); })
234  .Case([&](UnrankedMemRefType type) {
235  return convertUnrankedMemRefType(type);
236  })
237  .Case([&](VectorType type) { return convertVectorType(type); })
238  .Case([](LLVM::LLVMType type) { return type; })
239  .Default([](Type) { return Type(); });
240 }

288 /// Builds IR creating a MemRef descriptor that represents `type` and
289 /// populates it with static shape and stride information extracted from the
290 /// type.
291 MemRefDescriptor
292 MemRefDescriptor::fromStaticShape(OpBuilder &builder, Location loc,
293  LLVMTypeConverter &typeConverter,
294  MemRefType type, Value memory) {
295  assert(type.hasStaticShape() && "unexpected dynamic shape");
296  assert(type.getAffineMaps().empty() && "unexpected layout map");
297 
298  auto convertedType = typeConverter.convertType(type);
299  assert(convertedType && "unexpected failure in memref type conversion");
300 
301  auto descr = MemRefDescriptor::undef(builder, loc, convertedType);
302  descr.setAllocatedPtr(builder, loc, memory);
303  descr.setAlignedPtr(builder, loc, memory);
304  descr.setConstantOffset(builder, loc, 0);
305 
306  // Fill in sizes and strides, in reverse order to simplify stride
307  // calculation.
308  uint64_t runningStride = 1;
309  for (unsigned i = type.getRank(); i > 0; --i) {
310  unsigned dim = i - 1;
311  descr.setConstantSize(builder, loc, dim, type.getDimSize(dim));
312  descr.setConstantStride(builder, loc, dim, runningStride);
313  runningStride *= type.getDimSize(dim);
314  }
315  return descr;
316 }

497 struct FuncOpConversion : public LLVMLegalizationPattern<FuncOp> {
498  using LLVMLegalizationPattern<FuncOp>::LLVMLegalizationPattern;
499 
500  PatternMatchResult
501  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
502  ConversionPatternRewriter &rewriter) const override {
503  auto funcOp = cast<FuncOp>(op);
504  FunctionType type = funcOp.getType();
505 
506  // Store the positions of memref-typed arguments so that we can emit loads
507  // from them to follow the calling convention.
508  SmallVector<unsigned, 4> promotedArgIndices;
509  promotedArgIndices.reserve(type.getNumInputs());
510  for (auto en : llvm::enumerate(type.getInputs())) {
511  if (en.value().isa<MemRefType>() || en.value().isa<UnrankedMemRefType>())
512  promotedArgIndices.push_back(en.index());
513  }
514 
515  // Convert the original function arguments. Struct arguments are promoted to
516  // pointer to struct arguments to allow calling external functions with
517  // various ABIs (e.g. compiled from C/C++ on platform X).
518  auto varargsAttr = funcOp.getAttrOfType<BoolAttr>("std.varargs");
519  TypeConverter::SignatureConversion result(funcOp.getNumArguments());
520  auto llvmType = lowering.convertFunctionSignature(
521  funcOp.getType(), varargsAttr && varargsAttr.getValue(), result);
522 
523  // Only retain those attributes that are not constructed by build.
524  SmallVector<NamedAttribute, 4> attributes;
525  for (const auto &attr : funcOp.getAttrs()) {
526  if (attr.first.is(SymbolTable::getSymbolAttrName()) ||
527  attr.first.is(impl::getTypeAttrName()) ||
528  attr.first.is("std.varargs"))
529  continue;
530  attributes.push_back(attr);
531  }
532 
533  // Create an LLVM function, use external linkage by default until MLIR
534  // functions have linkage.
535  auto newFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
536  op->getLoc(), funcOp.getName(), llvmType, LLVM::Linkage::External,
537  attributes);
538  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
539  newFuncOp.end());
540 
541  // Tell the rewriter to convert the region signature.
542  rewriter.applySignatureConversion(&newFuncOp.getBody(), result);
543 
544  // Insert loads from memref descriptor pointers in function bodies.
545  if (!newFuncOp.getBody().empty()) {
546  Block *firstBlock = &newFuncOp.getBody().front();
547  rewriter.setInsertionPoint(firstBlock, firstBlock->begin());
548  for (unsigned idx : promotedArgIndices) {
549  BlockArgument arg = firstBlock->getArgument(idx);
550  Value loaded = rewriter.create<LLVM::LoadOp>(funcOp.getLoc(), arg);
551  rewriter.replaceUsesOfBlockArgument(arg, loaded);
552  }
553  }
554 
555  rewriter.eraseOp(op);
556  return matchSuccess();
557  }

580  assert(vectorType.getRank() > 1 && "expected >1D vector type");
581  NDVectorTypeInfo info;
582  info.llvmArrayTy =
583  converter.convertType(vectorType).dyn_cast<LLVM::LLVMType>();
584  if (!info.llvmArrayTy)
585  return info;
586  info.arraySizes.reserve(vectorType.getRank() - 1);
587  auto llvmTy = info.llvmArrayTy;
588  while (llvmTy.isArrayTy()) {
589  info.arraySizes.push_back(llvmTy.getArrayNumElements());
590  llvmTy = llvmTy.getArrayElementType();
591  }
592  if (!llvmTy.isVectorTy())
593  return info;
594  info.llvmVectorTy = llvmTy;
595  return info;
596 }

598 // Express `linearIndex` in terms of coordinates of `basis`.
599 // Returns the empty vector when linearIndex is out of the range [0, P] where
600 // P is the product of all the basis coordinates.
601 //
602 // Prerequisites:
603 // Basis is an array of nonnegative integers (signed type inherited from
604 // vector shape type).
605 static SmallVector<int64_t, 4> getCoordinates(ArrayRef<int64_t> basis,
606  unsigned linearIndex) {
607  SmallVector<int64_t, 4> res;
608  res.reserve(basis.size());
609  for (unsigned basisElement : llvm::reverse(basis)) {
610  res.push_back(linearIndex % basisElement);
611  linearIndex = linearIndex / basisElement;
612  }
613  if (linearIndex > 0)
614  return {};
615  std::reverse(res.begin(), res.end());
616  return res;
617 }
618 
619 // Iterate of linear index, convert to coords space and insert splatted 1-D
620 // vector in each position.
621 template <typename Lambda>
622 void nDVectorIterate(const NDVectorTypeInfo &info, OpBuilder &builder,
623  Lambda fun) {
624  unsigned ub = 1;
625  for (auto s : info.arraySizes)
626  ub *= s;
627  for (unsigned linearIndex = 0; linearIndex < ub; ++linearIndex) {
628  auto coords = getCoordinates(info.arraySizes, linearIndex);
629  // Linear index is out of bounds, we are done.
630  if (coords.empty())
631  break;
632  assert(coords.size() == info.arraySizes.size());
633  auto position = builder.getI64ArrayAttr(coords);
634  fun(position);
635  }
636 }

640 // LLVM Dialect Ops.
641 template <typename SourceOp, typename TargetOp>
642 struct OneToOneLLVMOpLowering : public LLVMLegalizationPattern<SourceOp> {
643  using LLVMLegalizationPattern<SourceOp>::LLVMLegalizationPattern;
644  using Super = OneToOneLLVMOpLowering<SourceOp, TargetOp>;
645 
646  // Convert the type of the result to an LLVM type, pass operands as is,
647  // preserve attributes.
648  PatternMatchResult
649  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
650  ConversionPatternRewriter &rewriter) const override {
651  unsigned numResults = op->getNumResults();
652 
653  Type packedType;
654  if (numResults != 0) {
655  packedType = this->lowering.packFunctionResults(op->getResultTypes());
656  if (!packedType)
657  return this->matchFailure();
658  }
659 
660  auto newOp = rewriter.create<TargetOp>(op->getLoc(), packedType, operands,
661  op->getAttrs());
662 
663  // If the operation produced 0 or 1 result, return them immediately.
664  if (numResults == 0)
665  return rewriter.eraseOp(op), this->matchSuccess();
666  if (numResults == 1)
667  return rewriter.replaceOp(op, newOp.getOperation()->getResult(0)),
668  this->matchSuccess();
669 
670  // Otherwise, it had been converted to an operation producing a structure.
671  // Extract individual results from the structure and return them as list.
672  SmallVector<Value, 4> results;
673  results.reserve(numResults);
674  for (unsigned i = 0; i < numResults; ++i) {
675  auto type = this->lowering.convertType(op->getResult(i).getType());
676  results.push_back(rewriter.create<LLVM::ExtractValueOp>(
677  op->getLoc(), type, newOp.getOperation()->getResult(0),
678  rewriter.getI64ArrayAttr(i)));
679  }
680  rewriter.replaceOp(op, results);
681  return this->matchSuccess();
682  }

702 // Basic lowering implementation for rewriting from Standard Ops to LLVM Dialect
703 // Ops for N-ary ops with one result. This supports higher-dimensional vector
704 // types.
705 template <typename SourceOp, typename TargetOp, unsigned OpCount>
706 struct NaryOpLLVMOpLowering : public LLVMLegalizationPattern<SourceOp> {
707  using LLVMLegalizationPattern<SourceOp>::LLVMLegalizationPattern;
708  using Super = NaryOpLLVMOpLowering<SourceOp, TargetOp, OpCount>;
709 
710  // Convert the type of the result to an LLVM type, pass operands as is,
711  // preserve attributes.
712  PatternMatchResult
713  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
714  ConversionPatternRewriter &rewriter) const override {
715  ValidateOpCount<SourceOp, OpCount>();
716  static_assert(
717  std::is_base_of<OpTrait::OneResult<SourceOp>, SourceOp>::value,
718  "expected single result op");
719  static_assert(std::is_base_of<OpTrait::SameOperandsAndResultType<SourceOp>,
720  SourceOp>::value,
721  "expected same operands and result type");
722 
723  // Cannot convert ops if their operands are not of LLVM type.
724  for (Value operand : operands) {
725  if (!operand || !operand.getType().isa<LLVM::LLVMType>())
726  return this->matchFailure();
727  }
728 
729  auto loc = op->getLoc();
730  auto llvmArrayTy = operands[0].getType().cast<LLVM::LLVMType>();
731 
732  if (!llvmArrayTy.isArrayTy()) {
733  auto newOp = rewriter.create<TargetOp>(
734  op->getLoc(), operands[0].getType(), operands, op->getAttrs());
735  rewriter.replaceOp(op, newOp.getResult());
736  return this->matchSuccess();
737  }
738 
739  auto vectorType = op->getResult(0).getType().dyn_cast<VectorType>();
740  if (!vectorType)
741  return this->matchFailure();
742  auto vectorTypeInfo = extractNDVectorTypeInfo(vectorType, this->lowering);
743  auto llvmVectorTy = vectorTypeInfo.llvmVectorTy;
744  if (!llvmVectorTy || llvmArrayTy != vectorTypeInfo.llvmArrayTy)
745  return this->matchFailure();
746 
747  Value desc = rewriter.create<LLVM::UndefOp>(loc, llvmArrayTy);
748  nDVectorIterate(vectorTypeInfo, rewriter, [&](ArrayAttr position) {
749  // For this unrolled `position` corresponding to the `linearIndex`^th
750  // element, extract operand vectors
751  SmallVector<Value, OpCount> extractedOperands;
752  for (unsigned i = 0; i < OpCount; ++i) {
753  extractedOperands.push_back(rewriter.create<LLVM::ExtractValueOp>(
754  loc, llvmVectorTy, operands[i], position));
755  }
756  Value newVal = rewriter.create<TargetOp>(
757  loc, llvmVectorTy, extractedOperands, op->getAttrs());
758  desc = rewriter.create<LLVM::InsertValueOp>(loc, llvmArrayTy, desc,
759  newVal, position);
760  });
761  rewriter.replaceOp(op, desc);
762  return this->matchSuccess();
763  }
764 };

879 // An `alloc` is converted into a definition of a memref descriptor value and
880 // a call to `malloc` to allocate the underlying data buffer. The memref
881 // descriptor is of the LLVM structure type where:
882 // 1. the first element is a pointer to the allocated (typed) data buffer,
883 // 2. the second element is a pointer to the (typed) payload, aligned to the
884 // specified alignment,
885 // 3. the remaining elements serve to store all the sizes and strides of the
886 // memref using LLVM-converted `index` type.
887 //
888 // Alignment is obtained by allocating `alignment - 1` more bytes than requested
889 // and shifting the aligned pointer relative to the allocated memory. If
890 // alignment is unspecified, the two pointers are equal.
891 struct AllocOpLowering : public LLVMLegalizationPattern<AllocOp> {
892  using LLVMLegalizationPattern<AllocOp>::LLVMLegalizationPattern;
893 
894  AllocOpLowering(LLVM::LLVMDialect &dialect_, LLVMTypeConverter &converter,
895  bool useAlloca = false)
896  : LLVMLegalizationPattern<AllocOp>(dialect_, converter),
897  useAlloca(useAlloca) {}
898 
899  PatternMatchResult match(Operation *op) const override {
900  MemRefType type = cast<AllocOp>(op).getType();
901  if (isSupportedMemRefType(type))
902  return matchSuccess();
903 
904  int64_t offset;
905  SmallVector<int64_t, 4> strides;
906  auto successStrides = getStridesAndOffset(type, strides, offset);
907  if (failed(successStrides))
908  return matchFailure();
909 
910  // Dynamic strides are ok if they can be deduced from dynamic sizes (which
911  // is guaranteed when succeeded(successStrides)). Dynamic offset however can
912  // never be alloc'ed.
913  if (offset == MemRefType::getDynamicStrideOrOffset())
914  return matchFailure();
915 
916  return matchSuccess();
917  }
918 
919  void rewrite(Operation *op, ArrayRef<Value> operands,
920  ConversionPatternRewriter &rewriter) const override {
921  auto loc = op->getLoc();
922  auto allocOp = cast<AllocOp>(op);
923  MemRefType type = allocOp.getType();
924 
925  // Get actual sizes of the memref as values: static sizes are constant
926  // values and dynamic sizes are passed to 'alloc' as operands. In case of
927  // zero-dimensional memref, assume a scalar (size 1).
928  SmallVector<Value, 4> sizes;
929  sizes.reserve(type.getRank());
930  unsigned i = 0;
931  for (int64_t s : type.getShape())
932  sizes.push_back(s == -1 ? operands[i++]
933  : createIndexConstant(rewriter, loc, s));
934  if (sizes.empty())
935  sizes.push_back(createIndexConstant(rewriter, loc, 1));
936 
937  // Compute the total number of memref elements.
938  Value cumulativeSize = sizes.front();
939  for (unsigned i = 1, e = sizes.size(); i < e; ++i)
940  cumulativeSize = rewriter.create<LLVM::MulOp>(
941  loc, getIndexType(), ArrayRef<Value>{cumulativeSize, sizes[i]});
942 
943  // Compute the size of an individual element. This emits the MLIR equivalent
944  // of the following sizeof(...) implementation in LLVM IR:
945  // %0 = getelementptr %elementType* null, %indexType 1
946  // %1 = ptrtoint %elementType* %0 to %indexType
947  // which is a common pattern of getting the size of a type in bytes.
948  auto elementType = type.getElementType();
949  auto convertedPtrType =
950  lowering.convertType(elementType).cast<LLVM::LLVMType>().getPointerTo();
951  auto nullPtr = rewriter.create<LLVM::NullOp>(loc, convertedPtrType);
952  auto one = createIndexConstant(rewriter, loc, 1);
953  auto gep = rewriter.create<LLVM::GEPOp>(loc, convertedPtrType,
954  ArrayRef<Value>{nullPtr, one});
955  auto elementSize =
956  rewriter.create<LLVM::PtrToIntOp>(loc, getIndexType(), gep);
957  cumulativeSize = rewriter.create<LLVM::MulOp>(
958  loc, getIndexType(), ArrayRef<Value>{cumulativeSize, elementSize});
959 
960  // Allocate the underlying buffer and store a pointer to it in the MemRef
961  // descriptor.
962  Value allocated = nullptr;
963  int alignment = 0;
964  Value alignmentValue = nullptr;
965  if (auto alignAttr = allocOp.alignment())
966  alignment = alignAttr.getValue().getSExtValue();
967 
968  if (useAlloca) {
969  allocated = rewriter.create<LLVM::AllocaOp>(loc, getVoidPtrType(),
970  cumulativeSize, alignment);
971  } else {
972  // Insert the `malloc` declaration if it is not already present.
973  auto module = op->getParentOfType<ModuleOp>();
974  auto mallocFunc = module.lookupSymbol<LLVM::LLVMFuncOp>("malloc");
975  if (!mallocFunc) {
976  OpBuilder moduleBuilder(
977  op->getParentOfType<ModuleOp>().getBodyRegion());
978  mallocFunc = moduleBuilder.create<LLVM::LLVMFuncOp>(
979  rewriter.getUnknownLoc(), "malloc",
980  LLVM::LLVMType::getFunctionTy(getVoidPtrType(), getIndexType(),
981  /*isVarArg=*/false));
982  }
983  if (alignment != 0) {
984  alignmentValue = createIndexConstant(rewriter, loc, alignment);
985  cumulativeSize = rewriter.create<LLVM::SubOp>(
986  loc,
987  rewriter.create<LLVM::AddOp>(loc, cumulativeSize, alignmentValue),
988  one);
989  }
990  allocated = rewriter
991  .create<LLVM::CallOp>(
992  loc, getVoidPtrType(),
993  rewriter.getSymbolRefAttr(mallocFunc), cumulativeSize)
994  .getResult(0);
995  }
996 
997  auto structElementType = lowering.convertType(elementType);
998  auto elementPtrType = structElementType.cast<LLVM::LLVMType>().getPointerTo(
999  type.getMemorySpace());
1000  Value bitcastAllocated = rewriter.create<LLVM::BitcastOp>(
1001  loc, elementPtrType, ArrayRef<Value>(allocated));
1002 
1003  int64_t offset;
1004  SmallVector<int64_t, 4> strides;
1005  auto successStrides = getStridesAndOffset(type, strides, offset);
1006  assert(succeeded(successStrides) && "unexpected non-strided memref");
1007  (void)successStrides;
1008  assert(offset != MemRefType::getDynamicStrideOrOffset() &&
1009  "unexpected dynamic offset");
1010 
1011  // 0-D memref corner case: they have size 1 ...
1012  assert(((type.getRank() == 0 && strides.empty() && sizes.size() == 1) ||
1013  (strides.size() == sizes.size())) &&
1014  "unexpected number of strides");
1015 
1016  // Create the MemRef descriptor.
1017  auto structType = lowering.convertType(type);
1018  auto memRefDescriptor = MemRefDescriptor::undef(rewriter, loc, structType);
1019  // Field 1: Allocated pointer, used for malloc/free.
1020  memRefDescriptor.setAllocatedPtr(rewriter, loc, bitcastAllocated);
1021 
1022  // Field 2: Actual aligned pointer to payload.
1023  Value bitcastAligned = bitcastAllocated;
1024  if (!useAlloca && alignment != 0) {
1025  assert(alignmentValue);
1026  // offset = (align - (ptr % align))% align
1027  Value intVal = rewriter.create<LLVM::PtrToIntOp>(
1028  loc, this->getIndexType(), allocated);
1029  Value ptrModAlign =
1030  rewriter.create<LLVM::URemOp>(loc, intVal, alignmentValue);
1031  Value subbed =
1032  rewriter.create<LLVM::SubOp>(loc, alignmentValue, ptrModAlign);
1033  Value offset = rewriter.create<LLVM::URemOp>(loc, subbed, alignmentValue);
1034  Value aligned = rewriter.create<LLVM::GEPOp>(loc, allocated.getType(),
1035  allocated, offset);
1036  bitcastAligned = rewriter.create<LLVM::BitcastOp>(
1037  loc, elementPtrType, ArrayRef<Value>(aligned));
1038  }
1039  memRefDescriptor.setAlignedPtr(rewriter, loc, bitcastAligned);
1040 
1041  // Field 3: Offset in aligned pointer.
1042  memRefDescriptor.setOffset(rewriter, loc,
1043  createIndexConstant(rewriter, loc, offset));
1044 
1045  if (type.getRank() == 0)
1046  // No size/stride descriptor in memref, return the descriptor value.
1047  return rewriter.replaceOp(op, {memRefDescriptor});
1048 
1049  // Fields 4 and 5: Sizes and strides of the strided MemRef.
1050  // Store all sizes in the descriptor. Only dynamic sizes are passed in as
1051  // operands to AllocOp.
1052  Value runningStride = nullptr;
1053  // Iterate strides in reverse order, compute runningStride and strideValues.
1054  auto nStrides = strides.size();
1055  SmallVector<Value, 4> strideValues(nStrides, nullptr);
1056  for (unsigned i = 0; i < nStrides; ++i) {
1057  int64_t index = nStrides - 1 - i;
1058  if (strides[index] == MemRefType::getDynamicStrideOrOffset())
1059  // Identity layout map is enforced in the match function, so we compute:
1060  // `runningStride *= sizes[index + 1]`
1061  runningStride =
1062  runningStride
1063  ? rewriter.create<LLVM::MulOp>(loc, runningStride,
1064  sizes[index + 1])
1065  : createIndexConstant(rewriter, loc, 1);
1066  else
1067  runningStride = createIndexConstant(rewriter, loc, strides[index]);
1068  strideValues[index] = runningStride;
1069  }
1070  // Fill size and stride descriptors in memref.
1071  for (auto indexedSize : llvm::enumerate(sizes)) {
1072  int64_t index = indexedSize.index();
1073  memRefDescriptor.setSize(rewriter, loc, index, indexedSize.value());
1074  memRefDescriptor.setStride(rewriter, loc, index, strideValues[index]);
1075  }
1076 
1077  // Return the final value of the descriptor.
1078  rewriter.replaceOp(op, {memRefDescriptor});
1079  }
1080 
1081  bool useAlloca;
1082 };

1085 // passes the pointer to the MemRef across function boundaries.

1087 struct CallOpInterfaceLowering : public LLVMLegalizationPattern<CallOpType> {

1089  using Super = CallOpInterfaceLowering<CallOpType>;
1090  using Base = LLVMLegalizationPattern<CallOpType>;

1092  PatternMatchResult
1093  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
1094  ConversionPatternRewriter &rewriter) const override {
1095  OperandAdaptor<CallOpType> transformed(operands);
1096  auto callOp = cast<CallOpType>(op);
1097 
1098  // Pack the result types into a struct.
1099  Type packedResult;
1100  unsigned numResults = callOp.getNumResults();
1101  auto resultTypes = llvm::to_vector<4>(callOp.getResultTypes());
1102 
1103  for (Type resType : resultTypes) {
1104  assert(!resType.isa<UnrankedMemRefType>() &&
1105  "Returning unranked memref is not supported. Pass result as an"
1106  "argument instead.");
1107  (void)resType;
1108  }
1109 
1110  if (numResults != 0) {
1111  if (!(packedResult = this->lowering.packFunctionResults(resultTypes)))
1112  return this->matchFailure();
1113  }
1114 
1115  auto promoted = this->lowering.promoteMemRefDescriptors(
1116  op->getLoc(), /*opOperands=*/op->getOperands(), operands, rewriter);
1117  auto newOp = rewriter.create<LLVM::CallOp>(op->getLoc(), packedResult,
1118  promoted, op->getAttrs());
1119 
1120  // If < 2 results, packing did not do anything and we can just return.
1121  if (numResults < 2) {
1122  rewriter.replaceOp(op, newOp.getResults());
1123  return this->matchSuccess();
1124  }
1125 
1126  // Otherwise, it had been converted to an operation producing a structure.
1127  // Extract individual results from the structure and return them as list.
1128  // TODO(aminim, ntv, riverriddle, zinenko): this seems like patching around
1129  // a particular interaction between MemRefType and CallOp lowering. Find a
1130  // way to avoid special casing.
1131  SmallVector<Value, 4> results;
1132  results.reserve(numResults);
1133  for (unsigned i = 0; i < numResults; ++i) {

1371 struct LoadStoreOpLowering : public LLVMLegalizationPattern<Derived> {
1372  using LLVMLegalizationPattern<Derived>::LLVMLegalizationPattern;
1373  using Base = LoadStoreOpLowering<Derived>;

1378  : this->matchFailure();

1381  // Given subscript indices and array sizes in row-major order,
1382  // i_n, i_{n-1}, ..., i_1
1383  // s_n, s_{n-1}, ..., s_1
1384  // obtain a value that corresponds to the linearized subscript
1385  // \sum_k i_k * \prod_{j=1}^{k-1} s_j
1386  // by accumulating the running linearized value.
1387  // Note that `indices` and `allocSizes` are passed in the same order as they
1388  // appear in load/store operations and memref type declarations.
1389  Value linearizeSubscripts(ConversionPatternRewriter &builder, Location loc,
1390  ArrayRef<Value> indices,
1391  ArrayRef<Value> allocSizes) const {
1392  assert(indices.size() == allocSizes.size() &&
1393  "mismatching number of indices and allocation sizes");
1394  assert(!indices.empty() && "cannot linearize a 0-dimensional access");
1395 
1396  Value linearized = indices.front();
1397  for (int i = 1, nSizes = allocSizes.size(); i < nSizes; ++i) {
1398  linearized = builder.create<LLVM::MulOp>(
1399  loc, this->getIndexType(),
1400  ArrayRef<Value>{linearized, allocSizes[i]});
1401  linearized = builder.create<LLVM::AddOp>(
1402  loc, this->getIndexType(), ArrayRef<Value>{linearized, indices[i]});
1403  }
1404  return linearized;
1405  }
1406 
1407  // This is a strided getElementPtr variant that linearizes subscripts as:
1408  // `base_offset + index_0 * stride_0 + ... + index_n * stride_n`.
1409  Value getStridedElementPtr(Location loc, Type elementTypePtr,
1410  Value descriptor, ArrayRef<Value> indices,
1411  ArrayRef<int64_t> strides, int64_t offset,
1412  ConversionPatternRewriter &rewriter) const {
1413  MemRefDescriptor memRefDescriptor(descriptor);
1414 
1415  Value base = memRefDescriptor.alignedPtr(rewriter, loc);
1416  Value offsetValue = offset == MemRefType::getDynamicStrideOrOffset()
1417  ? memRefDescriptor.offset(rewriter, loc)
1418  : this->createIndexConstant(rewriter, loc, offset);
1419 
1420  for (int i = 0, e = indices.size(); i < e; ++i) {
1421  Value stride = strides[i] == MemRefType::getDynamicStrideOrOffset()
1422  ? memRefDescriptor.stride(rewriter, loc, i)
1423  : this->createIndexConstant(rewriter, loc, strides[i]);
1424  Value additionalOffset =
1425  rewriter.create<LLVM::MulOp>(loc, indices[i], stride);
1426  offsetValue =
1427  rewriter.create<LLVM::AddOp>(loc, offsetValue, additionalOffset);
1428  }
1429  return rewriter.create<LLVM::GEPOp>(loc, elementTypePtr, base, offsetValue);
1430  }
1431 
1432  Value getDataPtr(Location loc, MemRefType type, Value memRefDesc,
1433  ArrayRef<Value> indices, ConversionPatternRewriter &rewriter,
1434  llvm::Module &module) const {
1435  LLVM::LLVMType ptrType = MemRefDescriptor(memRefDesc).getElementType();
1436  int64_t offset;
1437  SmallVector<int64_t, 4> strides;
1438  auto successStrides = getStridesAndOffset(type, strides, offset);
1439  assert(succeeded(successStrides) && "unexpected non-strided memref");
1440  (void)successStrides;
1441  return getStridedElementPtr(loc, ptrType, memRefDesc, indices, strides,
1442  offset, rewriter);
1443  }

1898 /// 2. Updates to the descriptor to introduce the data ptr, offset, size
1899 /// and stride.

1901 struct ViewOpLowering : public LLVMLegalizationPattern<ViewOp> {
1902  using LLVMLegalizationPattern<ViewOp>::LLVMLegalizationPattern;
1903 
1904  // Build and return the value for the idx^th shape dimension, either by
1905  // returning the constant shape dimension or counting the proper dynamic size.
1906  Value getSize(ConversionPatternRewriter &rewriter, Location loc,
1907  ArrayRef<int64_t> shape, ArrayRef<Value> dynamicSizes,
1908  unsigned idx) const {
1909  assert(idx < shape.size());
1910  if (!ShapedType::isDynamic(shape[idx]))
1911  return createIndexConstant(rewriter, loc, shape[idx]);
1912  // Count the number of dynamic dims in range [0, idx]
1913  unsigned nDynamic = llvm::count_if(shape.take_front(idx), [](int64_t v) {
1914  return ShapedType::isDynamic(v);
1915  });
1916  return dynamicSizes[nDynamic];
1917  }
1918 
1919  // Build and return the idx^th stride, either by returning the constant stride
1920  // or by computing the dynamic stride from the current `runningStride` and
1921  // `nextSize`. The caller should keep a running stride and update it with the
1922  // result returned by this function.
1923  Value getStride(ConversionPatternRewriter &rewriter, Location loc,
1924  ArrayRef<int64_t> strides, Value nextSize,
1925  Value runningStride, unsigned idx) const {
1926  assert(idx < strides.size());
1927  if (strides[idx] != MemRefType::getDynamicStrideOrOffset())
1928  return createIndexConstant(rewriter, loc, strides[idx]);
1929  if (nextSize)
1930  return runningStride
1931  ? rewriter.create<LLVM::MulOp>(loc, runningStride, nextSize)
1932  : nextSize;
1933  assert(!runningStride);
1934  return createIndexConstant(rewriter, loc, 1);

2020 static void ensureDistinctSuccessors(Block &bb) {
2021  auto *terminator = bb.getTerminator();
2022 
2023  // Find repeated successors with arguments.
2024  llvm::SmallDenseMap<Block *, SmallVector<int, 4>> successorPositions;
2025  for (int i = 0, e = terminator->getNumSuccessors(); i < e; ++i) {
2026  Block *successor = terminator->getSuccessor(i);
2027  // Blocks with no arguments are safe even if they appear multiple times
2028  // because they don't need PHI nodes.
2029  if (successor->getNumArguments() == 0)
2030  continue;
2031  successorPositions[successor].push_back(i);
2032  }
2033 
2034  // If a successor appears for the second or more time in the terminator,
2035  // create a new dummy block that unconditionally branches to the original
2036  // destination, and retarget the terminator to branch to this new block.
2037  // There is no need to pass arguments to the dummy block because it will be
2038  // dominated by the original block and can therefore use any values defined in
2039  // the original block.
2040  for (const auto &successor : successorPositions) {
2041  const auto &positions = successor.second;
2042  // Start from the second occurrence of a block in the successor list.
2043  for (auto position = std::next(positions.begin()), end = positions.end();
2044  position != end; ++position) {
2045  auto *dummyBlock = new Block();
2046  bb.getParent()->push_back(dummyBlock);
2047  auto builder = OpBuilder(dummyBlock);
2048  SmallVector<Value, 8> operands(
2049  terminator->getSuccessorOperands(*position));
2050  builder.create<BranchOp>(terminator->getLoc(), successor.first, operands);
2051  terminator->setSuccessor(dummyBlock, *position);
2052  for (int i = 0, e = terminator->getNumSuccessorOperands(*position); i < e;
2053  ++i)
2054  terminator->eraseSuccessorOperand(*position, i);
2055  }
2056  }
