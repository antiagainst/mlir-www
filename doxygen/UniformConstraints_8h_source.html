25 /// Factory methods for adding CAG constraints of various kinds suitable
26 /// for solving for uniform quantization.
27 class UniformConstraintsBuilder {
28 public:
29  UniformConstraintsBuilder(CAGSlice &slice) : slice(slice) {}
30 
31  /// Adds a coupling constraint between two nodes, effectively treating
32  /// them as a hard identity relationship.
33  void coupleAnchors(CAGAnchorNode *a, CAGAnchorNode *b);
34 
35  /// Applies statistics constraints to the given anchor, such that the solver
36  /// ensures that the statistics are representable by chosen types.
37  void applyStats(CAGAnchorNode *a, TensorAxisStatistics stats);
38 
39  /// Applies a constraint to a node which allows solutions that do not extend
40  /// beyond given min/max bounds (this is a hint that the tensor will not
41  /// take values outside of these bounds). If either minValue or maxValue is
42  /// NAN, then that side is considered open.
43  void clamp(CAGAnchorNode *a, APFloat minValue, APFloat maxValue);
44 
45  /// Propagates an explicit scale from an anchor that may have a uniform
46  /// |selectedType| to the |explicitScaleZeroPoint| field of the to node.
47  /// This is typically used with a to node that has a candidate quantized
48  /// type of |UniformExplicitFixedPointScale|, indicating that it can be
49  /// an arbitrary (signed) type that is expected to share the same scale
50  /// as the originating node.
51  void propagateExplicitScale(CAGAnchorNode *from, CAGAnchorNode *to);
52 
53 private:
54  CAGSlice &slice;
55 };
