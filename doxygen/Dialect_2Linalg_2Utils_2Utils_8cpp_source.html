33 Optional<RegionMatcher::BinaryOpKind>
34 RegionMatcher::matchAsScalarBinaryOp(GenericOp op) {
35  auto &region = op.region();
36  if (!has_single_element(region))
37  return llvm::None;
38 
39  Block &block = region.front();
40  if (block.getNumArguments() != 2 ||
41  !block.getArgument(0).getType().isIntOrFloat() ||
42  !block.getArgument(1).getType().isIntOrFloat())
43  return llvm::None;
44 
45  auto &ops = block.getOperations();
46  if (!has_single_element(block.without_terminator()))
47  return llvm::None;
48 
49  using mlir::matchers::m_Val;
50  auto a = m_Val(block.getArgument(0));
51  auto b = m_Val(block.getArgument(1));
52 
53  auto addPattern = m_Op<linalg::YieldOp>(m_Op<AddIOp>(a, b));
54  if (addPattern.match(&ops.back()))
55  return BinaryOpKind::IAdd;
56 
57  return llvm::None;
58 }

71 SmallVector<Value, 4> mlir::linalg::applyMapToValues(OpBuilder &b, Location loc,
72  AffineMap map,
73  ArrayRef<Value> values,
74  OperationFolder *folder) {
75  SmallVector<Value, 4> res;
76  res.reserve(map.getNumResults());
77  unsigned numDims = map.getNumDims();
78  // For each `expr` in `map`, applies the `expr` to the values extracted from
79  // ranges. If the resulting application can be folded into a Value, the
80  // folding occurs eagerly. Otherwise, an affine.apply operation is emitted.
81  for (auto expr : map.getResults()) {
82  AffineMap map = AffineMap::get(numDims, 0, expr);
83  res.push_back(emitOrFoldComposedAffineApply(b, loc, map, values, folder));
84  }
85  return res;
86 }

88 /// Returns all the operands of `linalgOp` that are not views.
89 /// Asserts that these operands are value types to allow transformations like
90 /// tiling to just use the values when cloning `linalgOp`.
91 SmallVector<Value, 4>
92 mlir::linalg::getAssumedNonViewOperands(LinalgOp linalgOp) {
93  auto *op = linalgOp.getOperation();
94  unsigned numViews = linalgOp.getNumInputsAndOutputs();
95  unsigned nOperands = op->getNumOperands() - numViews;
96  SmallVector<Value, 4> res;
97  res.reserve(nOperands);
98  for (unsigned i = 0; i < nOperands; ++i) {
99  res.push_back(op->getOperand(numViews + i));
100  auto t = res.back().getType();
101  (void)t;
102  assert((t.isIntOrIndexOrFloat() || t.isa<VectorType>()) &&
103  "expected scalar or vector type");
104  }
105  return res;
