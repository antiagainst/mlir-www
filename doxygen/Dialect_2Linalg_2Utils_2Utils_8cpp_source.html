45 SmallVector<Value, 4> mlir::linalg::applyMapToValues(OpBuilder &b, Location loc,
46  AffineMap map,
47  ArrayRef<Value> values,
48  OperationFolder *folder) {
49  SmallVector<Value, 4> res;
50  res.reserve(map.getNumResults());
51  unsigned numDims = map.getNumDims();
52  // For each `expr` in `map`, applies the `expr` to the values extracted from
53  // ranges. If the resulting application can be folded into a Value, the
54  // folding occurs eagerly. Otherwise, an affine.apply operation is emitted.
55  for (auto expr : map.getResults()) {
56  AffineMap map = AffineMap::get(numDims, 0, expr);
57  res.push_back(emitOrFoldComposedAffineApply(b, loc, map, values, folder));
58  }
59  return res;
60 }

62 /// Returns all the operands of `linalgOp` that are not views.
63 /// Asserts that these operands are value types to allow transformations like
64 /// tiling to just use the values when cloning `linalgOp`.
65 SmallVector<Value, 4>
66 mlir::linalg::getAssumedNonViewOperands(LinalgOp linalgOp) {
67  auto *op = linalgOp.getOperation();
68  unsigned numViews = linalgOp.getNumInputsAndOutputs();
69  unsigned nOperands = op->getNumOperands() - numViews;
70  SmallVector<Value, 4> res;
71  res.reserve(nOperands);
72  for (unsigned i = 0; i < nOperands; ++i) {
73  res.push_back(op->getOperand(numViews + i));
74  auto t = res.back().getType();
75  (void)t;
76  assert((t.isIntOrIndexOrFloat() || t.isa<VectorType>()) &&
77  "expected scalar or vector type");
78  }
79  return res;
