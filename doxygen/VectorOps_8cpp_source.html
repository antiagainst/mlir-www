94  parser.parseKeywordType("into", resultVectorType) ||
95  parser.resolveOperand(lhsInfo, types[0], result.operands) ||
96  parser.resolveOperand(rhsInfo, types[1], result.operands) ||
97  parser.resolveOperand(accInfo, resultVectorType, result.operands) ||
98  parser.addTypeToList(resultVectorType, result.types))
99  return failure();
100  result.attributes.assign(dictAttr.getValue().begin(),
101  dictAttr.getValue().end());
102  if (masksInfo.empty())
103  return success();

112  maskTypes.push_back(VectorType::get(rhsType.getShape(), maskElementType));
113  if (parser.resolveOperands(masksInfo, maskTypes, loc, result.operands))

150 static bool verifyOutputShape(
151  VectorType lhsType, VectorType rhsType, VectorType accType,
152  VectorType resType,
153  const std::vector<std::pair<int64_t, int64_t>> &contractingDimMap,
154  const std::vector<std::pair<int64_t, int64_t>> &batchDimMap) {
155  DenseSet<int64_t> lhsContractingDimSet;
156  DenseSet<int64_t> rhsContractingDimSet;
157  for (auto &dimPair : contractingDimMap) {
158  lhsContractingDimSet.insert(dimPair.first);
159  rhsContractingDimSet.insert(dimPair.second);
160  }
161  DenseSet<int64_t> rhsBatchDimSet;
162  for (auto &dimPair : batchDimMap)
163  rhsBatchDimSet.insert(dimPair.second);
164 
165  // Add free and batch dimensions from 'lhsType' to 'expectedResultDims'.
166  SmallVector<int64_t, 4> expectedResultDims;
167  for (int64_t i = 0, e = lhsType.getRank(); i < e; ++i) {
168  if (lhsContractingDimSet.count(i) > 0)
169  continue;
170  expectedResultDims.push_back(lhsType.getDimSize(i));
171  }
172 
173  // Add free dimensions from 'rhsType' to 'expectedResultDims'.
174  for (int64_t i = 0, e = rhsType.getRank(); i < e; ++i) {
175  if (rhsContractingDimSet.count(i) > 0 || rhsBatchDimSet.count(i) > 0)
176  continue;
177  expectedResultDims.push_back(rhsType.getDimSize(i));
178  }
179 
180  // Verify dimension from 'resType' against 'expectedResultDims'.
181  if (resType.getShape().size() != expectedResultDims.size() ||
182  accType.getShape().size() != expectedResultDims.size())
183  return false;
184  for (int64_t i = 0, e = resType.getRank(); i < e; ++i) {
185  if (resType.getDimSize(i) != expectedResultDims[i] ||
186  accType.getDimSize(i) != expectedResultDims[i])
187  return false;
188  }
189  return true;
190 }

192 static LogicalResult verify(ContractionOp op) {
193  auto lhsType = op.getLhsType();
194  auto rhsType = op.getRhsType();
195  auto accType = op.getAccType();
196  auto resType = op.getResultType();
197 
198  // Verify that an indexing map was specified for each vector operand.
199  if (op.indexing_maps().size() != 3)
200  return op.emitOpError("expected an indexing map for each vector operand");
201 
202  // Verify that each index map has 'numIterators' inputs, no symbols, and
203  // that the number of map outputs equals the rank of its associated
204  // vector operand.
205  unsigned numIterators = op.iterator_types().getValue().size();
206  for (auto it : llvm::enumerate(op.indexing_maps())) {
207  auto index = it.index();
208  auto map = it.value().cast<AffineMapAttr>().getValue();
209  if (map.getNumSymbols() != 0)
210  return op.emitOpError("expected indexing map ")
211  << index << " to have no symbols";
212  if (map.getNumDims() != numIterators)
213  return op.emitOpError("expected indexing map ")
214  << index << " to have " << numIterators << " number of inputs";
215  auto operandType = op.getOperand(index).getType().cast<VectorType>();
216  unsigned rank = operandType.getShape().size();
217  if (map.getNumResults() != rank)
218  return op.emitOpError("expected indexing map ")
219  << index << " to have " << rank << " number of outputs";
220  if (!map.isProjectedPermutation())
221  return op.emitOpError("expected indexing map ")
222  << index << " to be a projected permutation of its inputs";
223  }
224 
225  auto contractingDimMap = op.getContractingDimMap();
226  auto batchDimMap = op.getBatchDimMap();
227 
228  // Verify at least one contracting dimension pair was specified.
229  if (contractingDimMap.empty())
230  return op.emitOpError("expected at least one contracting dimension pair");
231 
232  // Verify contracting dimension map was properly constructed.
233  if (!verifyDimMap(lhsType, rhsType, contractingDimMap))
234  return op.emitOpError("invalid contracting dimension map");
235 
236  // Verify batch dimension map was properly constructed.
237  if (!verifyDimMap(lhsType, rhsType, batchDimMap))
238  return op.emitOpError("invalid batch dimension map");
239 
240  // Verify 'accType' and 'resType' shape.
241  if (!verifyOutputShape(lhsType, rhsType, accType, resType, contractingDimMap,
242  batchDimMap))
243  return op.emitOpError("invalid accumulator/result vector shape");
244 
245  // Verify that either two vector masks are set or none are set.
246  auto lhsMaskType = op.getLHSVectorMaskType();
247  auto rhsMaskType = op.getRHSVectorMaskType();
248  if ((lhsMaskType && !rhsMaskType) || (!lhsMaskType && rhsMaskType))
249  return op.emitOpError("invalid number of vector masks specified");
250  if (lhsMaskType && rhsMaskType) {
251  // Verify mask rank == argument rank.
252  if (lhsMaskType.getShape().size() != lhsType.getShape().size() ||
253  rhsMaskType.getShape().size() != rhsType.getShape().size())
254  return op.emitOpError("invalid vector mask rank");
255  }
256  return success();
257 }

274 getDimMap(ArrayRef<AffineMap> indexingMaps, ArrayAttr iteratorTypes,
275  StringRef targetIteratorTypeName, MLIRContext *context) {
276  std::vector<std::pair<int64_t, int64_t>> dimMap;
277  for (auto it : llvm::enumerate(iteratorTypes)) {

285  if (lhsDim >= 0 && rhsDim >= 0)
286  dimMap.push_back({lhsDim, rhsDim});
287  }

292  SmallVectorImpl<int64_t> &iterationBounds) {
293  auto lhsShape = getLhsType().getShape();
294  auto resShape = getResultType().getShape();
295  SmallVector<AffineMap, 4> indexingMaps(getIndexingMaps());
296  SmallVector<int64_t, 2> iterationShape;
297  for (auto it : llvm::enumerate(iterator_types())) {
298  // Search lhs/rhs map results for 'targetExpr'.
299  auto targetExpr = getAffineDimExpr(it.index(), getContext());
300  auto iteratorTypeName = it.value().cast<StringAttr>().getValue();
301  if (iteratorTypeName == getReductionIteratorTypeName()) {
302  // Get reduction dim size from lhs shape (same size in rhsShape).
303  int64_t lhsDimIndex = getResultIndex(indexingMaps[0], targetExpr);
304  assert(lhsDimIndex >= 0);
305  iterationBounds.push_back(lhsShape[lhsDimIndex]);
306  continue;
307  }
308  // Get parallel dimension size from result shape.
309  int64_t resDimIndex = getResultIndex(indexingMaps[2], targetExpr);
310  assert(resDimIndex >= 0);
311  iterationBounds.push_back(resShape[resDimIndex]);
312  }

477 static LogicalResult
478 isValidExtractOrInsertSlicesType(Operation *op, VectorType vectorType,
479  TupleType tupleType, ArrayRef<int64_t> sizes,
480  ArrayRef<int64_t> strides) {
481  // Check for non-unit strides.
482  // TODO(b/144845578) Support non-1 strides.
483  if (llvm::any_of(strides, [](int64_t s) { return s != 1; }))
484  return op->emitError("requires unit strides");
485  // Check that 'vectorType' rank matches rank of tuple element vectors.
486  unsigned rank = vectorType.getRank();
487  auto is_vector_type_of_rank = [&](Type t) {
488  return t.isa<VectorType>() && t.cast<VectorType>().getRank() == rank;
489  };
490  if (!llvm::all_of(tupleType.getTypes(), is_vector_type_of_rank))
491  return op->emitError("requires vector tuple elements of rank ") << rank;
492  // Check that 'sizes' and 'strides' are of size == 'rank'.
493  if (sizes.size() != rank || strides.size() != rank)
494  return op->emitError("requires sizes and strides of rank ") << rank;
495 
496  // Generate each slice shape based on 'sizes', 'strides' and 'vectorType',
497  // and verify that the same matches the corresponding tuple element 'i'.
498  auto shape = vectorType.getShape();
499  auto sliceStrides = computeStrides(shape, sizes);
500  for (int64_t i = 0, e = tupleType.size(); i < e; ++i) {
501  auto vectorOffsets = delinearize(sliceStrides, i);
502  auto elementOffsets =
503  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
504  auto sliceSizes = computeSliceSizes(shape, sizes, elementOffsets);
505  // Create slice VectorType type.
506  auto sliceVectorType =
507  VectorType::get(sliceSizes, vectorType.getElementType());
508  // Verify that 'sliceVectorType' matches tupleType.getTypes(i)
509  if (sliceVectorType != tupleType.getType(i))
510  return op->emitError("invalid tuple element type ") << sliceVectorType;
511  }
512  return success();
513 }

543 static LogicalResult verify(BroadcastOp op) {
544  VectorType srcVectorType = op.getSourceType().dyn_cast<VectorType>();
545  VectorType dstVectorType = op.getVectorType();
546  // Scalar to vector broadcast is always valid. A vector
547  // to vector broadcast needs some additional checking.
548  if (srcVectorType) {
549  int64_t srcRank = srcVectorType.getRank();
550  int64_t dstRank = dstVectorType.getRank();
551  if (srcRank > dstRank)
552  return op.emitOpError("source rank higher than destination rank");
553  // Source has an exact match or singleton value for all trailing dimensions
554  // (all leading dimensions are simply duplicated).
555  int64_t lead = dstRank - srcRank;
556  for (int64_t r = 0; r < srcRank; ++r) {
557  int64_t srcDim = srcVectorType.getDimSize(r);
558  int64_t dstDim = dstVectorType.getDimSize(lead + r);
