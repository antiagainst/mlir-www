147 /// Base class for the conversion patterns that require type changes. Specific
148 /// conversions must derive this class and implement least one `rewrite` method.
149 /// NOTE: These conversion patterns can only be used with the 'apply*' methods
150 /// below.
151 class ConversionPattern : public RewritePattern {
152 public:
153  /// Construct an ConversionPattern. `rootName` must correspond to the
154  /// canonical name of the first operation matched by the pattern.
155  ConversionPattern(StringRef rootName, PatternBenefit benefit,
156  MLIRContext *ctx)
157  : RewritePattern(rootName, benefit, ctx) {}
158 
159  /// Hook for derived classes to implement rewriting. `op` is the (first)
160  /// operation matched by the pattern, `operands` is a list of rewritten values
161  /// that are passed to this operation, `rewriter` can be used to emit the new
162  /// operations. This function must be reimplemented if the
163  /// ConversionPattern ever needs to replace an operation that does not
164  /// have successors. This function should not fail. If some specific cases of
165  /// the operation are not supported, these cases should not be matched.
166  virtual void rewrite(Operation *op, ArrayRef<Value> operands,
167  ConversionPatternRewriter &rewriter) const {
168  llvm_unreachable("unimplemented rewrite");
169  }
170 
171  /// Hook for derived classes to implement rewriting. `op` is the (first)
172  /// operation matched by the pattern, `properOperands` is a list of rewritten
173  /// values that are passed to the operation itself, `destinations` is a list
174  /// of (potentially rewritten) successor blocks, `operands` is a list of lists
175  /// of rewritten values passed to each of the successors, co-indexed with
176  /// `destinations`, `rewriter` can be used to emit the new operations. It must
177  /// be reimplemented if the ConversionPattern ever needs to replace a
178  /// terminator operation that has successors. This function should not fail
179  /// the pass. If some specific cases of the operation are not supported,
180  /// these cases should not be matched.
181  virtual void rewrite(Operation *op, ArrayRef<Value> properOperands,
182  ArrayRef<Block *> destinations,
183  ArrayRef<ArrayRef<Value>> operands,
184  ConversionPatternRewriter &rewriter) const {
185  llvm_unreachable("unimplemented rewrite for terminators");
186  }
187 
188  /// Hook for derived classes to implement combined matching and rewriting.
189  virtual PatternMatchResult
190  matchAndRewrite(Operation *op, ArrayRef<Value> properOperands,
191  ArrayRef<Block *> destinations,
192  ArrayRef<ArrayRef<Value>> operands,
193  ConversionPatternRewriter &rewriter) const {
194  if (!match(op))
195  return matchFailure();
196  rewrite(op, properOperands, destinations, operands, rewriter);
197  return matchSuccess();
198  }
199 
200  /// Hook for derived classes to implement combined matching and rewriting.
201  virtual PatternMatchResult
202  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
203  ConversionPatternRewriter &rewriter) const {
204  if (!match(op))
205  return matchFailure();
206  rewrite(op, operands, rewriter);
207  return matchSuccess();
208  }
209 
210  /// Attempt to match and rewrite the IR root at the specified operation.
211  PatternMatchResult matchAndRewrite(Operation *op,
212  PatternRewriter &rewriter) const final;
213 
214 private:
215  using RewritePattern::rewrite;
216 };

390 /// This class describes a specific conversion target.
391 class ConversionTarget {
392 public:
393  /// This enumeration corresponds to the specific action to take when
394  /// considering an operation legal for this conversion target.
395  enum class LegalizationAction {
396  /// The target supports this operation.
397  Legal,
398 
399  /// This operation has dynamic legalization constraints that must be checked
400  /// by the target.
401  Dynamic,
402 
403  /// The target explicitly does not support this operation.
404  Illegal,
405  };
406 
407  /// A structure containing additional information describing a specific legal
408  /// operation instance.
409  struct LegalOpDetails {
410  /// A flag that indicates if this operation is 'recursively' legal. This
411  /// means that if an operation is legal, either statically or dynamically,
412  /// all of the operations nested within are also considered legal.
413  bool isRecursivelyLegal = false;
414  };

416  /// The signature of the callback used to determine if an operation is
417  /// dynamically legal on the target.
418  using DynamicLegalityCallbackFn = std::function<bool(Operation *)>;
