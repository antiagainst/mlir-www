44 // Returns a result string which represents the direction vector (if there was

46 static std::string
47 getDirectionVectorStr(bool ret, unsigned numCommonLoops, unsigned loopNestDepth,
48  ArrayRef<DependenceComponent> dependenceComponents) {
49  if (!ret)
50  return "false";
51  if (dependenceComponents.empty() || loopNestDepth > numCommonLoops)
52  return "true";
53  std::string result;

56  if (dependenceComponents[i].lb.hasValue() &&

65  ubStr = std::to_string(dependenceComponents[i].ub.getValue());

72 // For each access in 'loadsAndStores', runs a dependence check between this
73 // "source" access and all subsequent "destination" accesses in
74 // 'loadsAndStores'. Emits the result of the dependence check as a note with
75 // the source access.
76 static void checkDependences(ArrayRef<Operation *> loadsAndStores) {
77  for (unsigned i = 0, e = loadsAndStores.size(); i < e; ++i) {
78  auto *srcOpInst = loadsAndStores[i];
79  MemRefAccess srcAccess(srcOpInst);
80  for (unsigned j = 0; j < e; ++j) {
81  auto *dstOpInst = loadsAndStores[j];
82  MemRefAccess dstAccess(dstOpInst);
83 
84  unsigned numCommonLoops =
85  getNumCommonSurroundingLoops(*srcOpInst, *dstOpInst);
86  for (unsigned d = 1; d <= numCommonLoops + 1; ++d) {
87  FlatAffineConstraints dependenceConstraints;
88  SmallVector<DependenceComponent, 2> dependenceComponents;
89  DependenceResult result = checkMemrefAccessDependence(
90  srcAccess, dstAccess, d, &dependenceConstraints,
91  &dependenceComponents);
92  assert(result.value != DependenceResult::Failure);
93  bool ret = hasDependence(result);
94  // TODO(andydavis) Print dependence type (i.e. RAW, etc) and print
95  // distance vectors as: ([2, 3], [0, 10]). Also, shorten distance
96  // vectors from ([1, 1], [3, 3]) to (1, 3).
97  srcOpInst->emitRemark("dependence from ")
98  << i << " to " << j << " at depth " << d << " = "
99  << getDirectionVectorStr(ret, numCommonLoops, d,
100  dependenceComponents);
101  }
102  }
103  }
104 }

106 // Walks the Function 'f' adding load and store ops to 'loadsAndStores'.
107 // Runs pair-wise dependence checks.
108 void TestMemRefDependenceCheck::runOnFunction() {
109  // Collect the loads and stores within the function.
110  loadsAndStores.clear();
111  getFunction().walk([&](Operation *op) {
112  if (isa<AffineLoadOp>(op) || isa<AffineStoreOp>(op))
113  loadsAndStores.push_back(op);
114  });
115 
116  checkDependences(loadsAndStores);
117 }
