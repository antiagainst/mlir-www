151 /// Get the number of bits require to store a value of the given shaped type.
152 /// Compute the value recursively since tensors are allowed to have vectors as
153 /// elements.
154 int64_t ShapedType::getSizeInBits() const {
155  assert(hasStaticShape() &&
156  "cannot get the bit size of an aggregate with a dynamic shape");
157 
158  auto elementType = getElementType();
159  if (elementType.isIntOrFloat())

325 /// Get or create a new MemRefType defined by the arguments. If the resulting
326 /// type would be ill-formed, return nullptr. If the location is provided,
327 /// emit detailed error messages. To emit errors when the location is unknown,
328 /// pass in an instance of UnknownLoc.
329 MemRefType MemRefType::getImpl(ArrayRef<int64_t> shape, Type elementType,
330  ArrayRef<AffineMap> affineMapComposition,
331  unsigned memorySpace,
332  Optional<Location> location) {
333  auto *context = elementType.getContext();
334 
335  // Check that memref is formed from allowed types.
336  if (!elementType.isIntOrFloat() && !elementType.isa<VectorType>())
337  return emitOptionalError(location, "invalid memref element type"),
338  MemRefType();
339 
340  for (int64_t s : shape) {
341  // Negative sizes are not allowed except for `-1` that means dynamic size.
342  if (s < -1)
343  return emitOptionalError(location, "invalid memref size"), MemRefType();
344  }
345 
346  // Check that the structure of the composition is valid, i.e. that each
347  // subsequent affine map has as many inputs as the previous map has results.
348  // Take the dimensionality of the MemRef for the first map.
349  auto dim = shape.size();
350  unsigned i = 0;
351  for (const auto &affineMap : affineMapComposition) {
352  if (affineMap.getNumDims() != dim) {
353  if (location)
354  emitError(*location)
355  << "memref affine map dimension mismatch between "
356  << (i == 0 ? Twine("memref rank") : "affine map " + Twine(i))
357  << " and affine map" << i + 1 << ": " << dim
358  << " != " << affineMap.getNumDims();
359  return nullptr;
360  }
361 
362  dim = affineMap.getNumResults();
363  ++i;
364  }
365 
366  // Drop identity maps from the composition.
367  // This may lead to the composition becoming empty, which is interpreted as an
368  // implicit identity.
369  SmallVector<AffineMap, 2> cleanedAffineMapComposition;
370  for (const auto &map : affineMapComposition) {
371  if (map.isIdentity())
372  continue;
373  cleanedAffineMapComposition.push_back(map);
374  }
375 
376  return Base::get(context, StandardTypes::MemRef, shape, elementType,
377  cleanedAffineMapComposition, memorySpace);
378 }

419 /// Given MemRef `sizes` that are either static or dynamic, returns the
420 /// canonical "contiguous" strides AffineExpr. Strides are multiplicative and
421 /// once a dynamic dimension is encountered, all canonical strides become
422 /// dynamic and need to be encoded with a different symbol.
423 /// For canonical strides expressions, the offset is always 0 and and fastest
424 /// varying stride is always `1`.
425 ///
426 /// Examples:
427 /// - memref<3x4x5xf32> has canonical stride expression `20*d0 + 5*d1 + d2`.
428 /// - memref<3x?x5xf32> has canonical stride expression `s0*d0 + 5*d1 + d2`.
429 /// - memref<3x4x?xf32> has canonical stride expression `s1*d0 + s0*d1 + d2`.
430 static AffineExpr makeCanonicalStridedLayoutExpr(ArrayRef<int64_t> sizes,
431  MLIRContext *context) {
432  AffineExpr expr;
433  bool dynamicPoisonBit = false;
434  unsigned nSymbols = 0;
435  int64_t runningSize = 1;
436  unsigned rank = sizes.size();
437  for (auto en : llvm::enumerate(llvm::reverse(sizes))) {
438  auto size = en.value();
439  auto position = rank - 1 - en.index();
440  // Degenerate case, no size =-> no stride
441  if (size == 0)
442  continue;
443  auto d = getAffineDimExpr(position, context);
444  // Static case: stride = runningSize and runningSize *= size.
445  if (!dynamicPoisonBit) {
446  auto cst = getAffineConstantExpr(runningSize, context);
447  expr = expr ? expr + cst * d : cst * d;
448  if (size > 0)
449  runningSize *= size;
450  else
451  // From now on bail into dynamic mode.
452  dynamicPoisonBit = true;
453  continue;
454  }
455  // Dynamic case, new symbol for each new stride.
456  auto sym = getAffineSymbolExpr(nSymbols++, context);
457  expr = expr ? expr + d * sym : d * sym;
458  }
459  return simplifyAffineExpr(expr, rank, nSymbols);
460 }

462 // Fallback cases for terminal dim/sym/cst that are not part of a binary op (
463 // i.e. single term). Accumulate the AffineExpr into the existing one.
464 static void extractStridesFromTerm(AffineExpr e,
465  AffineExpr multiplicativeFactor,
466  MutableArrayRef<AffineExpr> strides,
467  AffineExpr &offset) {
468  if (auto dim = e.dyn_cast<AffineDimExpr>())
469  strides[dim.getPosition()] =
470  strides[dim.getPosition()] + multiplicativeFactor;
471  else
472  offset = offset + e * multiplicativeFactor;
473 }
474 
475 /// Takes a single AffineExpr `e` and populates the `strides` array with the
476 /// strides expressions for each dim position.
477 /// The convention is that the strides for dimensions d0, .. dn appear in
478 /// order to make indexing intuitive into the result.
479 static LogicalResult extractStrides(AffineExpr e,
480  AffineExpr multiplicativeFactor,
481  MutableArrayRef<AffineExpr> strides,
482  AffineExpr &offset) {
483  auto bin = e.dyn_cast<AffineBinaryOpExpr>();
484  if (!bin) {
485  extractStridesFromTerm(e, multiplicativeFactor, strides, offset);
486  return success();
487  }
488 
489  if (bin.getKind() == AffineExprKind::CeilDiv ||
490  bin.getKind() == AffineExprKind::FloorDiv ||
491  bin.getKind() == AffineExprKind::Mod)
492  return failure();
493 
494  if (bin.getKind() == AffineExprKind::Mul) {
495  auto dim = bin.getLHS().dyn_cast<AffineDimExpr>();
496  if (dim) {
497  strides[dim.getPosition()] =
498  strides[dim.getPosition()] + bin.getRHS() * multiplicativeFactor;
499  return success();
500  }
501  // LHS and RHS may both contain complex expressions of dims. Try one path
502  // and if it fails try the other. This is guaranteed to succeed because
503  // only one path may have a `dim`, otherwise this is not an AffineExpr in
504  // the first place.
505  if (bin.getLHS().isSymbolicOrConstant())
506  return extractStrides(bin.getRHS(), multiplicativeFactor * bin.getLHS(),
507  strides, offset);
508  return extractStrides(bin.getLHS(), multiplicativeFactor * bin.getRHS(),
509  strides, offset);
510  }
511 
512  if (bin.getKind() == AffineExprKind::Add) {
513  auto res1 =
514  extractStrides(bin.getLHS(), multiplicativeFactor, strides, offset);
515  auto res2 =
516  extractStrides(bin.getRHS(), multiplicativeFactor, strides, offset);
517  return success(succeeded(res1) && succeeded(res2));
518  }
519 
520  llvm_unreachable("unexpected binary operation");
521 }

523 LogicalResult mlir::getStridesAndOffset(MemRefType t,
524  SmallVectorImpl<AffineExpr> &strides,
525  AffineExpr &offset) {
526  auto affineMaps = t.getAffineMaps();
527  // For now strides are only computed on a single affine map with a single
528  // result (i.e. the closed subset of linearization maps that are compatible
529  // with striding semantics).
530  // TODO(ntv): support more forms on a per-need basis.
531  if (affineMaps.size() > 1)
532  return failure();
533  if (affineMaps.size() == 1 && affineMaps[0].getNumResults() != 1)
534  return failure();
535 
536  auto zero = getAffineConstantExpr(0, t.getContext());
537  auto one = getAffineConstantExpr(1, t.getContext());
538  offset = zero;
539  strides.assign(t.getRank(), zero);
540 
541  AffineMap m;
542  if (!affineMaps.empty()) {
543  m = affineMaps.front();
544  assert(!m.isIdentity() && "unexpected identity map");
545  }
546 
547  // Canonical case for empty map.
548  if (!m) {
549  // 0-D corner case, offset is already 0.
550  if (t.getRank() == 0)
551  return success();
552  auto stridedExpr =
553  makeCanonicalStridedLayoutExpr(t.getShape(), t.getContext());
554  if (succeeded(extractStrides(stridedExpr, one, strides, offset)))
555  return success();
556  assert(false && "unexpected failure: extract strides in canonical layout");
557  }
558 
559  // Non-canonical case requires more work.
560  auto stridedExpr =
561  simplifyAffineExpr(m.getResult(0), m.getNumDims(), m.getNumSymbols());
562  if (failed(extractStrides(stridedExpr, one, strides, offset))) {
563  offset = AffineExpr();
564  strides.clear();
565  return failure();
566  }
567 
568  // Simplify results to allow folding to constants and simple checks.
569  unsigned numDims = m.getNumDims();
570  unsigned numSymbols = m.getNumSymbols();
571  offset = simplifyAffineExpr(offset, numDims, numSymbols);
572  for (auto &stride : strides)
573  stride = simplifyAffineExpr(stride, numDims, numSymbols);
574 
575  /// In practice, a strided memref must be internally non-aliasing. Test
576  /// against 0 as a proxy.
577  /// TODO(ntv) static cases can have more advanced checks.
578  /// TODO(ntv) dynamic cases would require a way to compare symbolic
579  /// expressions and would probably need an affine set context propagated
580  /// everywhere.
581  if (llvm::any_of(strides, [](AffineExpr e) {
582  return e == getAffineConstantExpr(0, e.getContext());
583  })) {
584  offset = AffineExpr();
585  strides.clear();
586  return failure();
587  }
588 
589  return success();
590 }

592 LogicalResult mlir::getStridesAndOffset(MemRefType t,
593  SmallVectorImpl<int64_t> &strides,
594  int64_t &offset) {
595  AffineExpr offsetExpr;
596  SmallVector<AffineExpr, 4> strideExprs;
597  if (failed(::getStridesAndOffset(t, strideExprs, offsetExpr)))
598  return failure();
599  if (auto cst = offsetExpr.dyn_cast<AffineConstantExpr>())
600  offset = cst.getValue();
601  else
602  offset = ShapedType::kDynamicStrideOrOffset;
603  for (auto e : strideExprs) {
604  if (auto c = e.dyn_cast<AffineConstantExpr>())
605  strides.push_back(c.getValue());
606  else
607  strides.push_back(ShapedType::kDynamicStrideOrOffset);
608  }
609  return success();
610 }

666 AffineMap mlir::makeStridedLinearLayoutMap(ArrayRef<int64_t> strides,
667  int64_t offset,
668  MLIRContext *context) {
669  AffineExpr expr;
670  unsigned nSymbols = 0;
671 
672  // AffineExpr for offset.
673  // Static case.
674  if (offset != MemRefType::getDynamicStrideOrOffset()) {
675  auto cst = getAffineConstantExpr(offset, context);
676  expr = cst;
677  } else {
678  // Dynamic case, new symbol for the offset.
679  auto sym = getAffineSymbolExpr(nSymbols++, context);
680  expr = sym;
681  }
682 
683  // AffineExpr for strides.
684  for (auto en : llvm::enumerate(strides)) {
685  auto dim = en.index();
686  auto stride = en.value();
687  assert(stride != 0 && "Invalid stride specification");
688  auto d = getAffineDimExpr(dim, context);
689  AffineExpr mult;
690  // Static case.
691  if (stride != MemRefType::getDynamicStrideOrOffset())
692  mult = getAffineConstantExpr(stride, context);
693  else
694  // Dynamic case, new symbol for each new stride.
695  mult = getAffineSymbolExpr(nSymbols++, context);
696  expr = expr + d * mult;
697  }
698 
699  return AffineMap::get(strides.size(), nSymbols, expr);
700 }

702 /// Return a version of `t` with identity layout if it can be determined
703 /// statically that the layout is the canonical contiguous strided layout.
704 /// Otherwise pass `t`'s layout into `simplifyAffineMap` and return a copy of
705 /// `t` with simplified layout.
706 /// If `t` has multiple layout maps or a multi-result layout, just return `t`.
707 MemRefType mlir::canonicalizeStridedLayout(MemRefType t) {
708  auto affineMaps = t.getAffineMaps();
709  // Already in canonical form.
710  if (affineMaps.empty())
711  return t;
712 
713  // Can't reduce to canonical identity form, return in canonical form.
714  if (affineMaps.size() > 1 || affineMaps[0].getNumResults() > 1)
715  return t;
716 
717  // If the canonical strided layout for the sizes of `t` is equal to the
718  // simplified layout of `t` we can just return an empty layout. Otherwise,
719  // just simplify the existing layout.
720  AffineExpr expr =
721  makeCanonicalStridedLayoutExpr(t.getShape(), t.getContext());
722  auto m = affineMaps[0];
723  auto simplifiedLayoutExpr =
724  simplifyAffineExpr(m.getResult(0), m.getNumDims(), m.getNumSymbols());
725  if (expr != simplifiedLayoutExpr)
726  return MemRefType::get(t.getShape(), t.getElementType(),
727  {AffineMap::get(m.getNumDims(), m.getNumSymbols(),
728  {simplifiedLayoutExpr})});
729 
730  return MemRefType::get(t.getShape(), t.getElementType(), {});
731 }
