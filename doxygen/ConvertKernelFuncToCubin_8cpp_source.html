43 /// A pass converting tagged kernel modules to cubin blobs.
44 ///
45 /// If tagged as a kernel module, each contained function is translated to NVVM
46 /// IR and further to PTX. A user provided CubinGenerator compiles the PTX to
47 /// GPU binary code, which is then attached as an attribute to the function. The
48 /// function body is erased.
49 class GpuKernelToCubinPass
50  : public OperationPass<GpuKernelToCubinPass, gpu::GPUModuleOp> {
51 public:
52  GpuKernelToCubinPass(
53  CubinGenerator cubinGenerator = compilePtxToCubinForTesting)
54  : cubinGenerator(cubinGenerator) {}
55 
56  void runOnOperation() override {
57  gpu::GPUModuleOp module = getOperation();
58 
59  // Make sure the NVPTX target is initialized.
60  LLVMInitializeNVPTXTarget();
61  LLVMInitializeNVPTXTargetInfo();
62  LLVMInitializeNVPTXTargetMC();
63  LLVMInitializeNVPTXAsmPrinter();
64 
65  auto llvmModule = translateModuleToNVVMIR(module);
66  if (!llvmModule)
67  return signalPassFailure();
68 
69  // Translate the module to CUBIN and attach the result as attribute to the
70  // module.
71  if (auto cubinAttr = translateGPUModuleToCubinAnnotation(
72  *llvmModule, module.getLoc(), module.getName()))
73  module.setAttr(kCubinAnnotation, cubinAttr);
74  else
75  signalPassFailure();
76  }
77 
78 private:
79  static OwnedCubin compilePtxToCubinForTesting(const std::string &ptx,
80  Location, StringRef);
81 
82  std::string translateModuleToPtx(llvm::Module &module,
83  llvm::TargetMachine &target_machine);
84 
85  /// Converts llvmModule to cubin using the user-provided generator. Location
86  /// is used for error reporting and name is forwarded to the CUBIN generator
87  /// to use in its logging mechanisms.
88  OwnedCubin convertModuleToCubin(llvm::Module &llvmModule, Location loc,
89  StringRef name);
90 
91  /// Translates llvmModule to cubin and returns the result as attribute.
92  StringAttr translateGPUModuleToCubinAnnotation(llvm::Module &llvmModule,
93  Location loc, StringRef name);
94 
95  CubinGenerator cubinGenerator;
96 };

122 OwnedCubin GpuKernelToCubinPass::convertModuleToCubin(llvm::Module &llvmModule,
123  Location loc,
124  StringRef name) {
125  std::unique_ptr<llvm::TargetMachine> targetMachine;
126  {
127  std::string error;
128  // TODO(herhut): Make triple configurable.
129  constexpr const char *cudaTriple = "nvptx64-nvidia-cuda";
130  llvm::Triple triple(cudaTriple);
131  const llvm::Target *target =
132  llvm::TargetRegistry::lookupTarget("", triple, error);
133  if (target == nullptr) {
134  emitError(loc, "cannot initialize target triple");
135  return {};
136  }
137  targetMachine.reset(
138  target->createTargetMachine(triple.str(), "sm_35", "+ptx60", {}, {}));
139  }
140 
141  // Set the data layout of the llvm module to match what the ptx target needs.
142  llvmModule.setDataLayout(targetMachine->createDataLayout());
143 
144  auto ptx = translateModuleToPtx(llvmModule, *targetMachine);
145 
146  return cubinGenerator(ptx, loc, name);
147 }
