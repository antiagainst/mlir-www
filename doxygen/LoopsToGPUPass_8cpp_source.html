49 // A pass that traverses top-level loops in the function and converts them to
50 // GPU launch operations. Nested launches are not allowed, so this does not
51 // walk the function recursively to avoid considering nested loops.
52 struct ForLoopMapper : public FunctionPass<ForLoopMapper> {
53  ForLoopMapper(unsigned numBlockDims, unsigned numThreadDims)
54  : numBlockDims(numBlockDims), numThreadDims(numThreadDims) {}
55 
56  void runOnFunction() override {
57  for (Block &block : getFunction())
58  for (Operation &op : llvm::make_early_inc_range(block)) {
59  if (auto forOp = dyn_cast<AffineForOp>(&op)) {
60  if (failed(convertAffineLoopNestToGPULaunch(forOp, numBlockDims,
61  numThreadDims)))
62  signalPassFailure();
63  } else if (auto forOp = dyn_cast<ForOp>(&op)) {
64  if (failed(convertLoopNestToGPULaunch(forOp, numBlockDims,
65  numThreadDims)))
66  signalPassFailure();
67  }
68  }
69  }
70 
71  unsigned numBlockDims;
72  unsigned numThreadDims;
73 };

75 // A pass that traverses top-level loops in the function and convertes them to
76 // GPU launch operations. The top-level loops itself does not have to be
77 // perfectly nested. The only requirement is that there be as many perfectly
78 // nested loops as the size of `numWorkGroups`. Within these any loop nest has
79 // to be perfectly nested upto depth equal to size of `workGroupSize`.
80 struct ImperfectlyNestedForLoopMapper
81  : public FunctionPass<ImperfectlyNestedForLoopMapper> {
82  ImperfectlyNestedForLoopMapper(ArrayRef<int64_t> numWorkGroups,
83  ArrayRef<int64_t> workGroupSize)
84  : numWorkGroups(numWorkGroups.begin(), numWorkGroups.end()),
85  workGroupSize(workGroupSize.begin(), workGroupSize.end()) {}
86 
87  void runOnFunction() override {
88  // Insert the num work groups and workgroup sizes as constant values. This
89  // pass is only used for testing.
90  FuncOp funcOp = getFunction();
91  OpBuilder builder(funcOp.getOperation()->getRegion(0));
92  SmallVector<Value, 3> numWorkGroupsVal, workGroupSizeVal;
93  for (auto val : numWorkGroups) {
94  auto constOp = builder.create<ConstantOp>(
95  funcOp.getLoc(), builder.getIntegerAttr(builder.getIndexType(), val));
96  numWorkGroupsVal.push_back(constOp);
97  }
98  for (auto val : workGroupSize) {
99  auto constOp = builder.create<ConstantOp>(
100  funcOp.getLoc(), builder.getIntegerAttr(builder.getIndexType(), val));
101  workGroupSizeVal.push_back(constOp);
102  }
103  for (Block &block : getFunction()) {
104  for (Operation &op : llvm::make_early_inc_range(block)) {
105  if (auto forOp = dyn_cast<ForOp>(&op)) {
106  if (failed(convertLoopToGPULaunch(forOp, numWorkGroupsVal,
107  workGroupSizeVal))) {
108  return signalPassFailure();
109  }
110  }
111  }
112  }
113  }
114  SmallVector<int64_t, 3> numWorkGroups;
115  SmallVector<int64_t, 3> workGroupSize;
116 };
