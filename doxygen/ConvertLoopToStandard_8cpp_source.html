37 // Create a CFG subgraph for the loop around its body blocks (if the body
38 // contained other loops, they have been already lowered to a flow of blocks).
39 // Maintain the invariants that a CFG subgraph created for any loop has a single
40 // entry and a single exit, and that the entry/exit blocks are respectively
41 // first/last blocks in the parent region. The original loop operation is
42 // replaced by the initialization operations that set up the initial value of
43 // the loop induction variable (%iv) and computes the loop bounds that are loop-
44 // invariant for affine loops. The operations following the original loop.for
45 // are split out into a separate continuation (exit) block. A condition block is
46 // created before the continuation block. It checks the exit condition of the
47 // loop and branches either to the continuation block, or to the first block of
48 // the body. Induction variable modification is appended to the last block of
49 // the body (which is the exit block from the body subgraph thanks to the
50 // invariant we maintain) along with a branch that loops back to the condition
51 // block.
52 //
53 // +---------------------------------+
54 // | <code before the ForOp> |
55 // | <compute initial %iv value> |
56 // | br cond(%iv) |
57 // +---------------------------------+
58 // |
59 // -------| |
60 // | v v
61 // | +--------------------------------+
62 // | | cond(%iv): |
63 // | | <compare %iv to upper bound> |
64 // | | cond_br %r, body, end |
65 // | +--------------------------------+
66 // | | |
67 // | | -------------|
68 // | v |
69 // | +--------------------------------+ |
70 // | | body-first: | |
71 // | | <body contents> | |
72 // | +--------------------------------+ |
73 // | | |
74 // | ... |
75 // | | |
76 // | +--------------------------------+ |
77 // | | body-last: | |
78 // | | <body contents> | |
79 // | | %new_iv =<add step to %iv> | |
80 // | | br cond(%new_iv) | |
81 // | +--------------------------------+ |
82 // | | |
83 // |----------- |--------------------
84 // v
85 // +--------------------------------+
86 // | end: |
87 // | <code after the ForOp> |
88 // +--------------------------------+
89 //
90 struct ForLowering : public OpRewritePattern<ForOp> {
91  using OpRewritePattern<ForOp>::OpRewritePattern;
92 
93  PatternMatchResult matchAndRewrite(ForOp forOp,
94  PatternRewriter &rewriter) const override;
95 };

97 // Create a CFG subgraph for the loop.if operation (including its "then" and
98 // optional "else" operation blocks). We maintain the invariants that the
99 // subgraph has a single entry and a single exit point, and that the entry/exit
100 // blocks are respectively the first/last block of the enclosing region. The
101 // operations following the loop.if are split into a continuation (subgraph
102 // exit) block. The condition is lowered to a chain of blocks that implement the
103 // short-circuit scheme. Condition blocks are created by splitting out an empty
104 // block from the block that contains the loop.if operation. They
105 // conditionally branch to either the first block of the "then" region, or to
106 // the first block of the "else" region. If the latter is absent, they branch
107 // to the continuation block instead. The last blocks of "then" and "else"
108 // regions (which are known to be exit blocks thanks to the invariant we
109 // maintain).
110 //
111 // +--------------------------------+
112 // | <code before the IfOp> |
113 // | cond_br %cond, %then, %else |
114 // +--------------------------------+
115 // | |
116 // | --------------|
117 // v |
118 // +--------------------------------+ |
119 // | then: | |
120 // | <then contents> | |
121 // | br continue | |
122 // +--------------------------------+ |
123 // | |
124 // |---------- |-------------
125 // | V
126 // | +--------------------------------+
127 // | | else: |
128 // | | <else contents> |
129 // | | br continue |
130 // | +--------------------------------+
131 // | |
132 // ------| |
133 // v v
134 // +--------------------------------+
135 // | continue: |
136 // | <code after the IfOp> |
137 // +--------------------------------+
138 //
139 struct IfLowering : public OpRewritePattern<IfOp> {
140  using OpRewritePattern<IfOp>::OpRewritePattern;
141 
142  PatternMatchResult matchAndRewrite(IfOp ifOp,
143  PatternRewriter &rewriter) const override;
144 };

154 PatternMatchResult
155 ForLowering::matchAndRewrite(ForOp forOp, PatternRewriter &rewriter) const {
156  Location loc = forOp.getLoc();
157 
158  // Start by splitting the block containing the 'loop.for' into two parts.
159  // The part before will get the init code, the part after will be the end
160  // point.
161  auto *initBlock = rewriter.getInsertionBlock();
162  auto initPosition = rewriter.getInsertionPoint();
163  auto *endBlock = rewriter.splitBlock(initBlock, initPosition);
164 
165  // Use the first block of the loop body as the condition block since it is
166  // the block that has the induction variable as its argument. Split out
167  // all operations from the first block into a new block. Move all body
168  // blocks from the loop body region to the region containing the loop.
169  auto *conditionBlock = &forOp.region().front();
170  auto *firstBodyBlock =
171  rewriter.splitBlock(conditionBlock, conditionBlock->begin());
172  auto *lastBodyBlock = &forOp.region().back();
173  rewriter.inlineRegionBefore(forOp.region(), endBlock);
174  auto iv = conditionBlock->getArgument(0);
175 
176  // Append the induction variable stepping logic to the last body block and
177  // branch back to the condition block. Construct an expression f :
178  // (x -> x+step) and apply this expression to the induction variable.
179  rewriter.eraseOp(lastBodyBlock->getTerminator());
180  rewriter.setInsertionPointToEnd(lastBodyBlock);
181  auto step = forOp.step();
182  auto stepped = rewriter.create<AddIOp>(loc, iv, step).getResult();
183  if (!stepped)
184  return matchFailure();
185  rewriter.create<BranchOp>(loc, conditionBlock, stepped);
186 
187  // Compute loop bounds before branching to the condition.
188  rewriter.setInsertionPointToEnd(initBlock);
189  Value lowerBound = forOp.lowerBound();
190  Value upperBound = forOp.upperBound();
191  if (!lowerBound || !upperBound)
192  return matchFailure();
193  rewriter.create<BranchOp>(loc, conditionBlock, lowerBound);
194 
195  // With the body block done, we can fill in the condition block.
196  rewriter.setInsertionPointToEnd(conditionBlock);
197  auto comparison =
198  rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, iv, upperBound);
199 
200  rewriter.create<CondBranchOp>(loc, comparison, firstBodyBlock,
201  ArrayRef<Value>(), endBlock, ArrayRef<Value>());
202  // Ok, we're done!
203  rewriter.eraseOp(forOp);
204  return matchSuccess();
205 }

207 PatternMatchResult
208 IfLowering::matchAndRewrite(IfOp ifOp, PatternRewriter &rewriter) const {
209  auto loc = ifOp.getLoc();
210 
211  // Start by splitting the block containing the 'loop.if' into two parts.
212  // The part before will contain the condition, the part after will be the
213  // continuation point.
214  auto *condBlock = rewriter.getInsertionBlock();
215  auto opPosition = rewriter.getInsertionPoint();
216  auto *continueBlock = rewriter.splitBlock(condBlock, opPosition);
217 
218  // Move blocks from the "then" region to the region containing 'loop.if',
219  // place it before the continuation block, and branch to it.
220  auto &thenRegion = ifOp.thenRegion();
221  auto *thenBlock = &thenRegion.front();
222  rewriter.eraseOp(thenRegion.back().getTerminator());
223  rewriter.setInsertionPointToEnd(&thenRegion.back());
224  rewriter.create<BranchOp>(loc, continueBlock);
225  rewriter.inlineRegionBefore(thenRegion, continueBlock);
226 
227  // Move blocks from the "else" region (if present) to the region containing
228  // 'loop.if', place it before the continuation block and branch to it. It
229  // will be placed after the "then" regions.
230  auto *elseBlock = continueBlock;
231  auto &elseRegion = ifOp.elseRegion();
232  if (!elseRegion.empty()) {
233  elseBlock = &elseRegion.front();
234  rewriter.eraseOp(elseRegion.back().getTerminator());
235  rewriter.setInsertionPointToEnd(&elseRegion.back());
236  rewriter.create<BranchOp>(loc, continueBlock);
237  rewriter.inlineRegionBefore(elseRegion, continueBlock);
238  }
239 
240  rewriter.setInsertionPointToEnd(condBlock);
241  rewriter.create<CondBranchOp>(loc, ifOp.condition(), thenBlock,
242  /*trueArgs=*/ArrayRef<Value>(), elseBlock,
243  /*falseArgs=*/ArrayRef<Value>());
244 
245  // Ok, we're done!
246  rewriter.eraseOp(ifOp);
247  return matchSuccess();
248 }

250 PatternMatchResult
251 ParallelLowering::matchAndRewrite(ParallelOp parallelOp,
252  PatternRewriter &rewriter) const {
253  Location loc = parallelOp.getLoc();
254  BlockAndValueMapping mapping;
255 
256  if (parallelOp.getNumResults() != 0) {
257  // TODO: Implement lowering of parallelOp with reductions.
258  return matchFailure();
259  }
260 
261  // For a parallel loop, we essentially need to create an n-dimensional loop
262  // nest. We do this by translating to loop.for ops and have those lowered in
263  // a further rewrite.
264  for (auto loop_operands :
265  llvm::zip(parallelOp.getInductionVars(), parallelOp.lowerBound(),
266  parallelOp.upperBound(), parallelOp.step())) {
267  Value iv, lower, upper, step;
268  std::tie(iv, lower, upper, step) = loop_operands;
269  ForOp forOp = rewriter.create<ForOp>(loc, lower, upper, step);
270  mapping.map(iv, forOp.getInductionVar());
271  rewriter.setInsertionPointToStart(forOp.getBody());
272  }
273 
274  // Now copy over the contents of the body.
275  for (auto &op : parallelOp.body().front().without_terminator())
276  rewriter.clone(op, mapping);
277 
278  rewriter.eraseOp(parallelOp);
279 
280  return matchSuccess();
281 }
