55 LogicalResult mlir::linalg::tileAndFuseLinalgOpAndSetMarker(
56  PatternRewriter &rewriter, Operation *op, ArrayRef<int64_t> sizes,
57  ArrayRef<int64_t> operandIndicesToFuse, StringRef linalgMarker) {
58  auto tileRes = tileLinalgOperation(rewriter, op, sizes);
59  if (!tileRes)
60  return failure();
61  tileRes->op.setAttr(LinalgTransforms::kLinalgTransformMarker,
62  rewriter.getStringAttr(linalgMarker));
63  Aliases aliases;
64  auto G = LinalgDependenceGraph::buildDependenceGraph(
65  aliases, op->getParentOfType<FuncOp>());
66  SmallVector<Operation *, 4> originalProducers;
67  for (auto operandIdx : operandIndicesToFuse) {
68  auto fusionRes = fuseProducerOf(rewriter, tileRes->op, operandIdx, G);
69  if (!fusionRes) {
70  // Linalg fusion requires tiled loops to even determine whether it is
71  // possible to fuse. As a consequence, the pattern may fail even though a
72  // tiled version of op has already been introduced.
73  // So we need to remove the tiled version ourselves in case of failure.
74  // Another possibility is to ensure the constraints on the pattern
75  // guarantee that fusion will occur and just assert here. As we develop
76  // more complex patterns we can choose what is best.
77  rewriter.eraseOp(tileRes->loops[0]);
78  return failure();
79  }
80  fusionRes->fusedProducer.setAttr(LinalgTransforms::kLinalgTransformMarker,
81  rewriter.getStringAttr(linalgMarker));
82  originalProducers.push_back(fusionRes->originalProducer);
83  }
84 
85  // The originalProducers can now be safely erased. This is similar to
86  // SSA-value use-def but in the world of buffer + structured ops.
87  for (auto *originalProducer : originalProducers)
88  rewriter.eraseOp(originalProducer);
89  return success();
90 }

92 bool mlir::linalg::detail::isProducedByOpOfTypeImpl(
93  Operation *consumerOp, Value consumedView,
94  function_ref<bool(Operation *)> isaOpType) {
95  LinalgOp consumer = dyn_cast<LinalgOp>(consumerOp);
96  assert(consumer.hasBufferSemantics() &&
97  "expected linalg op with buffer semantics");
98  if (!consumer)
99  return false;
100 
101  auto maybeConsumerIndex = consumer.getIndexOfInput(consumedView);
102  if (!maybeConsumerIndex)
103  return false;
104 
105  Aliases aliases;
106  auto G = LinalgDependenceGraph::buildDependenceGraph(
107  aliases, consumer.getParentOfType<FuncOp>());
108  for (auto dependence : G.getDependencesInto(
109  consumer, LinalgDependenceGraph::DependenceType::RAW)) {
110  auto producer = cast<LinalgOp>(dependence.dependentOpView.op);
111  if (!isProducerLastWriteOfView(G, consumer, consumedView, producer))
112  continue;
113  if (isaOpType(dependence.dependentOpView.op))
114  return true;
115  }
116  return false;
117 }

122 static bool hasMultiplyAddBody(linalg::GenericOp op) {
123  auto &r = op.region();
124  if (r.empty())
125  return false;
126  if (r.getBlocks().size() != 1)
127  return false;
128  auto &ops = r.front().getOperations();
129  if (ops.size() != 3)
130  return false;
131 
132  using mlir::matchers::m_Val;
133  auto a = m_Val(r.front().getArgument(0));
134  auto b = m_Val(r.front().getArgument(1));
135  auto c = m_Val(r.front().getArgument(2));
136  // TODO(ntv) Update this detection once we have matcher support for
137  // specifying that any permutation of operands matches.
138  auto pattern1 = m_Op<YieldOp>(m_Op<AddFOp>(m_Op<MulFOp>(a, b), c));
139  auto pattern2 = m_Op<YieldOp>(m_Op<AddFOp>(c, m_Op<MulFOp>(a, b)));
140  auto pattern3 = m_Op<YieldOp>(m_Op<AddFOp>(m_Op<MulFOp>(b, a), c));
141  auto pattern4 = m_Op<YieldOp>(m_Op<AddFOp>(c, m_Op<MulFOp>(b, a)));
142  return pattern1.match(&ops.back()) || pattern2.match(&ops.back()) ||
143  pattern3.match(&ops.back()) || pattern4.match(&ops.back());
144 }

148 static bool isMatmul(linalg::GenericOp genericOp) {
149  auto *ctx = genericOp.getContext();
150  auto m = getAffineDimExpr(0, ctx);
151  auto n = getAffineDimExpr(1, ctx);
152  auto k = getAffineDimExpr(2, ctx);
153  auto mapA = AffineMapAttr::get(AffineMap::get(3, 0, {m, k}));
154  auto mapB = AffineMapAttr::get(AffineMap::get(3, 0, {k, n}));
155  auto mapC = AffineMapAttr::get(AffineMap::get(3, 0, {m, n}));
156  auto maps = ArrayAttr::get({mapA, mapB, mapC}, ctx);
157  return genericOp.getNumInputs() == 2 && genericOp.getNumOutputs() == 1 &&
158  genericOp.indexing_maps() == maps && hasMultiplyAddBody(genericOp);
159 }

161 // TODO(ntv, ataei): This is in fact much more general than just vectorization
162 // for matmul and fill ops.
163 LogicalResult mlir::linalg::vectorizeLinalgOpPrecondition(Operation *op) {
164  auto linalgOp = cast<linalg::LinalgOp>(op);
165  // All types must be static shape to go to vector.
166  for (Value operand : linalgOp.getInputsAndOutputBuffers())
167  if (!operand.getType().cast<ShapedType>().hasStaticShape())
168  return failure();
169  for (Type outputTensorType : linalgOp.getOutputTensorTypes())
170  if (!outputTensorType.cast<ShapedType>().hasStaticShape())
171  return failure();
172  if (isa<linalg::MatmulOp>(op) || isa<linalg::FillOp>(op))
173  return success();
174 
175  auto genericOp = dyn_cast<linalg::GenericOp>(op);
176  if (!genericOp || !isMatmul(genericOp))
177  return failure();
178 
179  // TODO(ntv): non-identity layout.
180  auto isStaticMemRefWithIdentityLayout = [](Value v) {
181  auto m = v.getType().dyn_cast<MemRefType>();
182  if (!m || !m.hasStaticShape() || !m.getAffineMaps().empty())
183  return false;
184  return true;
185  };
186  if (!llvm::all_of(genericOp.getInputsAndOutputBuffers(),
187  isStaticMemRefWithIdentityLayout))
188  return failure();
189  return success();
190 }

192 SmallVector<Value, 0> mlir::linalg::vectorizeLinalgOp(PatternRewriter &rewriter,
193  Operation *op) {
194  using edsc::intrinsics::std_load;
195  using edsc::intrinsics::std_store;
196  using vector_contract = edsc::intrinsics::ValueBuilder<vector::ContractionOp>;
197  using vector_broadcast = edsc::intrinsics::ValueBuilder<vector::BroadcastOp>;
198  using vector_type_cast = edsc::intrinsics::ValueBuilder<vector::TypeCastOp>;
199 
200  assert(succeeded(vectorizeLinalgOpPrecondition(op)) &&
201  "DRR failure case must be a precondition");
202  auto linalgOp = cast<linalg::LinalgOp>(op);
203  assert(linalgOp.hasBufferSemantics() &&
204  "expected linalg op with buffer semantics");
205  edsc::ScopedContext scope(rewriter, op->getLoc());
206 
207  if (auto fillOp = dyn_cast<linalg::FillOp>(op)) {
208  // Vectorize fill as a vector.broadcast.
209  LLVM_DEBUG(dbgs() << "\n[" DEBUG_TYPE
210  "]: Rewrite linalg.fill as vector.broadcast: "
211  << *op << ":\n");
212  auto dstMemrefVec = vector_type_cast(fillOp.getOutputBuffer(0));
213  auto dstVec = std_load(dstMemrefVec);
214  auto resVec = vector_broadcast(dstVec, fillOp.value());
215  std_store(resVec, dstMemrefVec);
216  } else {
217  // Vectorize other ops as vector contraction (currently only matmul).
218  LLVM_DEBUG(dbgs() << "\n[" DEBUG_TYPE
219  "]: Rewrite linalg op as vector.contract: "
220  << *op << ":\n");
221  auto vA = std_load(vector_type_cast(linalgOp.getInput(0)));
222  auto vB = std_load(vector_type_cast(linalgOp.getInput(1)));
223  auto vectorMemRefC = vector_type_cast(linalgOp.getOutputBuffer(0));
224  auto vC = std_load(vectorMemRefC);
225  auto vRes = vector_contract(vA, vB, vC, linalgOp.indexing_maps(),
226  linalgOp.iterator_types());
227  std_store(vRes, vectorMemRefC);
228  }
229  return {};
230 }

236  Operation *op, ArrayRef<unsigned> permutation) {

241  return failure();

245  return failure();
246  return success();

249 SmallVector<Value, 0>
250 mlir::linalg::permuteGenericLinalgOp(PatternRewriter &rewriter, Operation *op,
251  ArrayRef<unsigned> permutation,
252  StringRef linalgMarker) {
253  LLVM_DEBUG(dbgs() << "\n[" DEBUG_TYPE "]: Permute dims for linalg op: " << *op
254  << ":\n");
255 
256  assert(succeeded(permuteGenericLinalgOpPrecondition(op, permutation)) &&
257  "DRR failure case must be a precondition");
258 
259  auto linOp = cast<LinalgOp>(op);
260  auto permutationMap = inversePermutation(
261  AffineMap::getPermutationMap(permutation, rewriter.getContext()));
262  SmallVector<AffineMap, 4> newIndexingMap;
263  auto indexingMaps = linOp.indexing_maps().getValue();
264  for (unsigned i = 0, e = linOp.getNumInputsAndOutputs(); i != e; ++i) {
265  AffineMap m = indexingMaps[i].cast<AffineMapAttr>().getValue().compose(
266  permutationMap);
267  newIndexingMap.push_back(m);
268  }
269  auto itTypes = linOp.iterator_types().getValue();
270  SmallVector<Attribute, 4> itTypesVector;
271  for (unsigned i = 0, e = itTypes.size(); i != e; ++i)
272  itTypesVector.push_back(itTypes[i]);
273  applyPermutationToVector(itTypesVector, permutation);
274  op->setAttr(getIndexingMapsAttrName(),
275  rewriter.getAffineMapArrayAttr(newIndexingMap));
276  op->setAttr(getIteratorTypesAttrName(), rewriter.getArrayAttr(itTypesVector));
277  op->setAttr(LinalgTransforms::kLinalgTransformMarker,
278  rewriter.getStringAttr(linalgMarker));
279  linOp.clone(rewriter, linOp.getLoc(), op->getOperands());
280  return {};
281 }

286 LogicalResult mlir::linalg::promoteSubviewsLinalgOpPrecondition(Operation *op) {
287  LinalgOp linOp = dyn_cast<LinalgOp>(op);
288  // Transformation applies to buffers only.
289  if (!linOp || !linOp.hasBufferSemantics())
290  return failure();
291  if (llvm::none_of(linOp.getInputsAndOutputBuffers(), [](Value v) {
292  return isa_and_nonnull<SubViewOp>(v.getDefiningOp());
293  }))
294  return failure();
295  return success();
296 }

298 SmallVector<Value, 0>
299 mlir::linalg::promoteSubviewsLinalgOp(PatternRewriter &rewriter,
300  Operation *op) {
301  LLVM_DEBUG(dbgs() << "\n[" DEBUG_TYPE "]: Promote subviews for linalg op: "
302  << *op << ":\n");
303 
304  assert(succeeded(promoteSubviewsLinalgOpPrecondition(op)) &&
305  "DRR failure case must be a precondition");
306 
307  LinalgOp linOp = cast<LinalgOp>(op);
308  assert(linOp.hasBufferSemantics() &&
309  "expected linalg op with buffer semantics");
310  SetVector<Value> subViews;
311  for (auto it : linOp.getInputsAndOutputBuffers())
312  if (auto sv = dyn_cast_or_null<SubViewOp>(it.getDefiningOp()))
313  subViews.insert(sv);
314  if (!subViews.empty()) {
315  promoteSubViewOperands(rewriter, linOp, subViews);
316  return {};
317  }
318  llvm_unreachable("DRR failure case must be a precondition");
319 }
