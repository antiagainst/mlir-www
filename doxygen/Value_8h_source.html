33 /// This class represents an instance of an SSA value in the MLIR system,
34 /// representing a computable value that has a type and a set of users. An SSA
35 /// value is either a BlockArgument or the result of an operation. Note: This
36 /// class has value-type semantics and is just a simple wrapper around a
37 /// ValueImpl that is either owner by a block(in the case of a BlockArgument) or
38 /// an Operation(in the case of an OpResult).
39 class Value {
40 public:
41  /// The enumeration represents the various different kinds of values the
42  /// internal representation may take. We steal 2 bits to support a total of 4
43  /// possible values.
44  enum class Kind {
45  /// The first N kinds are all inline operation results. An inline operation
46  /// result means that the kind represents the result number, and the owner
47  /// pointer is the owning `Operation*`. Note: These are packed first to make
48  /// result number lookups more efficient.
49  OpResult0 = 0,
50  OpResult1 = 1,
51 
52  /// The next kind represents a 'trailing' operation result. This is for
53  /// results with numbers larger than we can represent inline. The owner here
54  /// is an `TrailingOpResult*` that points to a trailing storage on the
55  /// parent operation.
56  TrailingOpResult = 2,
57 
58  /// The last kind represents a block argument. The owner here is a
59  /// `BlockArgumentImpl*`.
60  BlockArgument = 3
61  };
62 
63  /// This value represents the 'owner' of the value and its kind. See the
64  /// 'Kind' enumeration above for a more detailed description of each kind of
65  /// owner.
66  struct ImplTypeTraits : public llvm::PointerLikeTypeTraits<void *> {
67  // We know that all pointers within the ImplType are aligned by 8-bytes,
68  // meaning that we can steal up to 3 bits for the different values.
69  static constexpr int NumLowBitsAvailable = 3;
70  };
71  using ImplType = llvm::PointerIntPair<void *, 2, Kind, ImplTypeTraits>;
72 
73 public:
74  constexpr Value(std::nullptr_t) : ownerAndKind() {}
75  Value(ImplType ownerAndKind = {}) : ownerAndKind(ownerAndKind) {}
76  Value(const Value &) = default;
77  Value &operator=(const Value &) = default;
78 
79  template <typename U> bool isa() const {
80  assert(*this && "isa<> used on a null type.");
81  return U::classof(*this);
82  }
83  template <typename U> U dyn_cast() const {
84  return isa<U>() ? U(ownerAndKind) : U(nullptr);
85  }
86  template <typename U> U dyn_cast_or_null() const {
87  return (*this && isa<U>()) ? U(ownerAndKind) : U(nullptr);
88  }
89  template <typename U> U cast() const {
90  assert(isa<U>());
91  return U(ownerAndKind);
92  }
93 
94  operator bool() const { return ownerAndKind.getPointer(); }
95  bool operator==(const Value &other) const {
96  return ownerAndKind == other.ownerAndKind;
97  }
98  bool operator!=(const Value &other) const { return !(*this == other); }

104  MLIRContext *getContext() const { return getType().getContext(); }

107  ///

233 /// Block arguments are values.
234 class BlockArgument : public Value {

238  static bool classof(Value value) {
239  return value.getKind() == Kind::BlockArgument;
240  }
241 
242  /// Returns the block that owns this argument.
243  Block *getOwner() const { return getImpl()->owner; }
244 
245  /// Return the type of this value.
246  Type getType() const { return getImpl()->type; }
247 
248  /// Set the type of this value.
249  void setType(Type newType) { getImpl()->type = newType; }
250 
251  /// Returns the number of this argument.
252  unsigned getArgNumber() const;
253 
254 private:
255  /// Allocate a new argument with the given type and owner.
256  static BlockArgument create(Type type, Block *owner) {
257  return new detail::BlockArgumentImpl(type, owner);
258  }
259 
260  /// Destroy and deallocate this argument.
261  void destroy() { delete getImpl(); }
262 
263  /// Get a raw pointer to the internal implementation.
264  detail::BlockArgumentImpl *getImpl() const {
265  return reinterpret_cast<detail::BlockArgumentImpl *>(
266  ownerAndKind.getPointer());
267  }
268 
269  /// Allow access to `create` and `destroy`.
270  friend Block;
271 
272  /// Allow access to 'getImpl'.
273  friend Value;
