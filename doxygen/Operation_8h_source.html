516  /// Walk the operation in postorder, calling the callback for each nested
517  /// operation(including this one). The callback method can take any of the
518  /// following forms:
519  /// void(Operation*) : Walk all operations opaquely.
520  /// * op->walk([](Operation *nestedOp) { ...});
521  /// void(OpT) : Walk all operations of the given derived type.
522  /// * op->walk([](ReturnOp returnOp) { ...});
523  /// WalkResult(Operation*|OpT) : Walk operations, but allow for
524  /// interruption/cancellation.
525  /// * op->walk([](... op) {
526  /// // Interrupt, i.e cancel, the walk based on some invariant.
527  /// if (some_invariant)
528  /// return WalkResult::interrupt();
529  /// return WalkResult::advance();
530  /// });
531  template <typename FnT, typename RetT = detail::walkResultType<FnT>>
532  RetT walk(FnT &&callback) {
533  return detail::walkOperations(this, std::forward<FnT>(callback));
534  }
