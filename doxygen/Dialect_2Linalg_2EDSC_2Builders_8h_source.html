28 /// A LoopRangeBuilder is a generic NestedBuilder for loop.for operations.
29 /// More specifically it is meant to be used as a temporary object for
30 /// representing any nested MLIR construct that is "related to" an mlir::Value
31 /// (for now an induction variable).
32 class LoopRangeBuilder : public NestedBuilder {
33 public:
34  /// Constructs a new loop.for and captures the associated induction
35  /// variable. A ValueHandle pointer is passed as the first argument and is the
36  /// *only* way to capture the loop induction variable.
37  LoopRangeBuilder(ValueHandle *iv, ValueHandle range);
38  LoopRangeBuilder(ValueHandle *iv, Value range);
39  LoopRangeBuilder(ValueHandle *iv, SubViewOp::Range range);
40 
41  LoopRangeBuilder(const LoopRangeBuilder &) = delete;
42  LoopRangeBuilder(LoopRangeBuilder &&) = default;
43 
44  LoopRangeBuilder &operator=(const LoopRangeBuilder &) = delete;
45  LoopRangeBuilder &operator=(LoopRangeBuilder &&) = default;
46 
47  /// The only purpose of this operator is to serve as a sequence point so that
48  /// the evaluation of `fun` (which build IR snippets in a scoped fashion) is
49  /// scoped within a LoopRangeBuilder.
50  ValueHandle operator()(std::function<void(void)> fun = nullptr);
51 };

97 /// A StructuredIndexed represents an indexable quantity that is either:
98 /// 1. a captured value, which is suitable for buffer and tensor operands, or;
99 /// 2. a captured type, which is suitable for tensor return values.
100 ///
101 /// A StructuredIndexed itself is indexed and passed to `makeGenericLinalgOp`.
102 /// It enable an idiomatic syntax for index expressions such as:
103 ///
104 /// ```
105 /// StructuredIndexed A(buffer_or_tensor_value), B(buffer_or_tensor_value),
106 /// C(buffer_value_or_tensor_type);
107 /// makeGenericLinalgOp({A({m, n}), B({k, n})}, {C({m, n})}, ... );
108 /// ```
109 struct StructuredIndexed : public ValueHandle {
110  StructuredIndexed(Type type) : ValueHandle(type) {}
111  StructuredIndexed(Value value) : ValueHandle(value) {}
112  StructuredIndexed(ValueHandle valueHandle) : ValueHandle(valueHandle) {}
113  StructuredIndexed operator()(ArrayRef<AffineExpr> indexings) {
114  return StructuredIndexed(*this, indexings);
115  }
116 
117  ArrayRef<AffineExpr> getExprs() { return exprs; }
118 
119 private:
120  StructuredIndexed(Type t, ArrayRef<AffineExpr> indexings)
121  : ValueHandle(t), exprs(indexings.begin(), indexings.end()) {
122  assert(t.isa<RankedTensorType>() && "RankedTensor expected");
123  }
124  StructuredIndexed(Value v, ArrayRef<AffineExpr> indexings)
125  : ValueHandle(v), exprs(indexings.begin(), indexings.end()) {
126  assert((v.getType().isa<MemRefType>() ||
127  v.getType().isa<RankedTensorType>()) &&
128  "MemRef or RankedTensor expected");
129  }
130  StructuredIndexed(ValueHandle vh, ArrayRef<AffineExpr> indexings)
131  : ValueHandle(vh), exprs(indexings.begin(), indexings.end()) {}
132 
133  SmallVector<AffineExpr, 4> exprs;
134 };

138 /// Build a `linalg.generic` op with the specified `inputs`, `outputs` and
139 /// `region`.
140 ///
141 /// `otherValues` and `otherAttributes` may be passed and will be appended as
142 /// operands and attributes respectively.
143 ///
144 /// Prerequisites:
145 /// =============
146 ///
147 /// 1. `inputs` may contain StructuredIndexed that capture either buffer or
148 /// tensor values.
149 /// 2. `outputs` may contain StructuredIndexed that capture either buffer values
150 /// or tensor types. If both buffer values and tensor types are present, then
151 /// all buffer values must appear before any tensor type. Without this
152 /// restriction output tensor results would need to be reordered, which would
153 /// result in surprising behavior when combined with region definition.
154 Operation *makeGenericLinalgOp(
155  ArrayRef<IterType> iteratorTypes, ArrayRef<StructuredIndexed> inputs,
156  ArrayRef<StructuredIndexed> outputs,
157  function_ref<void(ArrayRef<BlockArgument>)> regionBuilder =
158  defaultRegionBuilder,
159  ArrayRef<Value> otherValues = {}, ArrayRef<Attribute> otherAttributes = {});

178 /// Build a linalg.pointwise, under the current ScopedContext, at the current
179 /// insert point, that computes:
180 /// ```
181 /// (i0, ..., in) = (par, ..., par)
182 /// |
183 /// | O...(some_subset...(i0, ..., in)) =
184 /// | some_pointwise_func...(I...(some_other_subset...(i0, ..., in)))
185 /// ```
186 ///
187 /// This is a very generic entry point that can be configured in many ways to
188 /// build a perfect loop nest of parallel loops with arbitrarily complex
189 /// innermost loop code and whatever (explicit) broadcast semantics.
190 ///
191 /// This can be used with both out-of-place and in-place semantics.
192 /// The client is responsible for ensuring the region operations are compatible
193 /// with in-place semantics and parallelism.
194
